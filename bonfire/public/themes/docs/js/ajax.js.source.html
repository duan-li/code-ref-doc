  84    Ajax.prototype.click = function (e) {
  85      var $this = $(this)
  86        , url = $this.attr('href')
  87        , method = $this.attr('data-method')
  88  
  89      if (!method) {
  90        method = 'get'
  91      }
  92  
  93      spin($this)
  94  
  95      e.preventDefault()
  96  
  97      $.ajax({
  98        url: url,
  99        type: method,
 100        dataType: 'json',
 101        statusCode: {
 102          200: function(data) {
 103            processData(data, $this)
 104          },
 105          500: function() {
 106            $this.trigger('bootstrap-ajax:error', [$this, 500])
 107          },
 108          404: function() {
 109            $this.trigger('bootstrap-ajax:error', [$this, 404])
 110          }
 111        }
 112      })

 115    Ajax.prototype.submit = function (e) {
 116      var $this = $(this)
 117        , url = $this.attr('action')
 118        , method = $this.attr('method')
 119        , data = new FormData($this[0])
 120        , button = $this.find('input[type=submit],button[type=submit]')
 121  
 122      button.attr('disabled', 'disabled')
 123  
 124      spin($this)
 125  
 126      e.preventDefault()
 127  
 128      $.ajax({
 129        url: url,
 130        type: method,
 131        data: data,
 132        dataType: 'json',
 133        cache: false,
 134        contentType: false,
 135        processData: false,
 136        statusCode: {
 137          200: function(data) {
 138              $this.find('input[type=text],textarea').val('')
 139              processData(data, $this)
 140          },
 141          500: function() {
 142            $this.trigger('bootstrap-ajax:error', [$this, 500])
 143          },
 144          404: function() {
 145            $this.trigger('bootstrap-ajax:error', [$this, 404])
 146          }
 147        },
 148        complete: function() {
 149          button.removeAttr('disabled')
 150        }
 151      })

 181    function spin($el) { // http://fgnass.github.com/spin.js/
 182      if ($.fn.spin !== undefined) {
 183        var spinner_selector = $el.attr('data-spinner')
 184          , spinner_selector_replace = $el.attr('data-spinner-replace')
 185          , spinner_selector_replace_closest = $el.attr('data-spinner-replace-closest')
 186          , opts = {lines: 11, length:0, width:6, radius:9, rotate:0, trail:89, speed:1.4}
 187  
 188          if (spinner_selector) {
 189            $(spinner_selector).spin(opts)
 190          } else if (spinner_selector_replace) {
 191            $(spinner_selector_replace).html("").spin(opts)
 192          } else if (spinner_selector_replace_closest) {
 193            $el.closest(spinner_selector_replace_closest).html("").spin(opts)
 194          } else {
 195            $el.html("").spin(opts)
 196          }
 197      }

 206    function processData(data, $el) {
 207      if (data.location) {
 208        window.location.href = data.location
 209      } else {
 210        var replace_selector = $el.attr('data-replace')
 211          , replace_closest_selector = $el.attr('data-replace-closest')
 212          , replace_inner_selector = $el.attr('data-replace-inner')
 213          , replace_closest_inner_selector = $el.attr('data-replace-closest-inner')
 214          , append_selector = $el.attr('data-append')
 215          , prepend_selector = $el.attr('data-prepend')
 216          , refresh_selector = $el.attr('data-refresh')
 217          , refresh_closest_selector = $el.attr('data-refresh-closest')
 218          , clear_selector = $el.attr('data-clear')
 219          , remove_selector = $el.attr('data-remove')
 220          , clear_closest_selector = $el.attr('data-clear-closest')
 221          , remove_closest_selector = $el.attr('data-remove-closest')
 222  
 223        if (replace_selector) {
 224          $(replace_selector).replaceWith(data.html)
 225        }
 226        if (replace_closest_selector) {
 227          $el.closest(replace_closest_selector).replaceWith(data.html)
 228        }
 229        if (replace_inner_selector) {
 230          $(replace_inner_selector).html(data.html)
 231        }
 232        if (replace_closest_inner_selector) {
 233          $el.closest(replace_closest_inner_selector).html(data.html)
 234        }
 235        if (append_selector) {
 236          $(append_selector).append(data.html)
 237        }
 238        if (prepend_selector) {
 239          $(prepend_selector).prepend(data.html)
 240        }
 241        if (refresh_selector) {
 242          $.each($(refresh_selector), function(index, value) {
 243            $.getJSON($(value).data('refresh-url'), function(data) {
 244              $(value).replaceWith(data.html)
 245            })
 246          })
 247        }
 248        if (refresh_closest_selector) {
 249          $.each($(refresh_closest_selector), function(index, value) {
 250            $.getJSON($(value).data('refresh-url'), function(data) {
 251              $el.closest($(value)).replaceWith(data.html)
 252            })
 253          })
 254        }
 255        if (clear_selector) {
 256          $(clear_selector).html('')
 257        }
 258        if (remove_selector) {
 259          $(remove_selector).remove()
 260        }
 261        if (clear_closest_selector) {
 262          $el.closest(clear_closest_selector).html('')
 263        }
 264        if (remove_closest_selector) {
 265          $el.closest(remove_closest_selector).remove()
 266        }
 267      }
 268  
 269      if (data.fragments) {
 270        for (var s in data.fragments) {
 271          $(s).replaceWith(data.fragments[s])
 272        }
 273      }
 274      if (data['inner-fragments']) {
 275        for (var i in data['inner-fragments']) {
 276          $(i).html(data['inner-fragments'][i])
 277        }
 278      }
 279      if (data['append-fragments']) {
 280        for (var a in data['append-fragments']) {
 281          $(a).append(data['append-fragments'][a])
 282        }
 283      }
 284      if (data['prepend-fragments']) {
 285        for (var p in data['prepend-fragments']) {
 286          $(p).prepend(data['prepend-fragments'][p])
 287        }
 288      }
 289      $el.trigger('bootstrap-ajax:success', [data, $el]);
 290    }
