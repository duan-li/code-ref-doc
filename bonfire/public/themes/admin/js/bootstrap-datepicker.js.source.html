 275      $.fn.datepicker = function ( option ) {
 276          return this.each(function () {
 277              var $this = $(this),
 278                  data = $this.data('datepicker'),
 279                  options = typeof option == 'object' && option;
 280              if (!data) {
 281                  $this.data('datepicker', (data = new Datepicker(this, $.extend({}, $.fn.datepicker.defaults,options))));
 282              }
 283              if (typeof option == 'string') data[option]();

 329          parseDate: function(date, format) {
 330              var parts = date.split(format.separator),
 331                  date = new Date(1970, 1, 1, 0, 0, 0),
 332                  val;
 333              if (parts.length == format.parts.length) {
 334                  for (var i=0, cnt = format.parts.length; i < cnt; i++) {
 335                      val = parseInt(parts[i], 10)||1;
 336                      switch(format.parts[i]) {
 337                          case 'dd':
 338                          case 'd':
 339                              date.setDate(val);
 340                              break;
 341                          case 'mm':
 342                          case 'm':
 343                              date.setMonth(val - 1);
 344                              break;
 345                          case 'yy':
 346                              date.setFullYear(2000 + val);
 347                              break;
 348                          case 'yyyy':
 349                              date.setFullYear(val);
 350                              break;
 351                      }
 352                  }
 353              }
