  43      /**
  44       * The main Lex parser method.  Essentially acts as dispatcher to
  45       * all of the helper parser methods.
  46       *
  47       * @param  string       $text     Text to parse
  48       * @param  array|object $data     Array or object to use
  49       * @param  mixed        $callback Callback to use for Callback Tags
  50       * @return string
  51       */
  52      public function parse($text, $data = array(), $callback = false, $allowPhp = false)
  53      {
  54  
  55          $this->setupRegex();
  56          $this->allowPhp = $allowPhp;
  57  
  58          // Is this the first time parse() is called?
  59          if (self::$data === null) {
  60              // Let's store the local data array for later use.
  61              self::$data = $data;
  62          } else {
  63              // Let's merge the current data array with the local scope variables
  64              // So you can call local variables from within blocks.
  65              $data = array_merge(self::$data, $data);
  66  
  67              // Since this is not the first time parse() is called, it's most definately a callback,
  68              // let's store the current callback data with the the local data
  69              // so we can use it straight after a callback is called.
  70              self::$callbackData = $data;
  71          }
  72  
  73          // The parseConditionals method executes any PHP in the text, so clean it up.
  74          if (! $allowPhp) {
  75              $text = str_replace(array('<?', '?>'), array('&lt;?', '?&gt;'), $text);
  76          }
  77  
  78          $text = $this->parseComments($text);
  79          $text = $this->extractNoparse($text);
  80          $text = $this->extractLoopedTags($text, $data, $callback);
  81  
  82          // Order is important here.  We parse conditionals first as to avoid
  83          // unnecessary code from being parsed and executed.
  84          $text = $this->parseConditionals($text, $data, $callback);
  85          $text = $this->injectExtractions($text, 'looped_tags');
  86          $text = $this->parseVariables($text, $data, $callback);
  87          $text = $this->injectExtractions($text, 'callback_blocks');
  88  
  89          if ($callback) {
  90              $text = $this->parseCallbackTags($text, $data, $callback);
  91          }
  92  
  93          // To ensure that {{ noparse }} is never parsed even during consecutive parse calls
  94          // set $cumulativeNoparse to true and use self::injectNoparse($text); immediately
  95          // before the final output is sent to the browser
  96          if (! $this->cumulativeNoparse) {
  97              $text = $this->injectExtractions($text);
  98          }
  99  
 100          return $text;
 101      }

 116      /**
 117       * Recursivly parses all of the variables in the given text and
 118       * returns the parsed text.
 119       *
 120       * @param  string       $text Text to parse
 121       * @param  array|object $data Array or object to use
 122       * @return string
 123       */
 124      public function parseVariables($text, $data, $callback = null)
 125      {
 126          $this->setupRegex();
 127  
 128          /**
 129           * $data_matches[][0][0] is the raw data loop tag
 130           * $data_matches[][0][1] is the offset of raw data loop tag
 131           * $data_matches[][1][0] is the data variable (dot notated)
 132           * $data_matches[][1][1] is the offset of data variable
 133           * $data_matches[][2][0] is the content to be looped over
 134           * $data_matches[][2][1] is the offset of content to be looped over
 135           */
 136          if (preg_match_all($this->variableLoopRegex, $text, $data_matches, PREG_SET_ORDER + PREG_OFFSET_CAPTURE)) {
 137              foreach ($data_matches as $index => $match) {
 138                  if ($loop_data = $this->getVariable($match[1][0], $data)) {
 139                      $looped_text = '';
 140                      foreach ($loop_data as $item_data) {
 141                          $str = $this->extractLoopedTags($match[2][0], $item_data, $callback);
 142                          $str = $this->parseConditionals($str, $item_data, $callback);
 143                          $str = $this->injectExtractions($str, 'looped_tags');
 144                          $str = $this->parseVariables($str, $item_data, $callback);
 145                          if ($callback !== null) {
 146                              $str = $this->parseCallbackTags($str, $item_data, $callback);
 147                          }
 148  
 149                          $looped_text .= $str;
 150                      }
 151                      $text = preg_replace('/'.preg_quote($match[0][0], '/').'/m', addcslashes($looped_text, '\\$'), $text, 1);
 152                  } else { // It's a callback block.
 153                      // Let's extract it so it doesn't conflict
 154                      // with the local scope variables in the next step.
 155                      $text = $this->createExtraction('callback_blocks', $match[0][0], $match[0][0], $text);
 156                  }
 157              }
 158          }
 159  
 160          /**
 161           * $data_matches[0] is the raw data tag
 162           * $data_matches[1] is the data variable (dot notated)
 163           */
 164          if (preg_match_all($this->variableTagRegex, $text, $data_matches)) {
 165              foreach ($data_matches[1] as $index => $var) {
 166                  if (($val = $this->getVariable($var, $data, '__lex_no_value__')) !== '__lex_no_value__') {
 167                      $text = str_replace($data_matches[0][$index], $val, $text);
 168                  }
 169              }
 170          }
 171  
 172          return $text;
 173      }

 175      /**
 176       * Parses all Callback tags, and sends them through the given $callback.
 177       *
 178       * @param  string $text           Text to parse
 179       * @param  mixed  $callback       Callback to apply to each tag
 180       * @param  bool   $inConditional Whether we are in a conditional tag
 181       * @return string
 182       */
 183      public function parseCallbackTags($text, $data, $callback)
 184      {
 185          $this->setupRegex();
 186          $inCondition = $this->inCondition;
 187  
 188          if ($inCondition) {
 189              $regex = '/\{\s*('.$this->variableRegex.')(\s+.*?)?\s*\}/ms';
 190          } else {
 191              $regex = '/\{\{\s*('.$this->variableRegex.')(\s+.*?)?\s*(\/)?\}\}/ms';
 192          }
 193          /**
 194           * $match[0][0] is the raw tag
 195           * $match[0][1] is the offset of raw tag
 196           * $match[1][0] is the callback name
 197           * $match[1][1] is the offset of callback name
 198           * $match[2][0] is the parameters
 199           * $match[2][1] is the offset of parameters
 200           * $match[3][0] is the self closure
 201           * $match[3][1] is the offset of closure
 202           */
 203          while (preg_match($regex, $text, $match, PREG_OFFSET_CAPTURE)) {
 204              $selfClosed = false;
 205              $parameters = array();
 206              $tag = $match[0][0];
 207              $start = $match[0][1];
 208              $name = $match[1][0];
 209              if (isset($match[2])) {
 210                  $cb_data = $data;
 211                  if ( !empty(self::$callbackData)) {
 212                      $cb_data = array_merge(self::$callbackData, $data);
 213                  }
 214                  $raw_params = $this->injectExtractions($match[2][0], '__cond_str');
 215                  $parameters = $this->parseParameters($raw_params, $cb_data, $callback);
 216              }
 217  
 218              if (isset($match[3])) {
 219                  $selfClosed = true;
 220              }
 221              $content = '';
 222  
 223              $temp_text = substr($text, $start + strlen($tag));
 224              if (preg_match('/\{\{\s*\/'.preg_quote($name, '/').'\s*\}\}/m', $temp_text, $match, PREG_OFFSET_CAPTURE) && ! $selfClosed) {
 225  
 226                  $content = substr($temp_text, 0, $match[0][1]);
 227                  $tag .= $content.$match[0][0];
 228  
 229                  // Is there a nested block under this one existing with the same name?
 230                  $nested_regex = '/\{\{\s*('.preg_quote($name, '/').')(\s.*?)\}\}(.*?)\{\{\s*\/\1\s*\}\}/ms';
 231                  if (preg_match($nested_regex, $content.$match[0][0], $nested_matches)) {
 232                      $nested_content = preg_replace('/\{\{\s*\/'.preg_quote($name, '/').'\s*\}\}/m', '', $nested_matches[0]);
 233                      $content = $this->createExtraction('nested_looped_tags', $nested_content, $nested_content, $content);
 234                  }
 235              }
 236  
 237              $replacement = call_user_func_array($callback, array($name, $parameters, $content));
 238              $replacement = $this->parseRecursives($replacement, $content, $callback);
 239  
 240              if ($inCondition) {
 241                  $replacement = $this->valueToLiteral($replacement);
 242              }
 243              $text = preg_replace('/'.preg_quote($tag, '/').'/m', addcslashes($replacement, '\\$'), $text, 1);
 244              $text = $this->injectExtractions($text, 'nested_looped_tags');
 245          }
 246  
 247          return $text;
 248      }

 250      /**
 251       * Parses all conditionals, then executes the conditionals.
 252       *
 253       * @param  string $text     Text to parse
 254       * @param  mixed  $data     Data to use when executing conditionals
 255       * @param  mixed  $callback The callback to be used for tags
 256       * @return string
 257       */
 258      public function parseConditionals($text, $data, $callback)
 259      {
 260          $this->setupRegex();
 261          preg_match_all($this->conditionalRegex, $text, $matches, PREG_SET_ORDER);
 262  
 263          $this->conditionalData = $data;
 264  
 265          /**
 266           * $matches[][0] = Full Match
 267           * $matches[][1] = Either 'if', 'unless', 'elseif', 'elseunless'
 268           * $matches[][2] = Condition
 269           */
 270          foreach ($matches as $match) {
 271              $this->inCondition = true;
 272  
 273              $condition = $match[2];
 274  
 275              // Extract all literal string in the conditional to make it easier
 276              if (preg_match_all('/(["\']).*?(?<!\\\\)\1/', $condition, $str_matches)) {
 277                  foreach ($str_matches[0] as $m) {
 278                      $condition = $this->createExtraction('__cond_str', $m, $m, $condition);
 279                  }
 280              }
 281              $condition = preg_replace($this->conditionalNotRegex, '$1!$2', $condition);
 282  
 283              if (preg_match_all($this->conditionalExistsRegex, $condition, $existsMatches, PREG_SET_ORDER)) {
 284                  foreach ($existsMatches as $m) {
 285                      $exists = 'true';
 286                      if ($this->getVariable($m[2], $data, '__doesnt_exist__') === '__doesnt_exist__') {
 287                          $exists = 'false';
 288                      }
 289                      $condition = $this->createExtraction('__cond_exists', $m[0], $m[1].$exists.$m[3], $condition);
 290                  }
 291              }
 292  
 293              $condition = preg_replace_callback('/\b('.$this->variableRegex.')\b/', array($this, 'processConditionVar'), $condition);
 294  
 295              if ($callback) {
 296                  $condition = preg_replace('/\b(?!\{\s*)('.$this->callbackNameRegex.')(?!\s+.*?\s*\})\b/', '{$1}', $condition);
 297                  $condition = $this->parseCallbackTags($condition, $data, $callback);
 298              }
 299  
 300              // Re-extract the strings that have now been possibly added.
 301              if (preg_match_all('/(["\']).*?(?<!\\\\)\1/', $condition, $str_matches)) {
 302                  foreach ($str_matches[0] as $m) {
 303                      $condition = $this->createExtraction('__cond_str', $m, $m, $condition);
 304                  }
 305              }
 306  
 307  
 308              // Re-process for variables, we trick processConditionVar so that it will return null
 309              $this->inCondition = false;
 310              $condition = preg_replace_callback('/\b('.$this->variableRegex.')\b/', array($this, 'processConditionVar'), $condition);
 311              $this->inCondition = true;
 312  
 313              // Re-inject any strings we extracted
 314              $condition = $this->injectExtractions($condition, '__cond_str');
 315              $condition = $this->injectExtractions($condition, '__cond_exists');
 316  
 317              $conditional = '<?php ';
 318  
 319              if ($match[1] == 'unless') {
 320                  $conditional .= 'if ( ! ('.$condition.'))';
 321              } elseif ($match[1] == 'elseunless') {
 322                  $conditional .= 'elseif ( ! ('.$condition.'))';
 323              } else {
 324                  $conditional .= $match[1].' ('.$condition.')';
 325              }
 326  
 327              $conditional .= ': ?>';
 328  
 329              $text = preg_replace('/'.preg_quote($match[0], '/').'/m', addcslashes($conditional, '\\$'), $text, 1);
 330          }
 331  
 332          $text = preg_replace($this->conditionalElseRegex, '<?php else: ?>', $text);
 333          $text = preg_replace($this->conditionalEndRegex, '<?php endif; ?>', $text);
 334  
 335          $text = $this->parsePhp($text);
 336          $this->inCondition = false;
 337  
 338          return $text;
 339      }

 341      /**
 342       * Goes recursively through a callback tag with a passed child array.
 343       *
 344       * @param  string $text      - The replaced text after a callback.
 345       * @param  string $orig_text - The original text, before a callback is called.
 346       * @param  mixed  $callback
 347       * @return string $text
 348       */
 349      public function parseRecursives($text, $orig_text, $callback)
 350      {
 351          // Is there a {{ *recursive [array_key]* }} tag here, let's loop through it.
 352          if (preg_match($this->recursiveRegex, $text, $match)) {
 353              $array_key = $match[1];
 354              $tag = $match[0];
 355              $next_tag = null;
 356              $children = self::$callbackData[$array_key];
 357              $child_count = count($children);
 358              $count = 1;
 359  
 360              // Is the array not multi-dimensional? Let's make it multi-dimensional.
 361              if ($child_count == count($children, COUNT_RECURSIVE)) {
 362                  $children = array($children);
 363                  $child_count = 1;
 364              }
 365  
 366              foreach ($children as $child) {
 367                  $has_children = true;
 368  
 369                  // If this is a object let's convert it to an array.
 370                  is_array($child) OR $child = (array) $child;
 371  
 372                  // Does this child not contain any children?
 373                  // Let's set it as empty then to avoid any errors.
 374                  if ( ! array_key_exists($array_key, $child)) {
 375                      $child[$array_key] = array();
 376                      $has_children = false;
 377                  }
 378  
 379                  $replacement = $this->parse($orig_text, $child, $callback, $this->allowPhp);
 380  
 381                  // If this is the first loop we'll use $tag as reference, if not
 382                  // we'll use the previous tag ($next_tag)
 383                  $current_tag = ($next_tag !== null) ? $next_tag : $tag;
 384  
 385                  // If this is the last loop set the next tag to be empty
 386                  // otherwise hash it.
 387                  $next_tag = ($count == $child_count) ? '' : md5($tag.$replacement);
 388  
 389                  $text = str_replace($current_tag, $replacement.$next_tag, $text);
 390  
 391                  if ($has_children) {
 392                      $text = $this->parseRecursives($text, $orig_text, $callback);
 393                  }
 394                  $count++;
 395              }
 396          }
 397  
 398          return $text;
 399      }

 451      /**
 452       * This is used as a callback for the conditional parser.  It takes a variable
 453       * and returns the value of it, properly formatted.
 454       *
 455       * @param  array  $match A match from preg_replace_callback
 456       * @return string
 457       */
 458      protected function processConditionVar($match)
 459      {
 460          $var = is_array($match) ? $match[0] : $match;
 461          if (in_array(strtolower($var), array('true', 'false', 'null', 'or', 'and')) or
 462              strpos($var, '__cond_str') === 0 or
 463              strpos($var, '__cond_exists') === 0 or
 464              is_numeric($var))
 465          {
 466              return $var;
 467          }
 468  
 469          $value = $this->getVariable($var, $this->conditionalData, '__processConditionVar__');
 470  
 471          if ($value === '__processConditionVar__') {
 472              return $this->inCondition ? $var : 'null';
 473          }
 474  
 475          return $this->valueToLiteral($value);
 476      }

 493       * @param  string $value Value to convert
 494       * @return string
 495       */
 496      protected function valueToLiteral($value)
 497      {
 498          if (is_object($value) and is_callable(array($value, '__toString'))) {
 499              return var_export((string) $value, true);

 501              return !empty($value) ? "true" : "false";

 507      /**
 508       * Sets up all the global regex to use the correct Scope Glue.
 509       *
 510       * @return void
 511       */
 512      protected function setupRegex()
 513      {
 514          if ($this->regexSetup) {
 515              return;
 516          }
 517          $glue = preg_quote($this->scopeGlue, '/');
 518  
 519          $this->variableRegex = $glue === '\\.' ? '[a-zA-Z0-9_'.$glue.']+' : '[a-zA-Z0-9_\.'.$glue.']+';
 520          $this->callbackNameRegex = $this->variableRegex.$glue.$this->variableRegex;
 521          $this->variableLoopRegex = '/\{\{\s*('.$this->variableRegex.')\s*\}\}(.*?)\{\{\s*\/\1\s*\}\}/ms';
 522          $this->variableTagRegex = '/\{\{\s*('.$this->variableRegex.')\s*\}\}/m';
 523  
 524          $this->callbackBlockRegex = '/\{\{\s*('.$this->variableRegex.')(\s.*?)\}\}(.*?)\{\{\s*\/\1\s*\}\}/ms';
 525  
 526          $this->recursiveRegex = '/\{\{\s*\*recursive\s*('.$this->variableRegex.')\*\s*\}\}/ms';
 527  
 528          $this->noparseRegex = '/\{\{\s*noparse\s*\}\}(.*?)\{\{\s*\/noparse\s*\}\}/ms';
 529  
 530          $this->conditionalRegex = '/\{\{\s*(if|unless|elseif|elseunless)\s*((?:\()?(.*?)(?:\))?)\s*\}\}/ms';
 531          $this->conditionalElseRegex = '/\{\{\s*else\s*\}\}/ms';
 532          $this->conditionalEndRegex = '/\{\{\s*endif\s*\}\}/ms';
 533          $this->conditionalExistsRegex = '/(\s+|^)exists\s+('.$this->variableRegex.')(\s+|$)/ms';
 534          $this->conditionalNotRegex = '/(\s+|^)not(\s+|$)/ms';
 535  
 536          $this->regexSetup = true;
 537  
 538          // This is important, it's pretty unclear by the documentation
 539          // what the default value is on <= 5.3.6
 540          ini_set('pcre.backtrack_limit', 1000000);
 541      }

 543      /**
 544       * Extracts the noparse text so that it is not parsed.
 545       *
 546       * @param  string $text The text to extract from
 547       * @return string
 548       */
 549      protected function extractNoparse($text)
 550      {
 551          /**
 552           * $matches[][0] is the raw noparse match
 553           * $matches[][1] is the noparse contents
 554           */
 555          if (preg_match_all($this->noparseRegex, $text, $matches, PREG_SET_ORDER)) {
 556              foreach ($matches as $match) {
 557                  $text = $this->createExtraction('noparse', $match[0], $match[1], $text);
 558              }

 561          return $text;

 564      /**
 565       * Extracts the looped tags so that we can parse conditionals then re-inject.
 566       *
 567       * @param  string $text The text to extract from
 568       * @return string
 569       */
 570      protected function extractLoopedTags($text, $data = array(), $callback = null)
 571      {
 572          /**
 573           * $matches[][0] is the raw match
 574           */
 575          if (preg_match_all($this->callbackBlockRegex, $text, $matches, PREG_SET_ORDER)) {
 576              foreach ($matches as $match) {
 577                  // Does this callback block contain parameters?
 578                  if ($this->parseParameters($match[2], $data, $callback)) {
 579                      // Let's extract it so it doesn't conflict with local variables when
 580                      // parseVariables() is called.
 581                      $text = $this->createExtraction('callback_blocks', $match[0], $match[0], $text);
 582                  } else {
 583                      $text = $this->createExtraction('looped_tags', $match[0], $match[0], $text);
 584                  }
 585              }
 586          }
 587  
 588          return $text;
 589      }

 591      /**
 592       * Extracts text out of the given text and replaces it with a hash which
 593       * can be used to inject the extractions replacement later.
 594       *
 595       * @param  string $type        Type of extraction
 596       * @param  string $extraction  The text to extract
 597       * @param  string $replacement Text that will replace the extraction when re-injected
 598       * @param  string $text        Text to extract out of
 599       * @return string
 600       */
 601      protected function createExtraction($type, $extraction, $replacement, $text)
 602      {
 603          $hash = md5($replacement);
 604          self::$extractions[$type][$hash] = $replacement;
 605  
 606          return str_replace($extraction, "{$type}_{$hash}", $text);
 607      }

 609      /**
 610       * Injects all of the extractions.
 611       *
 612       * @param  string $text Text to inject into
 613       * @return string
 614       */
 615      protected function injectExtractions($text, $type = null)
 616      {
 617          if ($type === null) {
 618              foreach (self::$extractions as $type => $extractions) {
 619                  foreach ($extractions as $hash => $replacement) {
 620                      if (strpos($text, "{$type}_{$hash}") !== false) {

 650       */
 651      protected function getVariable($key, $data, $default = null)
 652      {
 653          if (strpos($key, $this->scopeGlue) === false) {
 654              $parts = explode('.', $key);
 655          } else {
 656              $parts = explode($this->scopeGlue, $key);
 657          }
 658          foreach ($parts as $key_part) {
 659              if (is_array($data)) {
 660                  if ( ! array_key_exists($key_part, $data)) {
 661                      return $default;
 662                  }
 663  
 664                  $data = $data[$key_part];
 665              } elseif (is_object($data)) {
 666                  if ( ! isset($data->{$key_part})) {
 667                      return $default;
 668                  }
 669  
 670                  $data = $data->{$key_part};
 671              } else {
 672                  return $default;
 673              }
 674          }
