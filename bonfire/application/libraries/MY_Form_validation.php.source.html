 112      /**
 113       * Checks that a value is unique in the database
 114       *
 115       * i.e. '…|required|unique[users.name.id.4]|trim…'
 116       *
 117       * @abstract Rule to force value to be unique in table
 118       * @usage "unique[tablename.fieldname.(primaryKey-used-for-updates).(uniqueID-used-for-updates)]"
 119       * @access public
 120       *
 121       * @param mixed $value  The value to be checked
 122       * @param mixed $params The table and field to check against, if a second field is passed in this is used as "AND NOT EQUAL"
 123       *
 124       * @return bool
 125       */
 126  	function unique($value, $params)
 127      {
 128          $this->CI->form_validation->set_message('unique', lang('bf_form_unique'));
 129  
 130          // allow for more than 1 parameter
 131          $fields = explode(",", $params);
 132  
 133          // extract the first parameter
 134          list($table, $field) = explode(".", $fields[0], 2);
 135  
 136          // setup the db request
 137          $this->CI->db->select($field)->from($table)
 138              ->where($field, $value)->limit(1);
 139  
 140          // check if there is a second field passed in
 141          if (isset($fields[1]))
 142          {
 143              // this field is used to check that it is not the current record
 144              // eg select * from users where username='test' AND id != 4
 145  
 146              list($where_table, $where_field) = explode(".", $fields[1], 2);
 147  
 148              $where_value = $this->CI->input->post($where_field);
 149              if (isset($where_value))
 150              {
 151                  // add the extra where condition
 152                  $this->CI->db->where($where_field.' !=', $this->CI->input->post($where_field));
 153              }
 154          }
 155  
 156          // make the db request
 157          $query = $this->CI->db->get();
 158  
 159          if ($query->row())
 160          {
 161              return FALSE;
 162          }
 163          else
 164          {
 165              return TRUE;
 166          }
 167  
 168      }//end unique()

 240      /**
 241       * Check the entered password against the password strength settings.
 242       *
 243       * @access public
 244       *
 245       * @param string $str The password string to check
 246       *
 247       * @return bool
 248       */
 249  	public function valid_password($str)
 250      {
 251          // get the password strength settings from the database
 252          $min_length    = $this->CI->settings_lib->item('auth.password_min_length');
 253          $use_nums   = $this->CI->settings_lib->item('auth.password_force_numbers');
 254          $use_syms   = $this->CI->settings_lib->item('auth.password_force_symbols');
 255          $use_mixed  = $this->CI->settings_lib->item('auth.password_force_mixed_case');
 256  
 257          // Check length
 258          if (strlen($str) < $min_length)
 259          {
 260              $this->CI->form_validation->set_message('valid_password', str_replace('{min_length}', $min_length, lang('bf_form_valid_password') ) );
 261              return FALSE;
 262          }
 263  
 264          // Check numbers
 265          if ($use_nums)
 266          {
 267              if (0 === preg_match('/[0-9]/', $str))
 268              {
 269                  $this->CI->form_validation->set_message('valid_password', lang('bf_form_valid_password_nums'));
 270                  return FALSE;
 271              }
 272          }
 273  
 274          // Check Symbols
 275          if ($use_syms)
 276          {
 277              if (0 === preg_match('/[!@#$%^&*()._]/', $str))
 278              {
 279                  $this->CI->form_validation->set_message('valid_password', lang('bf_form_valid_password_syms'));
 280                  return FALSE;
 281              }
 282          }
 283  
 284          // Mixed Case?
 285          if ($use_mixed)
 286          {
 287              if (0 === preg_match('/[A-Z]/', $str))
 288              {
 289                  $this->CI->form_validation->set_message('valid_password', lang('bf_form_valid_password_mixed_1'));
 290                  return FALSE;
 291              }
 292  
 293              if (0 === preg_match('/[a-z]/', $str))
 294              {
 295                  $this->CI->form_validation->set_message('valid_password', lang('bf_form_valid_password_mixed_2'));
 296                  return FALSE;
 297              }
 298          }
 299  
 300          return TRUE;
 301  
 302      }//end valid_password()

 318  	public function allowed_types($str, $types = NULL)
 319      {
 320          if (!$types)
 321          {
 322              log_message('debug', 'form_validation method allowed_types was called without any allowed types.');
 323              return FALSE;
 324          }
 325  
 326          $type = explode('|', $types);
 327          $filetype = pathinfo($str['name'],PATHINFO_EXTENSION);
 328  
 329          if (!in_array($filetype, $type))
 330          {
 331              $this->CI->form_validation->set_message('allowed_types', lang('bf_form_allowed_types'));
 332              return FALSE;
 333          }
