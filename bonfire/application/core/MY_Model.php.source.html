 592      {
 593          if ($this->skip_validation === false) {
 594              $data = $this->validate($data, 'insert');
 595              if ($data === false) {
 596                  return false;
 597              }
 598          }
 599  
 600          $data = $this->trigger('before_insert', $data);
 601  
 602          if ($this->set_created === true && $this->log_user === true
 603              && ! array_key_exists($this->created_by_field, $data)
 604             ) {
 605              $data[$this->created_by_field] = $this->auth->user_id();
 606          }
 607  
 608          // Insert it
 609          $status = $this->db->insert($this->table_name, $data);
 610  
 611          if ($status == false) {
 612              $this->error = $this->get_db_error_message();
 613          } elseif ($this->return_insert_id) {
 614              $id = $this->db->insert_id();
 615  
 616              $status = $this->trigger('after_insert', $id);
 617          }
 618  
 619          return $status;

 665

 686      {

 692                      }

 696          {
 697              $where = array($this->key => $where);
 698          }
 699  
 700          $data = $this->trigger('before_update', $data);
 701  
 702          // Add the user id if using a modified_by field
 703          if ($this->set_modified === TRUE && $this->log_user === TRUE && !array_key_exists($this->modified_by_field, $data))
 704          {
 705              $data[$this->modified_by_field] = $this->auth->user_id();
 706          }
 707  
 708          if ($result = $this->db->update($this->table_name, $data, $where))
 709          {
 710              $this->trigger('after_update', array($data, $result));
 711              return TRUE;
 712          }
 713  
 714          return FALSE;

1233      /**
1234       * Triggers a model-specific event and call each of it's observers.
1235       *
1236       * @param string     $event     The name of the event to trigger
1237       * @param mixed     $data     The data to be passed to the callback functions.
1238       *
1239       * @return mixed
1240       */
1241  	public function trigger($event, $data=false)
1242      {
1243          if ( ! isset($this->$event) || ! is_array($this->$event))
1244          {
1245              return $data;
1246          }
1247  
1248          foreach ($this->$event as $method)
1249          {
1250              if (strpos($method, '('))
1251              {
1252                  preg_match('/([a-zA-Z0-9\_\-]+)(\(([a-zA-Z0-9\_\-\., ]+)\))?/', $method, $matches);
1253                  $this->callback_parameters = explode(',', $matches[3]);
1254              }

1256              $data = call_user_func_array(array($this, $method), array($data));
1257          }

1264      /**
1265       * Get the validation rules for the model
1266       *
1267       * @uses $empty_validation_rules Observer to generate validation rules if they are empty
1268       *
1269       * @param String $type Either 'update' or 'insert', appends rules set in $insert_validation_rules on insert
1270       *
1271       * @return array    The validation rules for the model or an empty array
1272       */
1273      public function get_validation_rules($type='update')
1274      {
1275          $temp_validation_rules = $this->validation_rules;
1276  
1277          /*
1278           * When $validation_rules is empty (or not an array), try to generate
1279           * them by triggering the $empty_validation_rules observer
1280           *
1281           * If the observer returns a non-empty array, set $validation_rules so
1282           * they aren't re-generated for this instance of the model
1283           */
1284  
1285          if (empty($temp_validation_rules) || ! is_array($temp_validation_rules))
1286          {
1287              $temp_validation_rules = $this->trigger('empty_validation_rules', $temp_validation_rules);
1288              if (empty($temp_validation_rules) || ! is_array($temp_validation_rules))
1289              {
1290                  return array();
1291              }
1292  
1293              $this->validation_rules = $temp_validation_rules;
1294          }
1295  
1296          // Any insert additions?
1297          if ($type == 'insert'
1298              && is_array($this->insert_validation_rules)
1299              && ! empty($this->insert_validation_rules)
1300             ) {
1301              // Get the index for each field in the validation rules
1302              $fieldIndexes = array();
1303              foreach ($temp_validation_rules as $key => $validation_rule) {
1304                  $fieldIndexes[$validation_rule['field']] = $key;
1305              }
1306  
1307              foreach ($this->insert_validation_rules as $key => $rule) {
1308                  if (is_array($rule)) {
1309                      $insert_rule = $rule;
1310                  } else {
1311                      // if $key isn't a field name and $insert_rule isn't an
1312                      // array, we probably can't do anything useful, so skip it
1313                      if (is_numeric($key)) {
1314                          continue;
1315                      }
1316  
1317                      $insert_rule = array(
1318                          'field' => $key,
1319                          'rules' => $rule,
1320                      );
1321                  }
1322  
1323                  /*
1324                   * if the field is already in the validation rules,
1325                   * we update the validation rule to merge the insert rule
1326                   * (or replace an empty rule)
1327                   */
1328                  if (isset($fieldIndexes[$insert_rule['field']])) {
1329                      $fieldKey = $fieldIndexes[$insert_rule['field']];
1330                      if (empty($temp_validation_rules[$fieldKey]['rules'])) {
1331                          $temp_validation_rules[$fieldKey]['rules'] = $insert_rule['rules'];
1332                      } else {
1333                          $temp_validation_rules[$fieldKey]['rules'] .= '|' . $insert_rule['rules'];
1334                      }
1335                  } else {
1336                      // Otherwise we add the insert rule to the validation rules
1337                      $temp_validation_rules[] = $insert_rule;
1338                  }
1339              }
1340          }
1341  
1342          return $temp_validation_rules;
1343      }

1347      /**
1348       * Validates the data passed into it based upon the form_validation rules
1349       * setup in the $this->validation_rules property.
1350       *
1351       * If $type == 'insert', any additional rules in the class var $insert_validation_rules
1352       * for that field will be added to the rules.
1353       *
1354       * @param  array $data      An array of data to validate
1355       * @param  string $type     Either 'update' or 'insert'.
1356       * @return array/bool       The original data or FALSE
1357       */
1358  	public function validate($data, $type='update')
1359      {
1360          if ($this->skip_validation) {
1361              return $data;
1362          }
1363  
1364          $current_validation_rules = $this->get_validation_rules($type);
1365  
1366          if (empty($current_validation_rules)) {
1367              return $data;
1368          }
1369  
1370          foreach ($data as $key => $val) {
1371              $_POST[$key] = $val;
1372          }
1373  
1374          $this->load->library('form_validation');
1375  
1376          /*
1377           * $current_validation_rules can be an array of rules, which we pass
1378           * to set_rules(), or a string that is passed to run(), which will
1379           * attempt to load the rules from a config file (run() will ignore
1380           * the input otherwise)
1381           */
1382          if (is_array($current_validation_rules)) {
1383              $this->form_validation->set_rules($current_validation_rules);
1384              $valid = $this->form_validation->run();
1385          } else {
1386              $valid = $this->form_validation->run($current_validation_rules);
1387          }
1388  
1389          if ($valid !== true) {
1390              return false;
1391          }
1392  
1393          return $data;
1394      }

1478       * @return string
1479       */
1480  	protected function get_db_error_message()
1481      {
1482          switch ($this->db->platform())
1483          {
1484              case 'cubrid':
1485                  return cubrid_errno($this->db->conn_id);
1486              case 'mssql':
1487                  return mssql_get_last_message();
1488              case 'mysql':
1489                  return mysql_error($this->db->conn_id);
1490              case 'mysqli':
1491                  return mysqli_error($this->db->conn_id);
1492              case 'oci8':
1493                  // If the error was during connection, no conn_id should be passed
1494                  $error = is_resource($this->db->conn_id) ? oci_error($this->db->conn_id) : oci_error();
1495                  return $error['message'];
1496              case 'odbc':
1497                  return odbc_errormsg($this->db->conn_id);
1498              case 'pdo':
1499                  $error_array = $this->db->conn_id->errorInfo();
1500                  return $error_array[2];
1501              case 'postgre':
1502                  return pg_last_error($this->db->conn_id);
1503              case 'sqlite':
1504                  return sqlite_error_string(sqlite_last_error($this->db->conn_id));
1505              case 'sqlsrv':
1506                  $error = array_shift(sqlsrv_errors());
1507                  return !empty($error['message']) ? $error['message'] : null;
1508              default:
1509                  /*
1510                   * !WARNING! $this->db->_error_message() is supposed to be private and
1511                   * possibly won't be available in future versions of CI
1512                   */
1513                  return $this->db->_error_message();
1514          }
