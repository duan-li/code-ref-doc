  38       * Load a language file
  39       *
  40       * This version always loads english first (as a fallback).
  41       * It will tolerate either file being missing (but not both).
  42       *
  43       * Parameters not documented here are listed by CI and MX Lang.
  44       * Compatibility with these parameters is provided, though
  45       * not all of the bugs have been implemented.  For full
  46       * details of bug compatibility, please read our code comments.
  47       *
  48       * The langfile parameter should not be considered optional (!)
  49       *
  50       * @param    string    the name of the language file to be loaded
  51       * @param    string    the language (english, etc.)
  52       * @return    void
  53       */
  54  	public function load($langfile = '', $idiom = '', $return = FALSE, $add_suffix = TRUE, $alt_path = '', $module = '')
  55      {
  56          if (is_array($langfile)) {
  57              foreach($langfile as $_lang) $this->load($_lang);
  58              return $return ? $this->language : TRUE;
  59          }
  60  
  61          // This check ignores $idiom, matching CI behaviour
  62          // (though we don't do the buggy add_suffix dance at this point).
  63          if (in_array($langfile . '_lang.php', $this->is_loaded, TRUE))
  64          {
  65              // Also we return a correct value here.
  66              // It's too hard to consistently match the CI bug where it returns void
  67              // (see the second is_loaded check in __load(), following the add_suffix dance).
  68              return $return ? $this->language : TRUE;
  69          }
  70  
  71          if ($idiom == '')
  72          {
  73              $config = get_config();
  74              $idiom = $config['language'];
  75          }
  76          if ($module == '')
  77          {
  78              $module = CI::$APP->router->fetch_module();
  79          }
  80  
  81          $loaded = $this->__load($langfile, 'english', $add_suffix, $alt_path, $module);
  82  
  83          if ($idiom != 'english')
  84          {
  85              $loaded_native = $this->__load($langfile, $idiom, $add_suffix, $alt_path, $module);
  86              if ($loaded_native)
  87              {
  88                  $loaded = array_merge($loaded, $loaded_native);

  90              else
  91              {

  94              }

 121      // @return array - empty means failure
 122  	private function __load($langfile, $idiom, $add_suffix, $alt_path, $module)
 123      {
 124          $lang = array();
 125  
 126          list($path, $file) = Modules::find($langfile.'_lang', $module, 'language/'.$idiom.'/');
 127          if ($path)
 128          {
 129              // Module file
 130              $lang = Modules::load_file($file, $path, 'lang');
 131  
 132              // Save full path for debug message below
 133              $file = $path . $file;
 134          }
 135  
 136          if (empty($lang))
 137          {
 138              // This code copied from CI Lang.  Obviously this means
 139              // passing $add_suffix=FALSE doesn't work for module files
 140              // (above); the resulting behaviour matches MX_Lang.
 141              $langfile = str_replace('.php', '', $langfile);
 142              if ($add_suffix == TRUE)
 143              {
 144                  // Extra period matches code from CI Lang
 145                  // so in reality this won't work if you pass the '_lang' in.
 146                  $langfile = str_replace('_lang.', '', $langfile).'_lang';
 147              }
 148              $langfile .= '.php';
 149              if (in_array($langfile, $this->is_loaded, TRUE))
 150              {
 151                  // Inefficient, but this should only happen in the add_suffix=FALSE bugpath.
 152                  // Note CI would incorrectly return void to the caller, but we need to
 153                  // let load() know we suceeded otherwise it'll abort the script.
 154                  return $this->language;
 155              }
 156  
 157              // Determine where the language file is and load it
 158              if ($alt_path != '' && file_exists($alt_path.'language/'.$idiom.'/'.$langfile))
 159              {
 160                  $file = $alt_path.'language/'.$idiom.'/'.$langfile;
 161                  include($file);
 162              }
 163              else
 164              {
 165                  foreach (get_instance()->load->get_package_paths(TRUE) as $package_path)
 166                  {
 167                      $file = $package_path.'language/'.$idiom.'/'.$langfile;
 168                      if (file_exists($file))
 169                      {
 170                          include $file;
 171                          if (!isset($lang))
 172                          {
 173                              log_message('error', "Language file contains no data? $file");
 174                          }
 175                          break;
 176                      }
 177                  }
 178              }
 179          }
 180  
 181          if (!empty($lang))
 182          {
 183              log_message('debug', 'Bonfire MY_Lang: Language file loaded: '.$file);
 184          }
 185  
 186          return $lang;
 187      }

 205
