 323              }
 324              else
 325              {
 326                  $modules = directory_map($path);
 327              }
 328  
 329              // If the element is not an array, we know that it's a file,
 330              // so we ignore it, otherwise it is assumbed to be a module.
 331              if ( ! is_array($modules) || ! count($modules))
 332              {
 333                  continue;
 334              }
 335  
 336              foreach ($modules as $mod_name => $values)
 337              {
 338                  if (is_array($values))
 339                  {
 340                      // Add just the specified folder for this module
 341                      if ( ! empty($module_folder) && isset($values[$module_folder]) && count($values[$module_folder]))
 342                      {
 343                          $files[$mod_name] = array(
 344                              $module_folder    => $values[$module_folder],
 345                          );
 346                      }
 347                      // Add the entire module
 348                      elseif (empty($module_folder))
 349                      {
 350                          $files[$mod_name] = $values;
 351                      }
 352                  }
 353              }
 354          }
 355  
 356          return count($files) ? $files : false;
 357      }

 521       * Implode an array with the key and value pair giving a glue,

 524       * Encode Query Strings

 533  	function array_implode($glue, $separator, $array)
 534      {
 535          if ( ! is_array( $array ) )
 536          {
 537              return $array;
 538          }
 539  
 540          $string = array();
 541  
 542          foreach ( $array as $key => $val )
 543          {
 544              if ( is_array( $val ) )
 545              {
 546                  $val = implode( ',', $val );
 547              }
 548  
 549              $string[] = "{$key}{$glue}{$val}";
 550          }
 551  
 552          return implode( $separator, $string );
