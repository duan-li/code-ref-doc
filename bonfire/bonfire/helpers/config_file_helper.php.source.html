 230       * @return string A string of text which will make up the array values in the config file.
 231       */
 232  	function config_array_output($array, $num_tabs=1)
 233      {
 234          if (!is_array($array))
 235          {
 236              return FALSE;
 237          }
 238  
 239          $tval  = 'array(';
 240  
 241          // allow for two-dimensional arrays
 242          $array_keys = array_keys($array);
 243          // check if they are basic numeric keys
 244          if (is_numeric($array_keys[0]) && $array_keys[0] == 0)
 245          {
 246              $tval .= "'".implode("','", $array)."'";
 247          }
 248          else
 249          {
 250              $tabs = "";
 251              for ($num=0;$num<$num_tabs;$num++)
 252              {
 253                  $tabs .= "\t";
 254              }
 255  
 256              // non-numeric keys
 257              foreach ($array as $key => $value)
 258              {
 259                  if(is_array($value))
 260                  {
 261                      $num_tabs++;
 262                      $tval .= "\n".$tabs."'".$key."' => ". config_array_output($value, $num_tabs). ",";
 263                  }
 264                  else
 265                  {
 266                      $tval .= "\n".$tabs."'".$key."' => '". $value. "',";
 267                  }
 268              }//end foreach
 269  
 270              $tval .= "\n".$tabs;
 271          }//end if
 272  
 273          $tval .= ')';
 274  
 275          return $tval;
 276  
 277      }//end config_array_output()
