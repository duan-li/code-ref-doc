 332

 335              show_error('The Database name must be set to connect to MongoDB.', 500);

 337          
 338          if (!empty($this->username) && !empty($this->password))
 339          {
 340              $connection_string .= "{$this->username}:{$this->password}@";
 341          }
 342          
 343          if (isset($this->port) && !empty($this->port))
 344          {
 345              $connection_string .= "{$this->hostname}:{$this->port}";
 346          } 
 347          else
 348          {
