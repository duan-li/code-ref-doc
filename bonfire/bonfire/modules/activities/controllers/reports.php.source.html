 226      /**
 227       * Delete the entries in the activity log for the specified area.
 228       *
 229       * @access private
 230       *
 231       * @param string $action The area we are in
 232       * @param string $which  A specific value to match, or "all"
 233       *
 234       * @return void
 235       */
 236  	private function _delete_activity($action, $which)
 237      {
 238          // check for permission to delete this
 239          $permission = str_replace('activity_', '',$action);
 240          if (!has_permission('Activities.'.ucfirst($permission).'.Delete')) {
 241              Template::set_message(lang('activity_restricted'), 'error');
 242              return;
 243          }
 244  
 245          if (empty($action))
 246          {
 247              Template::set_message('Delete section not specified', 'error');
 248              return;
 249          }
 250  
 251          if (empty($which))
 252          {
 253              Template::set_message('Delete value not specified', 'error');
 254              return;
 255          }
 256  
 257          // different delete where statement switch
 258          switch ($action)
 259          {
 260              case 'activity_date':
 261                  $value = 'activity_id';
 262              break;
 263  
 264              case 'activity_module':
 265                  $value = 'module';
 266              break;
 267  
 268              default:
 269                  $value = 'user_id';
 270              break;
 271          }
 272  
 273          // set a default delete then check if delete "all" was chosen
 274          $delete = ($value == 'activity_id') ? $value ." < '".$which."'" : $value ." = '".$which."'";
 275          if ($which == 'all')
 276          {
 277              $delete = $value ." != 'tsTyImbdOBOgwIqtb94N4Gr6ctatWVDnmYC3NfIfczzxPs0xZLNBnQs38dzBYn8'";
 278          }
 279  
 280          // check if they can delete their own stuff
 281          $delete .= (has_permission('Activities.Own.Delete')) ? '' : " AND user_id != '" . $this->auth->user_id()."'";
 282  
 283          $affected = $this->activity_model->delete_where($delete);
 284          if (is_numeric($affected))
 285          {
 286              Template::set_message(sprintf(lang('activity_deleted'), $affected), 'success');
 287              $this->activity_model->log_activity($this->auth->user_id(), 'deleted ' . $affected . ' activities', 'activities');
 288          }
 289          else if (isset($affected))
 290          {
 291              Template::set_message(lang('activity_nothing'),'attention');
 292          }
 293          else
 294          {
 295              Template::set_message('Error : '.$this->activity_model->error, 'error');
 296          }
 297  
 298      }//end _delete_activity()

 300      /**
 301       * Gets all the activity based on parameters passed
 302       *
 303       * @access public
 304       *
 305       * @param string $which      Filter the activities by type
 306       * @param bool   $find_value Value to filter by
 307       *
 308       * @return void
 309       */
 310  	private function _get_activity($which='activity_user',$find_value=FALSE)
 311      {
 312          // check if $find_value has anything in it
 313          if ($find_value === FALSE)
 314          {
 315              $find_value = ($this->input->post($which.'_select') == '') ? $this->uri->segment(5) : $this->input->post($which.'_select');
 316          }
 317  
 318          if (isset($_POST['delete']))
 319          {
 320              $this->_delete_activity($which, $find_value);
 321          }
 322  
 323          Template::set('filter', $this->input->post($which.'_select'));
 324  
 325          // set a couple default variables
 326          $options = array('all' => lang('activity_all'));
 327          $name = lang('activity_all');
 328  
 329          switch ($which)
 330          {
 331              case 'activity_module':
 332                  $modules = module_list();
 333                  foreach ($modules as $mod)
 334                  {
 335                      $options[$mod] = $mod;
 336  
 337                      if ($find_value == $mod)
 338                      {
 339                          $name = ucwords($mod);
 340                      }
 341                  }
 342                  $where = 'module';
 343              break;
 344  
 345              case 'activity_date':
 346                  foreach($this->activity_model->find_all_by('deleted', 0) as $e)
 347                  {
 348                      $options[$e->activity_id] = $e->created_on;
 349  
 350                      if ($find_value == $e->activity_id)
 351                      {
 352                          $name = $e->created_on;
 353                      }
 354                  }
 355                  $where = 'activity_id';
 356              break;
 357  
 358              case 'activity_own':
 359              default:
 360                  if (has_permission('Activities.User.View'))
 361                  {
 362                      foreach($this->user_model->where('users.deleted', 0)->find_all() as $e)
 363                      {
 364                          $options[$e->id] = $e->username;
 365  
 366                          if ($find_value == $e->id)
 367                          {
 368                              $name = $e->username;
 369                          }
 370                      }
 371                  }
 372                  else if (has_permission('Activities.Own.View'))
 373                  {
 374                      $options = array();
 375                      $options[$this->current_user->id] = $this->current_user->username;
 376                      $name = $this->current_user->username;
 377                  }
 378  
 379                  $where = 'user_id';
 380              break;
 381          }
 382  
 383          // set some vars for the view
 384          $vars = array(
 385              'which'            => $which,
 386              'view_which'    => ucwords(lang($which)),
 387              'name'            => $name,
 388              'delete_action'    => $where,
 389              'delete_id'        => $find_value
 390          );
 391          Template::set('vars', $vars);
 392  
 393          // if we have a filter, apply it here
 394          $this->db->order_by($where,'asc');
 395          if (!empty($find_value) && $find_value != 'all')
 396          {
 397              $where = ($where == 'activity_id') ? 'activity_id <' : $where;
 398              $this->db->where($where,$find_value);
 399  
 400              $this->db->where('activities.deleted', 0);
 401              $total = $this->activity_model->count_by($where, $find_value);
 402  
 403              // set this again for use in the main query
 404              $this->db->where($where,$find_value);
 405          }
 406          else
 407          {
 408              $total = $this->activity_model->count_by('activities.deleted', 0);
 409          }
 410  
 411          // does user have permission to see own records?
 412          if (!has_permission('Activities.Own.View'))
 413          {
 414              $this->db->where('activities.user_id != ', $this->auth->user_id());
 415          }
 416  
 417          // don't show the deleted records
 418          $this->db->where('activities.deleted', 0);
 419  
 420          // Pagination
 421          $this->load->library('pagination');
 422  
 423          $offset = $this->input->get('per_page');
 424  
 425          $limit = $this->settings_lib->item('site.list_limit');
 426  
 427          $this->pager['base_url']             = current_url() .'?';
 428          $this->pager['total_rows']             = $total;
 429          $this->pager['per_page']             = $limit;
 430          $this->pager['page_query_string']    = true;
 431  
 432          $this->pagination->initialize($this->pager);
 433  
 434          // get the activities
 435          $this->db->join('users', 'activities.user_id = users.id', 'left');
 436          $this->db->order_by('activity_id','desc'); // most recent stuff on top
 437          $this->db->select('activity, module, activities.created_on AS created, username');
 438          Template::set('activity_content', $this->activity_model->limit($limit, $offset)->find_all());
 439  
 440          Template::set('select_options', $options);
 441  
 442          Template::set_view('reports/view');
 443          Template::render();
 444  
 445      }//end _get_activity()
