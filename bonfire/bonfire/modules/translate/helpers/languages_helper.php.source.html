 131      {
 132  
 133          if (!is_dir($path))
 134          {
 135              return NULL;

 138          $files = array();
 139

 154       * Loads a single language file into an array.

 156       * @param string $filename The name of the file to locate. The file will be found by looking in all modules.

 162      {
 163          if (empty($filename))
 164          {
 165              return NULL;
 166          }
 167  
 168          $array = FALSE;
 169  
 170          // Is it the application_lang file?
 171          if ($filename == 'application_lang.php' || $filename == 'datatable_lang.php')
 172          {
 173              $path = APPPATH .'language/'. $language .'/'. $filename;
 174          }
 175          // Look in modules
 176          else
 177          {
 178              $module = str_replace('_lang.php', '', $filename);
 179  
 180              $path = module_file_path($module, 'language', $language .'/'. $filename);
 181          }
 182  
 183          // Load the actual array
 184          if (is_file($path))
 185          {
 186              include($path);
 187          }

 190          {
 191              $array = $lang;

 201  if (!function_exists('save_lang_file'))
 202  {
 203      /**
 204       * Save a language file
 205       *
 206       * @param string $filename The name of the file to locate. The file will be found by looking in all modules.
 207       * @param string $language The language to retrieve.
 208       * @param array  $settings An array of the language settings
 209       * @param bool   $return   TRUE to return the contents or FALSE to write to file
 210       *
 211       * @return mixed A string when the $return setting is TRUE
 212       */
 213  	function save_lang_file($filename=NULL, $language='english', $settings=NULL, $return=FALSE)
 214      {
 215          if (empty($filename) || !is_array($settings))
 216          {
 217              return FALSE;
 218          }
 219  
 220          // Is it the application_lang file?
 221          if ($filename == 'application_lang.php' || $filename == 'datatable_lang.php')
 222          {
 223              $orig_path = APPPATH .'language/english/'. $filename;
 224              $path = APPPATH .'language/'. $language .'/'. $filename;
 225          }
 226          // Look in core modules
 227          else
 228          {
 229              $module = str_replace('_lang.php', '', $filename);
 230  
 231              $orig_path = module_file_path($module, 'language', 'english/'. $filename);
 232              $path = module_file_path($module, 'language', $language .'/'. $filename);
 233  
 234              // If it's empty still, just grab the module path
 235              if (empty($path))
 236              {
 237                  $path = module_path($module, 'language');
 238              }
 239          }
 240  
 241          // Load the file so we can loop through the lines
 242          if (!is_file($orig_path))
 243          {
 244              return FALSE;
 245          }
 246          $contents = file_get_contents($orig_path);
 247          $contents = trim($contents) . "\n";
 248  
 249          if (!is_file($path))
 250          {
 251              // Create the folder...
 252              $folder = basename($path) == 'language' ? $path .'/'. $language : dirname($path);
 253  
 254              if (!is_dir($folder))
 255              {
 256                  mkdir($folder);
 257                  $path = basename($path) == 'language' ? $folder .'/'. $module .'_lang.php' : $path;
 258              }
 259          }
 260  
 261          // Save the file.
 262          foreach ($settings as $name => $val)
 263          {
 264              // Use strrpos() instead of strpos() so we don't lose data
 265              // when people have put duplicate keys in the english files
 266              $start = strrpos($contents, '$lang[\''.$name.'\']');
 267              if ($start === FALSE)
 268              {
 269                  // tried to add non-existent value?
 270                  return FALSE;
 271              }
 272              $end = strpos($contents, "\n", $start) + strlen("\n");
 273  
 274              if ($val !== '')
 275              {
 276                  $val = '\'' . addcslashes($val, '\'\\') .'\'';
 277                  $replace = '$lang[\''.$name.'\'] = ' . $val . ";\n";
 278              }
 279              else
 280              {
 281                  $replace = '// ' . substr($contents, $start, $end-$start);
 282              }
 283  
 284              $contents = substr($contents, 0, $start) . $replace . substr($contents, $end);
 285  
 286          }//end foreach
 287  
 288          // is the code we are producing OK?
 289          if (!is_null(eval(str_replace('<?php', '', $contents))))
 290          {
 291              return FALSE;
 292          }
 293  
 294          // Make sure the file still has the php opening header in it...
 295          if (strpos($contents, '<?php') === FALSE)
 296          {
 297              $contents = '<?php if ( ! defined(\'BASEPATH\')) exit(\'No direct script access allowed\');' . "\n\n" . $contents;
 298          }
 299  
 300          // Write the changes out...
 301          if (!function_exists('write_file'))
 302          {
 303              $CI = get_instance();
 304              $CI->load->helper('file');
 305          }
 306  
 307          if ($return == FALSE)
 308          {
 309              $result = write_file($path, $contents);
 310          }
 311          else
 312          {
 313              return $contents;
 314          }
 315  
 316          if ($result === FALSE)
 317          {
 318              return FALSE;
 319          }
 320          else
 321          {
 322              return TRUE;
 323          }
 324  
 325      }//end save_lang_file()
 326  }
