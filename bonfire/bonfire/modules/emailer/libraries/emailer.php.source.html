 105      /**
 106       * Handles sending the emails and routing to the appropriate methods
 107       * for queueing or sending.
 108       *
 109       * Information about the email should be sent in the $data
 110       * array. It looks like:
 111       *
 112       * $data = array(
 113       *     'to' => '',    // either string or array
 114       *     'subject' => '',    // string
 115       *     'message' => '',    // string
 116       *     'alt_message' => ''    // optional (text alt to html email)
 117       * );
 118       *
 119       * @access public
 120       *
 121       * @param array $data           An array of required information need to send the email.
 122       * @param bool  $queue_override (optional) Overrides the value of $queue_emails.
 123       *
 124       * @return bool TRUE/FALSE    Whether the operation was successful or not.
 125       */
 126  	public function send($data=array(), $queue_override=null)
 127      {
 128          // Make sure we have the information we need.
 129          $to = isset($data['to']) ? $data['to'] : FALSE;
 130          $from = settings_item('sender_email');
 131          $subject = isset($data['subject']) ? $data['subject'] : FALSE;
 132          $message = isset($data['message']) ? $data['message'] : FALSE;
 133          $alt_message = isset($data['alt_message']) ? $data['alt_message'] : FALSE;
 134  
 135          // If we don't have everything, return FALSE.
 136          if ($to == FALSE || $subject == FALSE || $message == FALSE)
 137          {
 138              $this->error = lang('em_missing_data');
 139              return FALSE;
 140          }
 141  
 142          // Wrap the $message in the email template.
 143          $mailtype = settings_item('mailtype');
 144          $templated = $message;
 145          if ($mailtype == 'html')
 146          {
 147              $templated  = $this->ci->load->view('emailer/email/_header', null, TRUE);
 148              $templated .= $message;
 149              $templated .= $this->ci->load->view('emailer/email/_footer', null, TRUE);
 150          }
 151  
 152          // Should we put it in the queue?
 153          if ($queue_override === TRUE || ($queue_override !== FALSE && $this->queue_emails == TRUE))
 154          {
 155              return $this->queue_email($to, $from, $subject, $templated, $alt_message);
 156          }
 157          // Otherwise, we're sending it right now.
 158          else
 159          {
 160              return $this->send_email($to, $from, $subject, $templated, $alt_message);
 161          }
 162  
 163      }//end send()

 214  	private function send_email($to=null, $from=null, $subject=null, $message=null, $alt_message=FALSE)
 215      {
 216          $this->ci->load->library('email');
 217          $this->ci->load->model('settings/settings_model', 'settings_model');
 218          $this->ci->email->initialize($this->ci->settings_model->select('name,value')->find_all_by('module', 'email'));
 219  
 220          $this->ci->email->set_newline("\r\n");
 221          $this->ci->email->to($to);
 222          $this->ci->email->from($from, settings_item('site.title'));
 223          $this->ci->email->subject($subject);
 224          $this->ci->email->message($message);
 225  
 226          if ($alt_message)
 227          {
 228              $this->ci->email->set_alt_message($alt_message);
 229          }
 230  
 231          if ((defined('ENVIRONMENT') && ENVIRONMENT == 'development') && $this->ci->config->item('emailer.write_to_file') === TRUE) {
 232              if (!function_exists('write_file')) {
 233                  $this->ci->load->helper('file');
 234              }
 235              write_file($this->ci->config->item('log_path').str_replace(" ","_",strtolower($subject)).substr(md5($to.time()),0,8).".html",$message);
 236              $result = TRUE;
 237          }
 238          else
 239          {
 240              $result = $this->ci->email->send();
 241          }
 242  
 243          if ($this->debug)
 244          {
 245              $this->debug_message = $this->ci->email->print_debugger();
 246          }
 247  
 248          return $result;

 254      /**
 255       * Process the email queue in chunks.
 256       *
 257       * Defaults to 33 which, if processed every 5 minutes, equals 400/hour
 258       * And should keep you safe with most ISP's. Always check your ISP's
 259       * terms of service to verify, though.
 260       *
 261       * @access public
 262       *
 263       * @param int $limit An int specifying how many emails to process at once.
 264       *
 265       * @return bool TRUE/FALSE    Whether the method was successful or not.
 266       */
 267  	public function process_queue($limit=33)
 268      {
 269          $success = TRUE;
 270  
 271          //$limit = 33; // 33 emails every 5 minutes = 400 emails/hour.
 272          $this->ci->load->library('email');
 273  
 274          $config_settings = $this->ci->settings_model->select('name,value')->find_all_by('module', 'email');
 275  
 276          // Grab records where success = 0
 277          $this->ci->db->limit($limit);
 278          $this->ci->db->where('success', 0);
 279          $query = $this->ci->db->get('email_queue');
 280  
 281          if ($query->num_rows() > 0)
 282          {
 283              $emails = $query->result();
 284          }
 285          else
 286          {
 287              return TRUE;
 288          }
 289  
 290          foreach($emails as $email)
 291          {
 292              echo '.';
 293  
 294              $this->ci->email->clear();
 295              $this->ci->email->initialize($config_settings);
 296  
 297              $this->ci->email->from(settings_item('sender_email'), settings_item('site.title'));
 298              $this->ci->email->to($email->to_email);
 299  
 300              $this->ci->email->subject($email->subject);
 301              $this->ci->email->message($email->message);
 302              $this->ci->email->set_newline("\r\n");
 303  
 304              if ($email->alt_message)
 305              {
 306                  $this->ci->email->set_alt_message($email->alt_message);
 307              }
 308  
 309              $prefix = $this->ci->db->dbprefix;
 310  
 311              if ($this->ci->email->send() === TRUE)
 312              {
 313                  // Email was successfully sent
 314                  $sql = "UPDATE {$prefix}email_queue SET success=1, attempts=attempts+1, last_attempt = NOW(), date_sent = NOW() WHERE id = " .$email->id;
 315                  $this->ci->db->query($sql);
 316              }
 317              else
 318              {
 319                  // Error sending email
 320                  $sql = "UPDATE {$prefix}email_queue SET attempts = attempts+1, last_attempt=NOW() WHERE id=". $email->id;
 321                  $this->ci->db->query($sql);
 322  
 323                  if ($this->debug)
 324                  {
 325                      $this->debug_message = $this->ci->email->print_debugger();
 326                  }
 327  
 328                  $success = FALSE;
 329              }
 330          }//end foreach
 331  
 332          return $success;
 333  
 334      }//end process_queue()
