 134  
 135          if (empty($id))
 136          {
 137              Template::set_message('Invalid Role ID.', 'error');
 138              redirect(SITE_AREA .'/settings/roles');
 139          }
 140  
 141          if (isset($_POST['save']))
 142          {
 143              if ($this->save_role('update', $id))
 144              {
 145                  Template::set_message('Role successfully saved.', 'success');
 146                  redirect(SITE_AREA .'/settings/roles');
 147              }
 148              else
 149              {
 150                  Template::set_message('There was a problem saving the role: '. $this->role_model->error);
 151              }
 152          }
 153          elseif (isset($_POST['delete']))
 154          {
 155              if ($this->role_model->delete($id))
 156              {
 157                  Template::set_message('The Role was successfully deleted.', 'success');
 158                  redirect(SITE_AREA .'/settings/roles');
 159              }
 160              else
 161              {
 162                  Template::set_message('We could not delete the role: '. $this->role_model->error, 'error');
 163              }
 164          }
 165  
 166          $role = $this->role_model->find($id);
 167          Template::set('role', $role);
 168          Template::set('contexts', list_contexts(true));
 169  
 170          $title = lang('bf_action_edit') . ' '. lang('matrix_role');
 171          Template::set('toolbar_title', isset($role->role_name) ? $title .': '. $role->role_name : $title);
 172          Template::set_view('settings/role_form');
 173          Template::render();

 181      /**
 182       * Builds the matrix for display in the role permissions form.
 183       *
 184       * @access private
 185       *
 186       * @return string The table(s) of settings, ready to be used in a form.
 187       */
 188  	public function matrix()
 189      {
 190          $id = (int)$this->uri->segment(5);
 191          $role = $this->role_model->find($id);
 192  
 193          // If the ID is empty, we are working with a new
 194          // role and won't have permissions to show.
 195          if ($id == 0)
 196          {
 197              return '<div class="alert alert-info">'. lang('role_new_permission_message') .'</div>';
 198          }
 199  
 200          // Verify role has permission to modify this role's access control
 201          if ($this->auth->has_permission('Permissions.'.ucwords($role->role_name).'.Manage')) {
 202              $permissions_full = $role->permissions;
 203  
 204              $role_permissions = $role->role_permissions;
 205  
 206              $template = array();
 207              foreach ($permissions_full as $key => $perm)
 208              {
 209                  $template[$perm->name]['perm_id'] = $perm->permission_id;
 210                  $template[$perm->name]['value'] = 0;
 211                  if(isset($role_permissions[$perm->permission_id]) )
 212                  {
 213                      $template[$perm->name]['value'] = 1;
 214                  }
 215              }
 216  
 217              // Extract our pieces from each permission
 218              $domains = array();
 219  
 220              foreach ($template as $key => $value)
 221              {
 222                  list($domain, $name, $action) = explode('.', $key);
 223  
 224                  // Add it to our domains if it's not already there.
 225                  if (!empty($domain) && !array_key_exists($domain, $domains))
 226                  {
 227                      $domains[$domain] = array();
 228                  }
 229  
 230                  // Add the preference to the domain array
 231                  if (!isset($domains[$domain][$name]))
 232                  {
 233                      $domains[$domain][$name] = array(
 234                          $action => $value
 235                      );
 236                  }
 237                  else
 238                  {
 239                      $domains[$domain][$name][$action] = $value;
 240                  }
 241  
 242                  // Store the actions separately for building the table header
 243                  if (!isset($domains[$domain]['actions']))
 244                  {
 245                      $domains[$domain]['actions'] = array();
 246                  }
 247  
 248                  if (!in_array($action, $domains[$domain]['actions']))
 249                  {
 250                      $domains[$domain]['actions'][] = $action;
 251                  }
 252              }//end foreach
 253  
 254              $auth_failed = '';
 255          }
 256          else
 257          {
 258              $auth_failed = lang('matrix_auth_fail');
 259              $domains = '';
 260          }//end if
 261  
 262          // Build the table(s) in the view to make things a little clearer,
 263          // and return it!
 264          return $this->load->view('settings/matrix', array('domains' => $domains, 'authentication_failed' => $auth_failed), TRUE);
 265  
 266      }//end matrix()

 275      /**
 276       * Saves the role record to the database
 277       *
 278       * @access private
 279       *
 280       * @param string $type The type of save operation (insert or edit)
 281       * @param int    $id   The record ID in the case of edit
 282       *
 283       * @return bool
 284       */
 285  	private function save_role($type='insert', $id=0)
 286      {
 287          if ($type == 'insert')
 288          {
 289              $this->form_validation->set_rules('role_name', 'lang:role_name', 'required|trim|unique[roles.role_name]|max_length[60]');
 290          }
 291          else
 292          {
 293              $_POST['role_id'] = $id;
 294              $this->form_validation->set_rules('role_name', 'lang:role_name', 'required|trim|unique[roles.role_name,roles.role_id]|max_length[60]');
 295          }
 296  
 297          $this->form_validation->set_rules('description', 'lang:bf_description', 'trim|max_length[255]');
 298          $this->form_validation->set_rules('login_destination', 'lang:role_login_destination', 'trim|max_length[255]');
 299          $this->form_validation->set_rules('default_context', 'lang:role_default_context', 'trim');
 300          $this->form_validation->set_rules('default', 'lang:role_default_role', 'trim|is_numeric|max_length[1]');
 301          $this->form_validation->set_rules('can_delete', 'lang:role_can_delete_role', 'trim|is_numeric|max_length[1]');
 302  
 303          if ($this->form_validation->run() === FALSE)
 304          {
 305              return FALSE;
 306          }
 307  
 308          unset($_POST['save']);
 309  
 310          // Grab our permissions out of the POST vars, if it's there.
 311          // We'll need it later.
 312          $permissions = $this->input->post('role_permissions');
 313          unset($_POST['role_permissions']);
 314  
 315          if ($type == 'insert')
 316          {
 317              $id = $this->role_model->insert($_POST);
 318  
 319              if (is_numeric($id))
 320              {
 321                  $return = TRUE;
 322              }
 323              else
 324              {
 325                  $return = FALSE;
 326              }
 327          }
 328          else if ($type == 'update')
 329          {
 330              $return = $this->role_model->update($id, $_POST);
 331          }
 332  
 333          // Add a new management permission for the role.
 334          if ($type ==  'insert')    {
 335              $add_perm = array(
 336                  'name'=>'Permissions.'.ucwords($this->input->post('role_name')).'.Manage',
 337                  'description'=>'To manage the access control permissions for the '.ucwords($this->input->post('role_name')).' role.',
 338                  'status'=>'active'
 339              );
 340  
 341              if ( $this->permission_model->insert($add_perm) ) {
 342                  // give current_role, or admin fallback, access to manage new role ACL
 343                  $assign_role = $this->session->userdata('role_id') ? $this->session->userdata('role_id') : 1;
 344                  $this->db->insert('role_permissions', array('role_id' => $assign_role, 'permission_id' => $this->db->insert_id()));
 345              }
 346              else
 347              {
 348                  $this->error = 'There was an error creating the ACL permission.';
 349              }
 350          }
 351          else
 352          {
 353              // grab the current role model name
 354              $current_name = $this->role_model->find($id)->role_name;
 355  
 356              // update the permission name (did it this way for brevity on the update_where line)
 357              $new_perm_name = 'Permissions.'.ucwords($this->input->post('role_name')).'.Manage';
 358              $old_perm_name = 'Permissions.'.ucwords($current_name).'.Manage';
 359              $this->permission_model->update_where('name',$old_perm_name,array('name'=>$new_perm_name));
 360          }
 361  
 362          // Save the permissions.
 363          if ($permissions && !$this->role_permission_model->set_for_role($id, $permissions))
 364          {
 365              $this->error = 'There was an error saving the permissions.';
 366          }
 367  
 368          unset($permissions);
 369          return $return;
 370  
 371      }//end save_role()

 405      /**
 406       * Update the role_permissions table.
 407       */
 408  	public function matrix_update()
 409      {
 410          // The profiler would add a lot of HTML to the end of the response.
 411          // This response is supposed to be single piece of text used by JS.
 412          $this->output->enable_profiler(FALSE);
 413  
 414          $pieces = explode(',',$this->input->post('role_perm'));
 415  
 416          if (!$this->auth->has_permission('Permissions.'.$this->role_model->find( (int) $pieces[0])->role_name.'.Manage'))
 417          {
 418              $this->output->set_output(lang("matrix_auth_fail"));
 419  
 420              return;
 421          }
 422  
 423          if ($this->input->post('action') == 'true')
 424          {
 425              if(is_numeric($this->role_permission_model->create_role_permissions($pieces[0],$pieces[1])))
 426              {
 427                  $msg = lang("matrix_insert_success");
 428              }
 429              else
 430              {
 431                  $msg = lang("matrix_insert_fail") . $this->role_permission_model->error;
 432              }
 433          }
 434          else
 435          {
 436              if($this->role_permission_model->delete_role_permissions($pieces[0], $pieces[1]))
 437              {
 438                  $msg = lang("matrix_delete_success");
 439              }
 440              else
 441              {
 442                  $msg = lang("matrix_delete_fail"). $this->role_permission_model->error;
 443              }
 444          }//end if
 445  
 446          $this->output->set_output($msg);
 447      }//end matrix_update()
