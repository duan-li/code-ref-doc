 217      /**
 218       * Deletes a role. By default, it will perform a soft_delete and
 219       * leave the permissions untouched. However, if $purge == TRUE, then
 220       * all permissions related to this role are also deleted.
 221       *
 222       * @access public
 223       *
 224       * @param int  $id    An integer with the role_id to delete.
 225       * @param bool $purge If FALSE, will perform a soft_delete. If TRUE, will remove the role and related permissions from db.
 226       *
 227       * @return bool TRUE/FALSE
 228       */
 229  	function delete($id=0, $purge=FALSE)
 230      {
 231          // We might not be allowed to delete this role.
 232          if ($this->can_delete_role($id) == FALSE)
 233          {
 234              $this->error = 'This role can not be deleted.';
 235              return FALSE;
 236          }
 237  
 238          if ($this->default_role_id() == $id)
 239          {
 240              $this->error = 'The default role can not be deleted.';
 241              return FALSE;
 242          }
 243  
 244          if ($purge === TRUE)
 245          {
 246              // temporarily set the soft_deletes to TRUE.
 247              $this->soft_deletes = FALSE;
 248          }
 249  
 250          // get the name for management deletion later
 251          $role = $this->role_model->find($id);
 252  
 253          // delete the record
 254          $deleted = parent::delete($id);
 255  
 256          if ($deleted === TRUE)
 257          {
 258              // Now update the users to the default role
 259              if ( ! class_exists('User_model'))
 260              {
 261                  $this->load->model('users/User_model','user_model');
 262              }
 263  
 264              $this->user_model->set_to_default_role($id);
 265  
 266              // now delete the role_permissions for this role
 267              $this->role_permission_model->delete_for_role($id);
 268  
 269              // now delete the manage permission for this role
 270              $permission_name = 'Permissions.' . ucwords($role->role_name) . '.Manage';
 271  
 272              if ( ! class_exists('Permission_model'))
 273              {
 274                  $this->load->model('permissions/permission_model');
 275              }
 276  
 277              $perm = $this->permission_model->find_by('name', $permission_name);
 278              if ($perm)
 279              {
 280                  // remove the role_permissions for this permission
 281                  $this->db->delete('role_permissions', array('permission_id' => $perm->permission_id));
 282  
 283                  if ($purge === TRUE)
 284                  {
 285                      $this->db->delete('permissions', array('name' => $permission_name));
 286                  }
 287                  else
 288                  {
 289                      $this->db->update('permissions', array('status' => 'inactive'), array('name' => $permission_name));
 290                  }
 291              }
 292          }//end if
 293  
 294          return $deleted;
 295  
 296      }//end delete()

 327      /**
 328       * Finds the permissions and role_permissions array for a single role.

 330       * @access public
 331       *
 332       * @param int $role A reference to an existing role object. This object is modified directly.
 333       *
 334       * @return void
 335       */
 336  	public function get_role_permissions(&$role)
 337      {
 338          if ( ! is_object($role))
 339          {
 340              return;
 341          }
 342  
 343          $permission_array = array();
 344  
 345          // Grab our permissions for the role.
 346          $permissions = $this->permission_model->find_all_by('status','active');
 347  
 348          // Permissions
 349          foreach ($permissions as $key => $permission)
 350          {
 351              $permission_array[$permission->name] = $permission;
 352          }
 353  
 354          $role->permissions = $permission_array;
 355  
 356          if ( ! class_exists('Role_permission_model'))
 357          {
 358              $this->load->model('roles/role_permission_model');
 359          }
 360  
 361          // Role Permissions
 362          $permission_array = array();
 363          $role_permissions = $this->role_permission_model->find_for_role($role->role_id);
 364  
 365          if (is_array($role_permissions) && count($role_permissions))
 366          {
 367              foreach ($role_permissions as $key => $permission)
 368              {
 369                  $permission_array[$permission->permission_id] = 1;
 370              }
 371          }
 372  
 373          $role->role_permissions = $permission_array;
 374          unset($permission_array);
 375  
 376      }//end get_role_permissions()
