 278      /**
 279       * Deletes a module and all of its files.
 280       *
 281       * @access public
 282       *
 283       * @return void
 284       */
 285      public function delete()
 286      {
 287          $module_name = $this->input->post('module');
 288  
 289          if ( ! empty($module_name))
 290          {
 291              $this->auth->restrict('Bonfire.Modules.Delete');
 292  
 293              $this->db->trans_begin();
 294  
 295              // check if there is a model to drop (non-table modules will have no model)
 296              $model_name = $module_name . '_model';
 297              if (module_file_path($module_name, 'models', $model_name . '.php'))
 298              {
 299                  // drop the table
 300                  $this->load->model($module_name . '/' . $model_name, 'mt');
 301                  $this->load->dbforge();
 302                  $this->dbforge->drop_table($this->mt->get_table());
 303              }
 304  
 305              // get any permission ids
 306              $query = $this->db->select('permission_id')
 307                  ->like('name', $module_name . '.', 'after')
 308                  ->get('permissions');
 309  
 310              if ($query->num_rows() > 0)
 311              {
 312                  foreach ($query->result_array() as $row)
 313                  {
 314                      // undo any permissions that exist
 315                      $this->db->where('permission_id', $row['permission_id'])
 316                          ->delete('permissions');
 317  
 318                      // and from the roles as well.
 319                      $this->db->where('permission_id', $row['permission_id'])
 320                              ->delete('role_permissions');
 321                  }
 322              }
 323  
 324              // drop the schema - old Migration schema method
 325              $module_name_lower = preg_replace("/[ -]/", "_", strtolower($module_name));
 326              if ($this->db->field_exists($module_name_lower . '_version', 'schema_version'))
 327              {
 328                  $this->dbforge->drop_column('schema_version', $module_name_lower . '_version');
 329              }
 330  
 331              // drop the Migration record - new Migration schema method
 332              if ($this->db->field_exists('version', 'schema_version'))
 333              {
 334                  $this->db->delete('schema_version', array('type' => $module_name_lower . '_'));
 335              }
 336  
 337              if ($this->db->trans_status() === FALSE)
 338              {
 339                  $this->db->trans_rollback();
 340                  Template::set_message(lang('mb_delete_trans_false'), $this->db->error, 'error');
 341              }
 342              else
 343              {
 344                  $this->db->trans_commit();
 345  
 346                  // database was successful in deleting everything. Now try to get rid of the files.
 347                  if (delete_files(module_path($module_name), true))
 348                  {
 349                      @rmdir(module_path($module_name.'/'));
 350  
 351                      // Log the activity
 352                      log_activity((integer) $this->current_user->id, lang('mb_act_delete').': ' . $module_name . ' : ' . $this->input->ip_address(), 'builder');
 353  
 354                      Template::set_message(lang('mb_delete_success'), 'success');
 355                  }
 356                  else
 357                  {
 358                      Template::set_message(lang('mb_delete_success') . lang('mb_delete_success_db_only'), 'info');
 359                  }
 360              }//end if
 361          }//end if
 362  
 363          redirect(SITE_AREA . '/developer/builder');
 364  
 365      }//end delete()

 483      /**
 484       * Returns an array with the structure and details for the fields in the specified
 485       * DB table.
 486       *
 487       * @access private
 488       *
 489       * @param string $table_name Name of the table to check
 490       *
 491       * @return mixed An array of fields or FALSE if the table does not exist
 492       */
 493      private function table_info($table_name)
 494      {
 495          $newfields = array();
 496  
 497          // check that the table exists in this database
 498          if ($this->db->table_exists($table_name))
 499          {
 500              $fields = $this->db->field_data($table_name);
 501  
 502              // We have a title - Edit it
 503              foreach ($fields as $field)
 504              {
 505                  $field_array = array();
 506  
 507                  $field_array['name'] = $field->name;
 508  
 509                  $type = '';
 510                  $max_length = null;
 511                  if (strpos($field->type, "("))
 512                  {
 513                      list($type, $max_length) = explode("--", str_replace("(", "--", str_replace(")", "", $field->type)));
 514                  }
 515                  else
 516                  {
 517                      $type = $field->type;
 518                  }
 519  
 520                  $field_array['type'] = strtoupper($type);
 521  
 522                  $values = '';
 523                  if (is_numeric($field->max_length))
 524                  {
 525                      $max_length = $field->max_length;
 526                  }
 527                  else
 528                  {
 529                      $values = $field->max_length;
 530                  }
 531                  $max_length = $max_length == null ? 1 : $max_length;
 532  
 533                  $primary_key = $field->primary_key == 1 ? 1 : 0;
 534  
 535                  $field_array['max_length']  = $max_length;
 536                  $field_array['values']      = $values;
 537                  $field_array['primary_key'] = $primary_key;
 538                  $field_array['default']     = $field->default;
 539  
 540                  $newfields[] = $field_array;
 541              } // end foreach
 542  
 543              return $newfields;
 544  
 545          }//end if
 546  
 547          return FALSE;
 548  
 549      }//end table_info()

 570          $role_id                = $this->input->post('role_id');

 573          $primary_key_field        = $this->input->post('primary_key_field');
 574          $form_error_delimiters    = explode(',', $this->input->post('form_error_delimiters'));
 575  
 576          if ($primary_key_field == '')
 577          {
 578              $primary_key_field = $this->options['primary_key_field'];
 579          }
 580  
 581          if ( ! is_array($form_error_delimiters) OR count($form_error_delimiters) != 2)
 582          {
 583              $form_error_delimiters = $this->options['$form_error_delimiters'];
 584          }
 585  
 586          $file_data = $this->modulebuilder->build_files($field_total, $module_name, $contexts, $action_names, $primary_key_field, $db_required, $form_error_delimiters, $module_description, $role_id, $table_name, $table_as_field_prefix);
 587  
 588          // make the variables available to the view file
 589          $data['module_name']        = $module_name;
 590          $data['module_name_lower']    = strtolower(preg_replace("/[ -]/", "_", $module_name));
 591          $data['controller_name']    = preg_replace("/[ -]/", "_", $module_name);
 592          $data['table_name']            = empty($table_name) ? strtolower(preg_replace("/[ -]/", "_", $module_name)) : $table_name;
 593  
 594          $data = $data + $file_data;
 595  
 596          // Allow for the Old method - update the schema first to prevent errors in duplicate column names due to Migrations.php caching db columns
 597          if ( ! $this->db->field_exists('version', 'schema_version'))
 598          {
 599              $this->dbforge->add_column('schema_version', array(
 600                  $data['module_name_lower'] . '_version' => array(
 601                      'type'            => 'INT',
 602                      'constraint'    => 4,
 603                      'null'            => true,
 604                      'default'        => 0,
 605                  ),
 606              ));
 607          }

 643      {
 644          for ($counter = 1; $this->field_total >= $counter; $counter++)
 645          {
 646              // nothing has been entered into this field so we don't need to check
 647              // or the field being checked is the same as the field we are checking from
 648              if ($_POST["view_field_name$counter"] == '' || $fieldno == $counter)
 649              {
 650                  continue;
 651              }
 652  
 653              if ($str == $_POST["view_field_name{$counter}"])
 654              {
 655                  $this->form_validation->set_message('_no_match', sprintf(lang('mb_validation_no_match'), lang('mb_form_field_details'), lang('mb_form_fieldname'), $fieldno, $counter));

 657              }
 658          }
