 312      /**
 313       * Perform a restore from a database backup.
 314       *
 315       * @access public
 316       *
 317       * @param string $filename Name of the file to restore
 318       *
 319       * @return void
 320       */
 321  	public function restore($filename=null)
 322      {
 323          Template::set('filename', $filename);
 324  
 325          if (isset($_POST['restore']) && !empty($filename))
 326          {
 327              // Load the file from disk.
 328              $this->load->helper('file');
 329              $file = file($this->backup_folder . $filename);
 330  
 331              $s = '';
 332              $templine = '';
 333  
 334              if (!empty($file))
 335              {
 336                  // Loop through each line
 337                  foreach ($file as $line)
 338                  {
 339                      // Skip it if it's a comment
 340                      if (substr(trim($line), 0, 1) != '#' || substr($line, 0 ,1) != '')
 341                      {
 342                          // Add this line to the current segment
 343                          $templine .= $line;
 344                          // If it has a semicolon at the end, it's the end of a query.
 345                          if (substr(trim($line), -1, 1) == ';')
 346                          {
 347                              // Perform the query...
 348                              if($this->db->query($templine))
 349                              {
 350                                  // Query Success
 351                                  $s .= "<strong style='color: green;'>Successfull Query</strong>: <span class='small'>$templine</span><br/>";
 352  
 353                                  // so reset our templine so we can start a new one
 354                                  $templine = '';
 355                              }
 356                              else {
 357                                  $s .= "<strong style='color:red'>Unsuccessful Query:</strong> $templine<br/><br/>";
 358                                  $templine = '';
 359                              }
 360                          }
 361                      }
 362                  }//end foreach
 363  
 364                  // Tell the results
 365                  Template::set('results', $s);
 366              }
 367              else
 368              {
 369                  // Couldn't read from file.
 370                  Template::set_message('Could not read the file: /application/db/backups/' . $filename . '.', 'error');
 371                  redirect(SITE_AREA .'/developer/database/backups');
 372              }
 373          }//end if
 374  
 375          // Show our verification screen.
 376          Template::set_view('developer/restore');
 377          Template::set('toolbar_title', 'Database Restore');
 378          Template::render();
 379  
 380      }//end restore()

 406                      $failed += 1;

 408              }

 458       *
 459       * @access public
 460       *
 461       * @param array $tables Array of table to drop
 462       *
 463       * @return bool
 464       */
 465  	public function drop($tables=null)
 466      {
 467          if (!empty($tables))
 468          {
 469              // Show our verification screen.
 470              Template::set('tables', $tables);
 471  
 472              Template::set_view('developer/drop');
 473              return TRUE;
 474          }
 475          else if (isset($_POST['tables']) && is_array($_POST['tables']))
 476          {
 477              // Actually delete the tables....
 478              $this->load->dbforge();
 479  
 480              foreach ($_POST['tables'] as $table)
 481              {
 482                  // dbforge automatically adds the prefix, so we need to remove it.
 483                  $prefix = $this->db->dbprefix;
 484  
 485                  if (strncmp($table, $prefix, strlen($prefix)) === 0)
 486                  {
 487                      $table = substr($table, strlen($prefix));
 488                      @$this->dbforge->drop_table($table);
 489                  }
 490              }
 491  
 492              $grammar = count($_POST['tables'] == 1) ? ' table' : ' tables';
 493              Template::set_message(count($_POST['tables']) .$grammar.' successfully dropped.', 'success');
 494              redirect(SITE_AREA .'/developer/database');
 495          }
 496          else
 497          {
 498              Template::set_message(lang('db_drop_none'), 'error');
 499              redirect(SITE_AREA .'/developer/database');
 500          }

 502      }//end drop()
