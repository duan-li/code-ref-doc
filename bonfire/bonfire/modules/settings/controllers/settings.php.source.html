 202      /**
 203       * Save the extended settings
 204       *
 205       * @access private
 206       *
 207       * @param    array    $extended_data    An array of settings to save
 208       *
 209       * @return    mixed/bool    TRUE or an inserted id if all settings saved successfully, else FALSE
 210       */
 211  	private function save_extended_settings($extended_data)
 212      {
 213          if ( ! is_array($extended_data)
 214              || empty($extended_data)
 215              || ! count($extended_data)
 216              )
 217          {
 218              return FALSE;
 219          }
 220  
 221          $setting = FALSE;
 222          foreach ($extended_data as $key => $value)
 223          {
 224              $setting = $this->settings_lib->set($key, $value);
 225              if ($setting === FALSE)
 226              {
 227                  return FALSE;
 228              }
 229          }
 230  
 231          return $setting;
 232      }// end save_extended_settings()
