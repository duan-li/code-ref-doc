 260  	public function is_installed()
 261      {
 262          // First check - Does a 'install/installed.txt' file exist? If so,
 263          // then we've likely already installed.
 264          if (is_file(APPPATH . 'config/installed.txt'))
 265          {
 266              return true;
 267          }
 268  
 269          // Does the database config exist?
 270          // If not, then we definitely haven't installed yet.
 271          if (!is_file(APPPATH . 'config/development/database.php'))
 272          {
 273              return false;
 274          }
 275  
 276          require(APPPATH . '/config/development/database.php');
 277  
 278          // If the $db['default'] doesn't exist then we can't
 279          // load our database.
 280          if (!isset($db) || !isset($db['default']))
 281          {
 282              $this->db_settings_exist = FALSE;
 283              return false;
 284          }
 285  
 286          $this->db_settings_exist = TRUE;
 287  
 288          // Just to be safe, also make sure that the database name
 289          // is actually specified.
 290          if (empty($db['default']['database']))
 291          {
 292              $this->db_exists = FALSE;
 293              return false;
 294          }
 295  
 296          $this->db_exists = TRUE;
 297  
 298          $this->ci->load->database($db['default']);
 299  
 300          // Does the users table exist?
 301          if (!$this->ci->db->table_exists('users'))
 302          {
 303              return false;
 304          }
 305  
 306          // Make sure at least one row exists in the users table.
 307          $query = $this->ci->db->get('users');
 308  
 309          if ($query->num_rows() == 0)
 310          {
 311              return false;
 312          }
 313  
 314          define('BF_INSTALLED', true);
 315  
 316          return true;
 317      }
