 239      /**
 240       * Renders links to stylesheets, with the $asset_url prepended.
 241       * If a single filename is passed, it will only create a single link
 242       * for that file, otherwise, it will include any styles that have
 243       * been added with add_css below. If no style is passed it will default
 244       * to the theme's style.css file.
 245       *
 246       * When passing a filename, the filepath should be relative to the site
 247       * root (where index.php resides).
 248       *
 249       * @access public
 250       * @static
 251       *
 252       * @param mixed  $style              The style(s) to have links rendered for.
 253       * @param string $media              The media to assign to the style(s) being passed in.
 254       * @param bool   $bypass_inheritance If TRUE, will skip the check for parent theme styles.
 255       * @param bool     $bypass_module      If TRUE, will not output the css file named after the controller or the module styles.
 256       *
 257       * @return string A string containing all necessary links.
 258       */
 259  	public static function css($style=null, $media='screen', $bypass_inheritance=FALSE, $bypass_module=FALSE)
 260      {
 261          $styles = array();
 262          $return = '';
 263  
 264          //Debugging issues with media being set to 1 on module_js
 265          if ($media == '1')
 266          {
 267              $media = 'screen';
 268          }
 269  
 270          // If no style(s) has been passed in, use all that have been added.
 271          if (empty($style) && self::$globals)
 272          {
 273              // Make sure to include a file based on media type.
 274              $styles[] = array(
 275                  'file'    => $media,
 276                  'media'    => $media,
 277              );
 278  
 279              $styles = array_merge(self::$styles, $styles);
 280          }
 281          // If an array has been passed, merge it with any added styles.
 282          else if (is_array($style))
 283          {
 284              $styles = array_merge($style, self::$styles);
 285          }
 286          // If a single style has been passed in, render it only.
 287          else
 288          {
 289              $styles[] = array(
 290                  'file'    => $style,
 291                  'media'    => $media,
 292              );
 293          }
 294  
 295          if ($bypass_module == FALSE)
 296          {
 297              // Add a style named for the controller so it will be looked for.
 298              $styles[] = array(
 299                  'file'    => self::$ci->router->class,
 300                  'media' => $media,
 301              );
 302  
 303              $mod_styles    = self::find_files(self::$module_styles, 'css', $bypass_inheritance);
 304          }
 305  
 306          $styles = self::find_files($styles, 'css', $bypass_inheritance);
 307  
 308          $combine = self::$ci->config->item('assets.css_combine');
 309  
 310          // Loop through the styles, spitting out links for each one.
 311          if ( ! $combine)
 312          {
 313              foreach ($styles as $s)
 314              {
 315                  if (is_array($s))
 316                  {
 317                      if (substr($s['file'], -4) != '.css')
 318                      {
 319                          $s['file'] .= '.css';
 320                      }
 321                  }
 322                  else
 323                  {
 324                      if (substr($s, -4) != '.css')
 325                      {
 326                          $s .= '.css';
 327                      }
 328                  }
 329  
 330                  $attr = array(
 331                      'rel'    => 'stylesheet',
 332                      'type'    => 'text/css',
 333                      'href'    => is_array($s) ? $s['file'] : $s,
 334                      'media'    => !empty($s['media']) ? $s['media'] : $media,
 335                  );
 336  
 337                  $return .= '<link' . self::attributes($attr) . " />\n";
 338              }
 339          }
 340          // add the combined css
 341          else
 342          {
 343              $return .= self::combine_css($styles, $media);
 344          }
 345  
 346          if ($bypass_module == FALSE)
 347          {
 348              // Make sure we include module styles
 349              $return .= self::combine_css($mod_styles, $media, 'module');
 350          }
 351  
 352          return $return;
 353  
 354      }//end css()

 357      /**
 358       * Does the actual work of generating the combined css code.
 359       *
 360       * It is called by the css() method.
 361       *
 362       * @access public
 363       * @static
 364       *
 365       * @param array  $files An array of file arrays
 366       * @param string $media The media to assign to the style(s) being passed in.
 367       * @param string $type  Either a string 'module' or empty - defines which scripts are being combined
 368       *
 369       * @return string
 370       */
 371  	public static function combine_css($files=array(), $media='screen', $type='')
 372      {
 373          // Are there any scripts to include?
 374          if (count($files) == 0)
 375          {
 376              return;
 377          }
 378  
 379          $output = '';
 380          $min    = '';
 381  
 382          // Add the theme name to the filename
 383          // to account for different frontend/backend themes.
 384          $theme = trim(Template::get('active_theme'), '/');
 385          $theme = empty($theme) ? trim(Template::get('default_theme'), '/') : $theme;
 386  
 387          $file_name = $theme . '_' . self::$ci->router->fetch_module() . '_' . self::$ci->router->fetch_class();
 388  
 389          if (self::$ci->config->item('assets.encrypt_name') == TRUE)
 390          {
 391              $file_name = md5($file_name);
 392          }
 393  
 394          if ($type == 'module')
 395          {
 396              $file_name .= '_mod';
 397          }
 398          else
 399          {
 400              $file_name .= '_combined';
 401          }
 402  
 403          if (self::$ci->config->item('assets.css_minify') == TRUE)
 404          {
 405              // since $file_name is passed to generate_file, we don't want to add .min to it here,
 406              // but we need to add it to the href attribute below
 407              $min = '.min';
 408          }
 409  
 410          //Debugging issues with media being set to 1 on module_js
 411          if ($media == '1')
 412          {
 413              $media = 'screen';
 414          }
 415  
 416          // Create our link attributes
 417          $attr = array(
 418              'rel'    => 'stylesheet',
 419              'type'    => 'text/css',
 420              'href'    => base_url() . self::$asset_base . '/' . self::$asset_cache_folder . '/' . $file_name . $min . '.css',
 421              'media'    => $media,
 422          );
 423  
 424          if (self::generate_file($files, $file_name, 'css'))
 425          {
 426              $output .= '<link' . self::attributes($attr) . " />\n";
 427          }
 428  
 429          return $output;
 430  
 431      }//end combine_css()

 662      {

 673          }
 674          // If a string is passed, it's a single script

 689          $output  = self::external_js();

 698      /**
 699       * Does the actual work of generating the links to the js files.
 700       * It is called by the js() method, but can be used on it's own.
 701       *
 702       * If no script are passed into the first parameter, links are created for
 703       * all scripts within the self::$external_scripts array. If one or
 704       * more scripts are passed in the first parameter, only these script files
 705       * will be used to create links with, and any stored in self::$external_scripts
 706       * will be ignored.
 707       *
 708       * Note that links will not be rendered for files that cannot be found, though
 709       * scripts will full urls are not checked, but are simply included.
 710       *
 711       * @access public
 712       * @static
 713       *
 714       * @param mixed $new_js  Either a string or an array containing the names of files to link to.
 715       * @param bool  $list    If TRUE, will echo out a list of scriptnames, enclosed in quotes and comma separated. Convenient for using with third-party js loaders.
 716       * @param bool  $add_ext Automatically add the .js extension when adding files
 717       * @param bool    $bypass_globals    If TRUE, bypass global scripts for this call to external_js
 718       *
 719       * @return string
 720       */
 721  	public static function external_js($new_js=null, $list=FALSE, $add_ext=TRUE, $bypass_globals=FALSE)
 722      {
 723          $return = '';
 724          $scripts = array();
 725  
 726          // If scripts were passed, they override all other scripts.
 727          if ( ! empty($new_js))
 728          {
 729              if (is_string($new_js))
 730              {
 731                  $scripts[] = $new_js;
 732              }
 733              else if (is_array($new_js))
 734              {
 735                  $scripts = $new_js;
 736              }
 737          }
 738          else
 739          {
 740              $scripts = self::$external_scripts;
 741          }
 742  
 743          // Make sure we check for a 'global.js' file.
 744          if ($bypass_globals == FALSE && self::$globals)
 745          {
 746                  $scripts[] = 'global';
 747          }
 748  
 749          // Add a style named for the controller so it will be looked for.
 750          $scripts[] = self::$ci->router->class;
 751  
 752          // Prep our scripts array with only files
 753          // that actually can be found.
 754          $scripts = self::find_files($scripts, 'js');
 755  
 756          // We either combine the files into one...
 757          if ((empty($new_js) || is_array($new_js))
 758              && $list==FALSE
 759              && self::$ci->config->item('assets.js_combine')
 760          )
 761          {
 762              $src = self::combine_js($scripts);
 763  
 764              if ($list)
 765              {
 766                  $return .= '"'. $src .'", ';
 767              }
 768              else
 769              {
 770                  $attr = array(
 771                      'src'   => $src,
 772                      'type'  => 'text/javascript'
 773                  );
 774  
 775                  $return .= '<script'. self::attributes($attr) ." ></script>\n";
 776              }
 777          }
 778          // Or generate individual links
 779          else
 780          {
 781              // Check for HTTPS or HTTP connection
 782              $http_protocol = ! empty($_SERVER['HTTPS']) && strtolower($_SERVER['HTTPS']) !== 'off' ? 'https' : 'http';
 783  
 784              foreach ($scripts as $script)
 785              {
 786                  if (TRUE === $add_ext && substr($script, -3) != '.js')
 787                  {
 788                      $script .= '.js';
 789                  }
 790  
 791                  $attr = array('type' => 'text/javascript');
 792  
 793                  // If $script has a full url built in, leave it alone
 794                  if (strpos($script, $http_protocol . ':') !== FALSE    // Absolute URL with current protocol, which should be more likely
 795                      || strpos($script, '//') === 0                    // Protocol-relative URL
 796                      || strpos($script, 'https:') !== FALSE            // We should assume $http_protocol is most likely 'http', so check 'https' next
 797                      || strpos($script, 'http:') !== FALSE            // Finally, check 'http' in case $http_protocol is 'https'
 798                  )
 799                  {
 800                      $attr['src'] = $script;
 801                  }
 802                  // Otherwise, build the full url
 803                  else
 804                  {
 805                      $attr['src'] = base_url() . self::$asset_base . '/' . self::$asset_folders['js'] . '/' . $script;
 806                  }
 807  
 808                  if ($list)
 809                  {
 810                      $return .= '"' . $attr['src'] . '", ';
 811                  }
 812                  else
 813                  {
 814                      $return .= '<script' . self::attributes($attr) . " ></script>\n";
 815                  }
 816              }
 817          }
 818  
 819          return trim($return, ', ');
 820  
 821      }//end external_js()

 943      /**
 944       * Does the actual work of generating the combined js code.
 945       *
 946       * It is called by the external_js() and module_js() methods.
 947       *
 948       * @access public
 949       * @static
 950       *
 951       * @param array $files An array of files to be combined.
 952       * @param string $type Either a string 'module' or empty. Helps determine the file name.
 953       *
 954       * @return string
 955       */
 956  	public static function combine_js($files=array(), $type='')
 957      {
 958          // Are there any scripts to include?
 959          if (is_array($files) && count($files) == 0)
 960          {
 961              return;
 962          }
 963  
 964          $output = '';
 965          $min    = '';
 966  
 967          $theme = Template::get('active_theme');
 968          $theme = empty($theme) ? Template::get('default_theme') : $theme;
 969  
 970          // get the module name
 971          $module_name = self::$ci->router->fetch_module();
 972  
 973          // get the uri segments
 974          $uri_segments = self::$ci->uri->segment_array();
 975  
 976          // get the module name from the uri segment
 977          $context_key = array_search(self::$ci->router->fetch_class(), $uri_segments);
 978          if (FALSE !== $context_key)
 979          {
 980              $module_key = $context_key + 1;
 981              if (isset($uri_segments[$module_key]))
 982              {
 983                  $module_name = $uri_segments[$module_key];
 984              }
 985          }
 986  
 987          $file_name = trim($theme,'/') .'_'. $module_name .'_'. self::$ci->router->fetch_class();
 988  
 989          if (self::$ci->config->item('assets.encrypt_name'))
 990          {
 991              $file_name = md5($file_name);
 992          }
 993  
 994          $file_name .= $type == 'module' ? '_mod' : '_combined';
 995  
 996          if (self::$ci->config->item('assets.js_minify') == TRUE)
 997          {
 998              // since $file_name is passed to generate_file, we don't want to add .min to it here,
 999              // but we need to add it to the URL below
1000              $min = '.min';
1001          }
1002  
1003          // Create our shell opening
1004          if (self::generate_file($files, $file_name, 'js'))
1005          {
1006              $output .= base_url() . self::$asset_base . '/' . self::$asset_cache_folder . '/' . $file_name . $min . '.js';
1007          }
1008  
1009          return $output;
1010  
1011      }//end combine_js()

1122      /**
1123       * Locates file by looping through the active and default themes, and
1124       * then the assets folder (as specified in the config file).
1125       *
1126       * Files are searched for in this order...
1127       *     1 - active_theme/
1128       *     2 - active_theme/type/
1129       *     3 - default_theme/
1130       *     4 - default_theme/type/
1131       *     5 - asset_base/type
1132       *
1133       * Where 'type' is either 'css' or 'js'.
1134       *
1135       * If the file is not found, it is removed from the array. If the file
1136       * is found, a full url is created, using base_path(), unless the path
1137       * already includes 'http' at the beginning of the filename, in which case
1138       * it is simply included in the return files.
1139       *
1140       * For CSS files, if a script of the same name is found in both the
1141       * default_theme and the active_theme folders (or their type sub-folder),
1142       * they are both returned, with the default_theme linked to first, so that
1143       * active_theme styles can override those in the default_theme without
1144       * having to recreate the entire stylesheet.
1145       *
1146       * @access private
1147       * @static
1148       *
1149       * @param array  $files     Array of files
1150       * @param string $file_name Name of the file to generate
1151       * @param string $file_type Either 'css' or 'js'.
1152       *
1153       * @return bool True if file generated successfully, False if there were errors.
1154       */
1155  	private static function generate_file($files=array(), $file_name, $file_type='css')
1156      {
1157          if (count($files) == 0)
1158          {
1159              // While the file wasn't actually created,
1160              // there weren't any errors, either.
1161              return TRUE;
1162          }
1163  
1164          $site_path = Template::get('site_path');
1165  
1166          // Where to save the combined file to.
1167          $cache_path = $site_path . self::$asset_base . '/' . self::$asset_cache_folder . '/';
1168  
1169          // full file path - without the extension
1170          $file_path = $cache_path . $file_name;
1171  
1172          if (self::$ci->config->item('assets.' . $file_type . '_minify'))
1173          {
1174              $file_path .= '.min';
1175          }
1176  
1177          $file_path .= '.' . $file_type;
1178  
1179          $modified_time    = 0;    // Holds the last modified date of all included files.
1180          $actual_file_time = 0;    // The modified time of the combined file.
1181  
1182          // If the combined file already exists,
1183          // we need to grab the last modified time.
1184          if (is_file($file_path))
1185          {
1186              $actual_file_time = filemtime($file_path);
1187          }
1188  
1189          foreach ($files as $key => $file)
1190          {
1191              $app_file = '';
1192  
1193              // Javascript
1194              if ($file_type == 'js')
1195              {
1196                  if (is_array($file))
1197                  {
1198                      $app_file = $file['server_path'];
1199                  }
1200                  else
1201                  {
1202                      $app_file = $site_path . str_replace(base_url(), '', $file);
1203                  }
1204                  $app_file = strpos($app_file, '.js') ? $app_file : $app_file .'.js';
1205              }
1206              // CSS
1207              else
1208              {
1209                  $app_file = $file['server_path'];
1210              }
1211  
1212              $files_array[$key] = $app_file;
1213  
1214              // By this point, we already know that the files exist,
1215              // so just grab the modified time. If it is higher than
1216              // the previous files' modified times, keep it.
1217              $modified_time = max(filemtime($app_file), $modified_time);
1218  
1219          }//end foreach
1220  
1221          $asset_output = '';
1222  
1223          if ($actual_file_time < $modified_time)
1224          {
1225              // write to the file
1226              foreach ($files_array as $key => $file)
1227              {
1228                  $file_output = file_get_contents($file);
1229  
1230                  if ( ! empty($file_output))
1231                  {
1232                      $asset_output .= $file_output. PHP_EOL;
1233                  }
1234              }
1235  
1236              switch ($file_type)
1237              {
1238                  case 'js':
1239                      if (config_item('assets.js_minify'))
1240                      {
1241                          $asset_output = JSMin::minify($asset_output);
1242                      }
1243                      break;
1244                  case 'css':
1245                      if (config_item('assets.css_minify'))
1246                      {
1247                          $asset_output = CSSMin::minify($asset_output);
1248                      }
1249                      break;
1250                  default:
1251                      throw new LoaderException("Unknown file type - {$file_type}.");
1252                      break;
1253              }
1254  
1255              self::$ci->load->helper('file');
1256  
1257              if ( ! is_dir($cache_path))
1258              {
1259                  @mkdir($cache_path);
1260              }
1261  
1262              if ( ! write_file($file_path, $asset_output))
1263              {
1264                  return FALSE;
1265              }
1266          }
1267          elseif ($actual_file_time == 0)
1268          {
1269              return FALSE;
1270          }
1271  
1272          return TRUE;
1273  
1274      }//end generate_file()

1277      /**
1278       * Locates file by looping through the active and default themes, and
1279       * then the assets folder (as specified in the config file).
1280       *
1281       * Files are searched for in this order...
1282       *     1 - active_theme/
1283       *     2 - active_theme/type/
1284       *     3 - default_theme/
1285       *     4 - default_theme/type/
1286       *     5 - asset_base/type
1287       *
1288       * Where 'type' is either 'css' or 'js'.
1289       *
1290       * If the file is not found, it is removed from the array. If the file
1291       * is found, a full url is created, using base_path(), unless the path
1292       * already includes 'http' at the beginning of the filename, in which case
1293       * it is simply included in the return files.
1294       *
1295       * For CSS files, if a script of the same name is found in both the
1296       * default_theme and the active_theme folders (or their type sub-folder),
1297       * they are both returned, with the default_theme linked to first, so that
1298       * active_theme styles can override those in the default_theme without
1299       * having to recreate the entire stylesheet.
1300       *
1301       * @access private
1302       * @static
1303       *
1304       * @param array  $files              An array of file names to search for.
1305       * @param string $type               Either 'css' or 'js'.
1306       * @param bool   $bypass_inheritance If TRUE, will skip the check for parent theme styles.
1307       *
1308       * @return array The complete list of files with url paths.
1309       */
1310  	private static function find_files($files=array(), $type='css', $bypass_inheritance=FALSE)
1311      {
1312          // Grab the theme paths from the template library.
1313          $paths = Template::get('theme_paths');
1314          $site_path = Template::get('site_path');
1315          $active_theme = Template::get('active_theme');
1316          $default_theme = Template::get('default_theme');
1317  
1318          $new_files = array();
1319  
1320          $clean_type = $type;
1321          $type = '.' . $type;
1322  
1323          if (self::$debug)
1324          {
1325              echo "Active Theme = {$active_theme}<br/>
1326                  Default Theme = {$default_theme}<br/>
1327                  Site Path = {$site_path}<br/>
1328                  File(s) to find: ";
1329              print_r($files);
1330          }
1331  
1332          // Check for HTTPS or HTTP connection
1333          $http_protocol = ! empty($_SERVER['HTTPS']) && strtolower($_SERVER['HTTPS']) !== 'off' ? 'https' : 'http';
1334  
1335          $rtl = self::$rtl_postfix;
1336          $rtl_set = lang('bf_language_direction') == $rtl;
1337  
1338          foreach ($files as $file)
1339          {
1340              // If it's an array, we're dealing with css and it has both
1341              // a file and media keys. Store them for later use.
1342              if (is_array($file))
1343              {
1344                  if ($type == '.css')
1345                  {
1346                      $media = $file['media'];
1347                  }
1348                  $module    = isset($file['module']) ? $file['module'] : '';
1349                  $file = $file['file'];
1350              }
1351  
1352              // Strip out the file type for consistency
1353              $file = str_replace($type, '', $file);
1354  
1355              // If it contains an external URL, we're all done here.
1356              if (strpos((string)$file, $http_protocol . ':') !== FALSE    // Absolute URL with current protocol, which should be more likely
1357                      || strpos((string)$file, '//') === 0                // Protocol-relative URL
1358                      || strpos((string)$file, 'https:') !== FALSE        // We should assume $http_protocol is most likely 'http', so check 'https' next
1359                      || strpos((string)$file, 'http:') !== FALSE            // Finally, check 'http' in case $http_protocol is 'https'
1360                 )
1361              {
1362                  $new_files[] = !empty($media) ? array('file' => $file, 'media' => $media) : $file;
1363                  continue;
1364              }
1365  
1366              $found = FALSE;
1367  
1368              // Is it a module file?
1369              if ( ! empty($module))
1370              {
1371                  $file_path_name = $file;
1372  
1373                  // try the /module/assets folder
1374                  $path = module_file_path($module, 'assets', $file_path_name . $type);
1375  
1376                  // If path is empty, try the /module/assets/type folder
1377                  if (empty($path))
1378                  {
1379                      $file_path_name = $clean_type . '/' . $file;
1380                      $path = module_file_path($module, 'assets', $file_path_name . $type);
1381                  }
1382  
1383                  // If we have a path and the language is right-to-left, try adding -rtl to the file name
1384                  if ( ! empty($path) && $rtl_set)
1385                  {
1386                      $path_rtl = module_file_path($module, 'assets', $file_path_name . '-' . $rtl . $type);
1387                      if ( ! empty($path_rtl))
1388                      {
1389                          $path = $path_rtl;
1390                      }
1391                  }
1392  
1393                  if (self::$debug)
1394                  {
1395                      echo "[Assets] Looking for MODULE asset at: {$path}<br/>" . PHP_EOL;
1396                  }
1397  
1398                  // if we found the file, add it to our array for output
1399                  if ( ! empty($path))
1400                  {
1401                      $file_path = '';
1402  
1403                      $file = array(
1404                          'file'            => $file_path,
1405                          'server_path'    => $path
1406                      );
1407                      if (isset($media))
1408                      {
1409                          $file['media'] = $media;
1410                      }
1411  
1412                      $new_files[] = $file;
1413                  }
1414  
1415                  continue;
1416              }
1417              // Non-module files
1418              else
1419              {
1420                  $media = empty($media) ? '' : $media;
1421                  // We need to check all of the possible theme_paths
1422                  foreach ($paths as $path)
1423                  {
1424                      if (self::$debug)
1425                      {
1426                          echo "[Assets] Looking in:
1427                              <ul>
1428                                  <li>{$site_path}{$path}/{$default_theme}{$file}{$type}</li>
1429                                  <li>{$site_path}{$path}/{$default_theme}{$clean_type}/{$file}{$type}</li>";
1430  
1431                          if ( ! empty($active_theme))
1432                          {
1433                              echo "
1434                                  <li>{$site_path}{$path}/{$active_theme}{$file}{$type}</li>
1435                                  <li>{$site_path}{$path}/{$active_theme}{$clean_type}/{$file}{$type}</li>";
1436                          }
1437  
1438                          echo "
1439                                  <li>{$site_path}" . self::$asset_base . "/{$clean_type}/{$file}{$type}</li>
1440                              </ul>" . PHP_EOL;
1441                      }
1442  
1443                      /*
1444                       * If default_theme and active_theme are the same,
1445                       * checking the default_theme would just repeat the
1446                       * active_theme section below, resulting in duplicates
1447                       */
1448                      if ( ! $bypass_inheritance && $default_theme !== $active_theme)
1449                      {
1450                          /*
1451                              DEFAULT THEME
1452  
1453                              First, check the default theme. Add it to the array. We check here first so that it
1454                              will get overwritten by anything in the active theme.
1455                          */
1456                          if (($file_array = self::get_file_array($site_path, $path . '/' . $default_theme, $file, $type, $media)))
1457                          {
1458                              $new_files[] = $file_array;
1459                              $found = TRUE;
1460                          }
1461                          /*
1462                              If it wasn't found in the default theme root folder, look in default_theme/$type/
1463                          */
1464                          else if (($file_array = self::get_file_array($site_path, $path . '/' . $default_theme . $clean_type . '/', $file, $type, $media)))
1465                          {
1466                              $new_files[] = $file_array;
1467                              $found = TRUE;
1468                          }
1469                      }
1470                      /*
1471                          ACTIVE THEME
1472  
1473                          By grabbing a copy from both the default theme and the active theme, we can
1474                          handle simple CSS-only overrides for a theme, completely changing its appearance
1475                          through a simple child css file.
1476                      */
1477                      if ( ! empty($active_theme)) // separated this because the else if below should not run if $active_theme is empty
1478                      {
1479                          if (($file_array = self::get_file_array($site_path, $path . '/' . $active_theme, $file, $type, $media)))
1480                          {
1481                              $new_files[] = $file_array;
1482                              $found = TRUE;
1483                          }
1484                          /*
1485                              If it wasn't found in the active theme root folder, look in active_theme/$type/
1486                          */
1487                          else if (($file_array = self::get_file_array($site_path, $path . '/' . $active_theme . $clean_type . '/', $file, $type, $media)))
1488                          {
1489                              $new_files[] = $file_array;
1490                              $found = TRUE;
1491                          }
1492                      }
1493                      /*
1494                          ASSET BASE
1495  
1496                          If the file hasn't been found, yet, we have one more place to look for it:
1497                          in the folder specified by 'assets.base_folder', and under the $type sub-folder.
1498                      */
1499                      if ( ! $found)
1500                      {
1501                          // Assets/type folder
1502                          if (($file_array = self::get_file_array($site_path, self::$asset_base . '/' . $clean_type . '/', $file, $type, $media)))
1503                          {
1504                              $new_files[] = $file_array;
1505                          }
1506                          /*
1507                              ASSETS ROOT
1508  
1509                              Finally, one last check to see if it is simply under assets/. This is useful for
1510                              keeping collections of scripts (say, TinyMCE or MarkItUp together for easy upgrade.
1511                          */
1512                          else if (($file_array = self::get_file_array($site_path, self::$asset_base . '/', $file, $type, $media)))
1513                          {
1514                              $new_files[] = $file_array;
1515                          }
1516                      }    // if (!$found)
1517                  }    // foreach ($paths as $path)
1518              }    // else
1519          }//end foreach
1520  
1521          return $new_files;
1522  
1523      }//end find_files()

1525      /**
1526       * Get the file array for a given file and path
1527       *
1528       * @access private
1529       * @static
1530       *
1531       * @param string $site_path The base server path
1532       * @param string $path      The path to the file (appended to base_url() and $site_path)
1533       * @param string $file      The name of the file, without the extension
1534       * @param string $type      File extension, including '.'
1535       * @param string $media     media type, e.g. 'screen' or 'print'
1536       *
1537       * @return mixed    false if the file wasn't found, the URL of the file if $media is empty,
1538       *                     or an array containing the file (URL), media, and server_path
1539       */
1540  	private static function get_file_array($site_path='', $path='', $file='', $type='', $media='')
1541      {
1542          if (empty($file) || empty($type))
1543          {
1544              return false;
1545          }
1546          $return = false;
1547          $rtl = self::$rtl_postfix;
1548          $file_name = $path . $file;
1549  
1550          if (is_file($site_path . $file_name . $type))
1551          {
1552              $file_path = base_url() . $file_name . $type;
1553              $server_path = $site_path . $file_name . $type;
1554              if (lang('bf_language_direction') == $rtl)
1555              {
1556                  if (is_file($site_path . $file_name . '-' . $rtl . $type))
1557                  {
1558                      $file_path = base_url() . $file_name . '-' . $rtl . $type;
1559                      $server_path = $site_path . $file_name . '-' . $rtl . $type;
1560                  }
1561              }
1562              $return = empty($media) ? $file_path : array('file' => $file_path, 'media' => $media, 'server_path' => $server_path);
1563  
1564              if (self::$debug) echo "[Assets] Found file at: <b>{$site_path}{$file_name}{$type}</b><br/>";
1565          }
1566          return $return;
1567      }
