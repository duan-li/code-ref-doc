 123      /**
 124       * Returns the HTML from the template based on the field passed in
 125       *
 126       * @access public
 127       * @static
 128       *
 129       * @param string $name       Name of the field
 130       * @param array  $properties Field settings
 131       *
 132       * @return string HTML for the required field
 133       */
 134  	public static function field($name, $properties=array())
 135      {
 136          if (!isset($properties['name']))
 137          {
 138              $properties['name'] = $name;
 139          }
 140  
 141          $input    = '';
 142          $error_class = '';
 143          $error = '';
 144          $help = '';
 145  
 146          if (isset($properties['help']))
 147          {
 148              $help = $properties['help'];
 149              unset($properties['help']);
 150          }
 151  
 152          switch ($properties['type'])
 153          {
 154              case 'hidden':
 155                  break;
 156              case 'radio':
 157              case 'checkbox':
 158                  break;
 159              case 'select':
 160                  break;
 161              case 'textarea':
 162                  break;
 163              case 'state':
 164                  $input = self::state($properties);
 165                  break;
 166              default:
 167                  $input = self::input($properties);
 168                  break;
 169          }
 170  
 171          $return = str_replace('{label}', self::label($properties['label']), self::$template);
 172          $return = str_replace('{input}', $input, $return);
 173          $return = str_replace('{help}', $help, $return);
 174          $return = str_replace('{error_class}', $error_class, $return);
 175          $return = str_replace('{error}', $error, $return);
 176  
 177          return $return;
 178  
 179      }//end field()

 335  	private static function attr_to_string($attr)
 336      {
 337          $attr_str = '';
 338  
 339          if ( ! is_array($attr))
 340          {
 341              $attr = (array) $attr;
 342          }
 343  
 344          foreach ($attr as $property => $value)
 345          {
 346              if ($property == 'label')
 347              {
 348                  continue;
 349              }
 350              if ($property == 'value')
 351              {
 352                  $value = self::prep_value($value);
 353              }
 354              $attr_str .= $property . '="' . $value . '" ';
 355          }
 356  
 357          // We strip off the last space for return
