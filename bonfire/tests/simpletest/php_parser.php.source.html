  21  /**
  22   *    Compounded regular expression. Any of
  23   *    the contained patterns could match and
  24   *    when one does, it's label is returned.
  25   *    @package SimpleTest
  26   *    @subpackage WebTester
  27   */
  28  class ParallelRegex {
  29      private $patterns;
  30      private $labels;
  31      private $regex;
  32      private $case;
  33  
  34      /**
  35       *    Constructor. Starts with no patterns.
  36       *    @param boolean $case    True for case sensitive, false
  37       *                            for insensitive.
  38       *    @access public
  39       */
  40      function __construct($case) {
  41          $this->case = $case;
  42          $this->patterns = array();
  43          $this->labels = array();
  44          $this->regex = null;
  45      }
  46  
  47      /**
  48       *    Adds a pattern with an optional label.
  49       *    @param string $pattern      Perl style regex, but ( and )
  50       *                                lose the usual meaning.
  51       *    @param string $label        Label of regex to be returned
  52       *                                on a match.
  53       *    @access public
  54       */
  55      function addPattern($pattern, $label = true) {
  56          $count = count($this->patterns);
  57          $this->patterns[$count] = $pattern;
  58          $this->labels[$count] = $label;
  59          $this->regex = null;
  60      }
  61  
  62      /**
  63       *    Attempts to match all patterns at once against
  64       *    a string.
  65       *    @param string $subject      String to match against.
  66       *    @param string $match        First matched portion of
  67       *                                subject.
  68       *    @return boolean             True on success.
  69       *    @access public
  70       */
  71      function match($subject, &$match) {
  72          if (count($this->patterns) == 0) {
  73              return false;
  74          }
  75          if (! preg_match($this->getCompoundedRegex(), $subject, $matches)) {
  76              $match = '';
  77              return false;
  78          }
  79          $match = $matches[0];
  80          for ($i = 1; $i < count($matches); $i++) {
  81              if ($matches[$i]) {
  82                  return $this->labels[$i - 1];
  83              }
  84          }
  85          return true;
  86      }
  87  
  88      /**
  89       *    Compounds the patterns into a single
  90       *    regular expression separated with the
  91       *    "or" operator. Caches the regex.
  92       *    Will automatically escape (, ) and / tokens.
  93       *    @param array $patterns    List of patterns in order.
  94       *    @access private
  95       */
  96      protected function getCompoundedRegex() {
  97          if ($this->regex == null) {
  98              for ($i = 0, $count = count($this->patterns); $i < $count; $i++) {
  99                  $this->patterns[$i] = '(' . str_replace(
 100                          array('/', '(', ')'),
 101                          array('\/', '\(', '\)'),
 102                          $this->patterns[$i]) . ')';
 103              }
 104              $this->regex = "/" . implode("|", $this->patterns) . "/" . $this->getPerlMatchingFlags();
 105          }
 106          return $this->regex;
 107      }
 108  
 109      /**
 110       *    Accessor for perl regex mode flags to use.
 111       *    @return string       Perl regex flags.
 112       *    @access private
 113       */
 114      protected function getPerlMatchingFlags() {
 115          return ($this->case ? "msS" : "msSi");
 116      }
 117  }

 180  class SimpleLexer {
 181      private $regexes;
 182      private $parser;
 183      private $mode;
 184      private $mode_handlers;
 185      private $case;
 186  
 187      /**
 188       *    Sets up the lexer in case insensitive matching
 189       *    by default.
 190       *    @param SimpleSaxParser $parser  Handling strategy by
 191       *                                    reference.
 192       *    @param string $start            Starting handler.
 193       *    @param boolean $case            True for case sensitive.
 194       *    @access public
 195       */
 196      function __construct($parser, $start = "accept", $case = false) {
 197          $this->case = $case;
 198          $this->regexes = array();
 199          $this->parser = $parser;
 200          $this->mode = new SimpleStateStack($start);
 201          $this->mode_handlers = array($start => $start);
 202      }
 203  
 204      /**
 205       *    Adds a token search pattern for a particular
 206       *    parsing mode. The pattern does not change the
 207       *    current mode.
 208       *    @param string $pattern      Perl style regex, but ( and )
 209       *                                lose the usual meaning.
 210       *    @param string $mode         Should only apply this
 211       *                                pattern when dealing with
 212       *                                this type of input.
 213       *    @access public
 214       */
 215      function addPattern($pattern, $mode = "accept") {
 216          if (! isset($this->regexes[$mode])) {
 217              $this->regexes[$mode] = new ParallelRegex($this->case);
 218          }
 219          $this->regexes[$mode]->addPattern($pattern);
 220          if (! isset($this->mode_handlers[$mode])) {
 221              $this->mode_handlers[$mode] = $mode;
 222          }
 223      }
 224  
 225      /**
 226       *    Adds a pattern that will enter a new parsing
 227       *    mode. Useful for entering parenthesis, strings,
 228       *    tags, etc.
 229       *    @param string $pattern      Perl style regex, but ( and )
 230       *                                lose the usual meaning.
 231       *    @param string $mode         Should only apply this
 232       *                                pattern when dealing with
 233       *                                this type of input.
 234       *    @param string $new_mode     Change parsing to this new
 235       *                                nested mode.
 236       *    @access public
 237       */
 238      function addEntryPattern($pattern, $mode, $new_mode) {
 239          if (! isset($this->regexes[$mode])) {
 240              $this->regexes[$mode] = new ParallelRegex($this->case);
 241          }
 242          $this->regexes[$mode]->addPattern($pattern, $new_mode);
 243          if (! isset($this->mode_handlers[$new_mode])) {
 244              $this->mode_handlers[$new_mode] = $new_mode;
 245          }
 246      }
 247  
 248      /**
 249       *    Adds a pattern that will exit the current mode
 250       *    and re-enter the previous one.
 251       *    @param string $pattern      Perl style regex, but ( and )
 252       *                                lose the usual meaning.
 253       *    @param string $mode         Mode to leave.
 254       *    @access public
 255       */
 256      function addExitPattern($pattern, $mode) {
 257          if (! isset($this->regexes[$mode])) {
 258              $this->regexes[$mode] = new ParallelRegex($this->case);
 259          }
 260          $this->regexes[$mode]->addPattern($pattern, "__exit");
 261          if (! isset($this->mode_handlers[$mode])) {
 262              $this->mode_handlers[$mode] = $mode;
 263          }
 264      }

 266      /**
 267       *    Adds a pattern that has a special mode. Acts as an entry
 268       *    and exit pattern in one go, effectively calling a special
 269       *    parser handler for this token only.
 270       *    @param string $pattern      Perl style regex, but ( and )
 271       *                                lose the usual meaning.
 272       *    @param string $mode         Should only apply this
 273       *                                pattern when dealing with
 274       *                                this type of input.
 275       *    @param string $special      Use this mode for this one token.
 276       *    @access public
 277       */
 278      function addSpecialPattern($pattern, $mode, $special) {
 279          if (! isset($this->regexes[$mode])) {
 280              $this->regexes[$mode] = new ParallelRegex($this->case);
 281          }
 282          $this->regexes[$mode]->addPattern($pattern, "_$special");
 283          if (! isset($this->mode_handlers[$special])) {
 284              $this->mode_handlers[$special] = $special;
 285          }
 286      }
 287  
 288      /**
 289       *    Adds a mapping from a mode to another handler.
 290       *    @param string $mode        Mode to be remapped.
 291       *    @param string $handler     New target handler.
 292       *    @access public
 293       */
 294      function mapHandler($mode, $handler) {
 295          $this->mode_handlers[$mode] = $handler;
 296      }
 297  
 298      /**
 299       *    Splits the page text into tokens. Will fail
 300       *    if the handlers report an error or if no
 301       *    content is consumed. If successful then each
 302       *    unparsed and parsed token invokes a call to the
 303       *    held listener.
 304       *    @param string $raw        Raw HTML text.
 305       *    @return boolean           True on success, else false.
 306       *    @access public
 307       */
 308      function parse($raw) {
 309          if (! isset($this->parser)) {
 310              return false;
 311          }
 312          $length = strlen($raw);
 313          while (is_array($parsed = $this->reduce($raw))) {
 314              list($raw, $unmatched, $matched, $mode) = $parsed;
 315              if (! $this->dispatchTokens($unmatched, $matched, $mode)) {
 316                  return false;
 317              }
 318              if ($raw === '') {
 319                  return true;
 320              }
 321              if (strlen($raw) == $length) {
 322                  return false;
 323              }
 324              $length = strlen($raw);
 325          }
 326          if (! $parsed) {
 327              return false;
 328          }
 329          return $this->invokeParser($raw, LEXER_UNMATCHED);
 330      }
 331  
 332      /**
 333       *    Sends the matched token and any leading unmatched
 334       *    text to the parser changing the lexer to a new
 335       *    mode if one is listed.
 336       *    @param string $unmatched    Unmatched leading portion.
 337       *    @param string $matched      Actual token match.
 338       *    @param string $mode         Mode after match. A boolean
 339       *                                false mode causes no change.
 340       *    @return boolean             False if there was any error
 341       *                                from the parser.
 342       *    @access private
 343       */
 344      protected function dispatchTokens($unmatched, $matched, $mode = false) {
 345          if (! $this->invokeParser($unmatched, LEXER_UNMATCHED)) {
 346              return false;
 347          }
 348          if (is_bool($mode)) {
 349              return $this->invokeParser($matched, LEXER_MATCHED);
 350          }
 351          if ($this->isModeEnd($mode)) {
 352              if (! $this->invokeParser($matched, LEXER_EXIT)) {
 353                  return false;
 354              }
 355              return $this->mode->leave();
 356          }
 357          if ($this->isSpecialMode($mode)) {
 358              $this->mode->enter($this->decodeSpecial($mode));
 359              if (! $this->invokeParser($matched, LEXER_SPECIAL)) {
 360                  return false;
 361              }
 362              return $this->mode->leave();
 363          }
 364          $this->mode->enter($mode);
 365          return $this->invokeParser($matched, LEXER_ENTER);
 366      }
 367  
 368      /**
 369       *    Tests to see if the new mode is actually to leave
 370       *    the current mode and pop an item from the matching
 371       *    mode stack.
 372       *    @param string $mode    Mode to test.
 373       *    @return boolean        True if this is the exit mode.
 374       *    @access private
 375       */
 376      protected function isModeEnd($mode) {
 377          return ($mode === "__exit");
 378      }
 379  
 380      /**
 381       *    Test to see if the mode is one where this mode
 382       *    is entered for this token only and automatically
 383       *    leaves immediately afterwoods.
 384       *    @param string $mode    Mode to test.
 385       *    @return boolean        True if this is the exit mode.
 386       *    @access private
 387       */
 388      protected function isSpecialMode($mode) {
 389          return (strncmp($mode, "_", 1) == 0);
 390      }
 391  
 392      /**
 393       *    Strips the magic underscore marking single token
 394       *    modes.
 395       *    @param string $mode    Mode to decode.
 396       *    @return string         Underlying mode name.
 397       *    @access private
 398       */
 399      protected function decodeSpecial($mode) {
 400          return substr($mode, 1);
 401      }
 402  
 403      /**
 404       *    Calls the parser method named after the current
 405       *    mode. Empty content will be ignored. The lexer
 406       *    has a parser handler for each mode in the lexer.
 407       *    @param string $content        Text parsed.
 408       *    @param boolean $is_match      Token is recognised rather
 409       *                                  than unparsed data.
 410       *    @access private
 411       */
 412      protected function invokeParser($content, $is_match) {
 413          if (($content === '') || ($content === false)) {
 414              return true;
 415          }
 416          $handler = $this->mode_handlers[$this->mode->getCurrent()];
 417          return $this->parser->$handler($content, $is_match);
 418      }
 419  
 420      /**
 421       *    Tries to match a chunk of text and if successful
 422       *    removes the recognised chunk and any leading
 423       *    unparsed data. Empty strings will not be matched.
 424       *    @param string $raw         The subject to parse. This is the
 425       *                               content that will be eaten.
 426       *    @return array/boolean      Three item list of unparsed
 427       *                               content followed by the
 428       *                               recognised token and finally the
 429       *                               action the parser is to take.
 430       *                               True if no match, false if there
 431       *                               is a parsing error.
 432       *    @access private
 433       */
 434      protected function reduce($raw) {
 435          if ($action = $this->regexes[$this->mode->getCurrent()]->match($raw, $match)) {
 436              $unparsed_character_count = strpos($raw, $match);
 437              $unparsed = substr($raw, 0, $unparsed_character_count);
 438              $raw = substr($raw, $unparsed_character_count + strlen($match));
 439              return array($raw, $unparsed, $match, $action);
 440          }
 441          return true;
 442      }
 443  }

 533          $this->mapHandler('uq_attribute', 'acceptAttributeToken');
 534          $this->addSpecialPattern('=\s*[^>\s]*', 'tag', 'uq_attribute');

 584      /**
 585       *    Accepts a token from the tag mode. If the
 586       *    starting element completes then the element
 587       *    is dispatched and the current attributes
 588       *    set back to empty. The element or attribute
 589       *    name is converted to lower case.
 590       *    @param string $token     Incoming characters.
 591       *    @param integer $event    Lexer event type.
 592       *    @return boolean          False if parse error.
 593       *    @access public
 594       */
 595      function acceptStartToken($token, $event) {
 596          if ($event == LEXER_ENTER) {
 597              $this->tag = strtolower(substr($token, 1));
 598              return true;
 599          }
 600          if ($event == LEXER_EXIT) {
 601              $success = $this->listener->startElement(
 602                      $this->tag,
 603                      $this->attributes);
 604              $this->tag = '';
 605              $this->attributes = array();
 606              return $success;
 607          }
 608          if ($token != '=') {
 609              $this->current_attribute = strtolower(html_entity_decode($token, ENT_QUOTES));
 610              $this->attributes[$this->current_attribute] = '';
 611          }
 612          return true;
 613      }
 614  
 615      /**
 616       *    Accepts a token from the end tag mode.
 617       *    The element name is converted to lower case.
 618       *    @param string $token     Incoming characters.
 619       *    @param integer $event    Lexer event type.
 620       *    @return boolean          False if parse error.
 621       *    @access public
 622       */
 623      function acceptEndToken($token, $event) {
 624          if (! preg_match('/<\/(.*)>/', $token, $matches)) {
 625              return false;
 626          }
 627          return $this->listener->endElement(strtolower($matches[1]));
 628      }

 630      /**
 631       *    Part of the tag data.
 632       *    @param string $token     Incoming characters.
 633       *    @param integer $event    Lexer event type.
 634       *    @return boolean          False if parse error.
 635       *    @access public
 636       */
 637      function acceptAttributeToken($token, $event) {
 638          if ($this->current_attribute) {
 639              if ($event == LEXER_UNMATCHED) {
 640                  $this->attributes[$this->current_attribute] .=
 641                          html_entity_decode($token, ENT_QUOTES);
 642              }
 643              if ($event == LEXER_SPECIAL) {
 644                  $this->attributes[$this->current_attribute] .=
 645                          preg_replace('/^=\s*/' , '', html_entity_decode($token, ENT_QUOTES));
 646              }
 647          }
 648          return true;
 649      }

 765      /**
 766       *    Start of element event. Opens a new tag.
 767       *    @param string $name         Element name.
 768       *    @param hash $attributes     Attributes without content
 769       *                                are marked as true.
 770       *    @return boolean             False on parse error.
 771       *    @access public
 772       */
 773      function startElement($name, $attributes) {
 774          $factory = new SimpleTagBuilder();
 775          $tag = $factory->createTag($name, $attributes);
 776          if (! $tag) {
 777              return true;
 778          }
 779          if ($tag->getTagName() == 'label') {
 780              $this->acceptLabelStart($tag);
 781              $this->openTag($tag);
 782              return true;
 783          }
 784          if ($tag->getTagName() == 'form') {
 785              $this->acceptFormStart($tag);
 786              return true;
 787          }
 788          if ($tag->getTagName() == 'frameset') {
 789              $this->acceptFramesetStart($tag);
 790              return true;
 791          }
 792          if ($tag->getTagName() == 'frame') {
 793              $this->acceptFrame($tag);
 794              return true;
 795          }
 796          if ($tag->isPrivateContent() && ! isset($this->private_content_tag)) {
 797              $this->private_content_tag = &$tag;
 798          }
 799          if ($tag->expectEndTag()) {
 800              $this->openTag($tag);
 801              return true;
 802          }
 803          $this->acceptTag($tag);
 804          return true;
 805      }

 807      /**
 808       *    End of element event.
 809       *    @param string $name        Element name.
 810       *    @return boolean            False on parse error.
 811       *    @access public
 812       */
 813      function endElement($name) {
 814          if ($name == 'label') {
 815              $this->acceptLabelEnd();
 816              return true;
 817          }
 818          if ($name == 'form') {
 819              $this->acceptFormEnd();
 820              return true;
 821          }
 822          if ($name == 'frameset') {
 823              $this->acceptFramesetEnd();
 824              return true;
 825          }
 826          if ($this->hasNamedTagOnOpenTagStack($name)) {
 827              $tag = array_pop($this->tags[$name]);
 828              if ($tag->isPrivateContent() && $this->private_content_tag->getTagName() == $name) {
 829                  unset($this->private_content_tag);
 830              }
 831              $this->addContentTagToOpenTags($tag);
 832              $this->acceptTag($tag);
 833              return true;
 834          }
 835          return true;
 836      }

 913       *    @param SimpleTag $tag        Tag to accept.
 914       *    @access public

 916      protected function acceptTag($tag) {
 917          if ($tag->getTagName() == "a") {
 918              $this->page->addLink($tag);
 919          } elseif ($tag->getTagName() == "base") {
 920              $this->page->setBase($tag->getAttribute('href'));
 921          } elseif ($tag->getTagName() == "title") {
 922              $this->page->setTitle($tag);

 924              for ($i = 0; $i < count($this->open_forms); $i++) {
 925                  $this->open_forms[$i]->addWidget($tag);
