 257      private function getPageMethods() {
 258          $methods = array();
 259          foreach (get_class_methods('SimplePage') as $method) {
 260              if (strtolower($method) == strtolower('SimplePage')) {
 261                  continue;
 262              }
 263              if (strtolower($method) == strtolower('getFrameset')) {
 264                  continue;
 265              }
 266              if (strncmp($method, '_', 1) == 0) {
 267                  continue;
 268              }
 269              if (in_array($method, array('setTitle', 'setBase', 'setForms', 'normalise', 'setFrames', 'addLink'))) {
 270                  continue;
 271              }
 272              $methods[] = $method;
