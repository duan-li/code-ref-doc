 118      /**
 119       *    Uses reflection to run every method within itself
 120       *    starting with the string "test" unless a method
 121       *    is specified.
 122       *    @param SimpleReporter $reporter    Current test reporter.
 123       *    @return boolean                    True if all tests passed.
 124       *    @access public
 125       */
 126      function run($reporter) {
 127          $context = SimpleTest::getContext();
 128          $context->setTest($this);
 129          $context->setReporter($reporter);
 130          $this->reporter = $reporter;
 131          $started = false;
 132          foreach ($this->getTests() as $method) {
 133              if ($reporter->shouldInvoke($this->getLabel(), $method)) {
 134                  $this->skip();
 135                  if ($this->should_skip) {
 136                      break;
 137                  }
 138                  if (! $started) {
 139                      $reporter->paintCaseStart($this->getLabel());
 140                      $started = true;
 141                  }
 142                  $invoker = $this->reporter->createInvoker($this->createInvoker());
 143                  $invoker->before($method);
 144                  $invoker->invoke($method);
 145                  $invoker->after($method);
 146              }
 147          }
 148          if ($started) {
 149              $reporter->paintCaseEnd($this->getLabel());
 150          }
 151          unset($this->reporter);
 152          $context->setTest(null);
 153          return $reporter->getStatus();
 154      }

 364  /**
 365   *  Helps to extract test cases automatically from a file.
 366   *    @package      SimpleTest
 367   *    @subpackage   UnitTester
 368   */
 369  class SimpleFileLoader {
 370  
 371      /**
 372       *    Builds a test suite from a library of test cases.
 373       *    The new suite is composed into this one.
 374       *    @param string $test_file        File name of library with
 375       *                                    test case classes.
 376       *    @return TestSuite               The new test suite.
 377       *    @access public
 378       */
 379      function load($test_file) {
 380          $existing_classes = get_declared_classes();
 381          $existing_globals = get_defined_vars();
 382          include_once($test_file);
 383          $new_globals = get_defined_vars();
 384          $this->makeFileVariablesGlobal($existing_globals, $new_globals);
 385          $new_classes = array_diff(get_declared_classes(), $existing_classes);
 386          if (empty($new_classes)) {
 387              $new_classes = $this->scrapeClassesFromFile($test_file);
 388          }
 389          $classes = $this->selectRunnableTests($new_classes);
 390          return $this->createSuiteFromClasses($test_file, $classes);
 391      }
 392  
 393      /**
 394       *    Imports new variables into the global namespace.
 395       *    @param hash $existing   Variables before the file was loaded.
 396       *    @param hash $new        Variables after the file was loaded.
 397       *    @access private
 398       */
 399      protected function makeFileVariablesGlobal($existing, $new) {
 400          $globals = array_diff(array_keys($new), array_keys($existing));
 401          foreach ($globals as $global) {
 402              $GLOBALS[$global] = $new[$global];
 403          }
 404      }
 405  
 406      /**
 407       *    Lookup classnames from file contents, in case the
 408       *    file may have been included before.
 409       *    Note: This is probably too clever by half. Figuring this
 410       *    out after a failed test case is going to be tricky for us,
 411       *    never mind the user. A test case should not be included
 412       *    twice anyway.
 413       *    @param string $test_file        File name with classes.
 414       *    @access private
 415       */
 416      protected function scrapeClassesFromFile($test_file) {
 417          preg_match_all('~^\s*class\s+(\w+)(\s+(extends|implements)\s+\w+)*\s*\{~mi',
 418                          file_get_contents($test_file),
 419                          $matches );
 420          return $matches[1];
 421      }
 422  
 423      /**
 424       *    Calculates the incoming test cases. Skips abstract
 425       *    and ignored classes.
 426       *    @param array $candidates   Candidate classes.
 427       *    @return array              New classes which are test
 428       *                               cases that shouldn't be ignored.
 429       *    @access public
 430       */
 431      function selectRunnableTests($candidates) {
 432          $classes = array();
 433          foreach ($candidates as $class) {
 434              if (TestSuite::getBaseTestCase($class)) {
 435                  $reflection = new SimpleReflection($class);
 436                  if ($reflection->isAbstract()) {
 437                      SimpleTest::ignore($class);
 438                  } else {
 439                      $classes[] = $class;
 440                  }
 441              }
 442          }
 443          return $classes;
 444      }

 446      /**
 447       *    Builds a test suite from a class list.
 448       *    @param string $title       Title of new group.
 449       *    @param array $classes      Test classes.
 450       *    @return TestSuite          Group loaded with the new
 451       *                               test cases.
 452       *    @access public
 453       */
 454      function createSuiteFromClasses($title, $classes) {
 455          if (count($classes) == 0) {
 456              $suite = new BadTestSuite($title, "No runnable test cases in [$title]");
 457              return $suite;
 458          }
 459          SimpleTest::ignoreParentsIfIgnored($classes);
 460          $suite = new TestSuite($title);
 461          foreach ($classes as $class) {
 462              if (! SimpleTest::isIgnored($class)) {
 463                  $suite->add($class);
 464              }
 465          }
 466          return $suite;
 467      }

 516          if (! is_string($test_case)) {

 548      /**
 549       *    Invokes run() on all of the held test cases, instantiating
 550       *    them if necessary.
 551       *    @param SimpleReporter $reporter    Current test reporter.
 552       *    @access public
 553       */
 554      function run($reporter) {
 555          $reporter->paintGroupStart($this->getLabel(), $this->getSize());
 556          for ($i = 0, $count = count($this->test_cases); $i < $count; $i++) {
 557              if (is_string($this->test_cases[$i])) {
 558                  $class = $this->test_cases[$i];
 559                  $test = new $class();
 560                  $test->run($reporter);
 561                  unset($test);
 562              } else {
 563                  $this->test_cases[$i]->run($reporter);
 564              }
 565          }
 566          $reporter->paintGroupEnd($this->getLabel());
 567          return $reporter->getStatus();
 568      }

 570      /**
 571       *    Number of contained test cases.
 572       *    @return integer     Total count of cases in the group.
 573       *    @access public
 574       */
 575      function getSize() {
 576          $count = 0;
 577          foreach ($this->test_cases as $case) {
 578              if (is_string($case)) {
 579                  if (! SimpleTest::isIgnored($case)) {
 580                      $count++;
 581                  }
 582              } else {
 583                  $count += $case->getSize();
 584              }
 585          }
 586          return $count;
 587      }

 589      /**
 590       *    Test to see if a class is derived from the
 591       *    SimpleTestCase class.
 592       *    @param string $class     Class name.
 593       *    @access public
 594       */
 595      static function getBaseTestCase($class) {
 596          while ($class = get_parent_class($class)) {
 597              $class = strtolower($class);
 598              if ($class == 'simpletestcase' || $class == 'testsuite') {
 599                  return $class;
 600              }
 601          }
 602          return false;
