  50          $periods = array("moment", "min", "hour", "day", "week",
  51          "month", "year", "decade");
  52  
  53          $lengths = array("60","60","24","7","4.35","12","10", "10");
  54  
  55          if ($difference >= 0)
  56          {
  57              // this was in the past
  58              $ending = "ago";

  68          {

  73  
  74          if ($difference != 1)

  76              $periods[$j].= "s";

  97      /**
  98       * Returns the difference between two dates.
  99       *
 100       * @param $start mixed The start date in either unix timestamp or a format that can be used within strtotime().
 101       * @param $end mixed The ending date in either unix timestamp or a format that can be used within strtotime().
 102       * @param $interval string A string with the interval to use. Choices 'week', 'day', 'hour', 'minute'.
 103       * @param $reformat boolean If TRUE, will reformat the time using strtotime().
 104       *
 105       * @return int A number representing the difference between the two dates in the interval desired.
 106       */
 107  	function date_difference($start=null, $end=null, $interval='day', $reformat=false)
 108      {
 109          if (is_null($start))
 110          {
 111              return false;
 112          }
 113  
 114          if (is_null($end))
 115          {
 116              $end = date('Y-m-d H:i:s');
 117          }
 118  
 119          $times = array(
 120              'week'        => 604800,
 121              'day'        => 86400,
 122              'hour'        => 3600,
 123              'minute'    => 60
 124          );
 125  
 126          if ($reformat === true)
 127          {
 128              $start     = strtotime($start);
 129              $end    = strtotime($end);
 130          }
 131  
 132          $diff = $end - $start;
 133  
 134          return round($diff / $times[$interval]);
 135  
 136      }//end date_difference()

 150       * @example echo user_time();
 151       * @example echo user_time($timestamp, 'EET', 'l jS \of F Y h:i:s A');
 152       *
 153       * @param int    $timestamp A UNIX timestamp. If non is given current time will be used.
 154       * @param string $timezone  The timezone we want to convert to. If none is given a current logged user timezone will be used.
 155       * @param string $format    The format of the outputted date/time string
 156       *
 157       * @return string A string formatted according to the given format using the given timestamp and given timezone or the current time if no timestamp is given.
 158       */
 159  	function user_time($timestamp = NULL, $timezone = NULL, $format = 'r')
 160      {
 161          if ( ! $timezone)
 162          {
 163              $CI =& get_instance();
 164              $CI->load->library('users/auth');
 165              if ($CI->auth->is_logged_in())
 166              {
 167                  $timezone = standard_timezone($CI->auth->user()->timezone);
 168              }
 169          }
 170  
 171          $timestamp = ($timestamp) ? $timestamp : time();
 172  
 173          $dtzone = new DateTimeZone($timezone);
 174          $dtime = new DateTime();
 175  
 176          return $dtime->setTimestamp($timestamp)->setTimeZone($dtzone)->format($format);
 177  
 178      }//end user_time()
