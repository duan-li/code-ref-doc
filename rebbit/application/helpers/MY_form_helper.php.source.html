  41  {
  42      /**
  43       * Used by many of the new functions to wrap the input in the correct
  44       * tags so that the styling is automatic.
  45       *
  46       * @access private
  47       *
  48       * @param string $type    A string with the name of the element type.
  49       * @param string $data    Either a string with the element name, or an array of key/value pairs of all attributes.
  50       * @param string $value   Either a string with the value, or blank if an array is passed to the $data param.
  51       * @param string $label   A string with the label of the element.
  52       * @param string $extra   A string with any additional items to include, like Javascript.
  53       * @param string $tooltip A string for inline help or a tooltip icon
  54       *
  55       * @return string A string with the formatted input element, label tag and wrapping divs.
  56       */
  57  	function _form_common($type='text', $data='', $value='', $label='', $extra='', $tooltip = '')
  58      {
  59          $defaults = array('type' => $type, 'name' => (( ! is_array($data)) ? $data : ''), 'value' => $value);
  60  
  61          // If name is empty at this point, try to grab it from the $data array
  62          if (empty($defaults['name']) && is_array($data) && isset($data['name']))
  63          {
  64              $defaults['name'] = $data['name'];
  65              unset($data['name']);
  66          }
  67  
  68          // If label is empty at this point, try to grab it from the $data array
  69          if (empty($label) && is_array($data) && isset($data['label']))
  70          {
  71              $label = $data['label'];
  72              unset($data['label']);
  73          }
  74  
  75          // If tooltip is empty at this point, try to grab it from the $data array
  76          if (empty($tooltip) && is_array($data) && isset($data['tooltip']))
  77          {
  78              $tooltip = $data['tooltip'];
  79              unset($data['tooltip']);
  80          }
  81  
  82          $error = '';
  83  
  84          if (function_exists('form_error'))
  85          {
  86              if (form_error($defaults['name']))
  87              {
  88                  $error   = ' error';
  89                  $tooltip = '<span class="help-inline">' . form_error($defaults['name']) . '</span>' . PHP_EOL;
  90              }
  91          }
  92  
  93          $output = _parse_form_attributes($data, $defaults);
  94  
  95          $output = <<<EOL
  96  
  97  <div class="control-group {$error}">
  98      <label class="control-label" for="{$defaults['name']}">{$label}</label>
  99      <div class="controls">
 100           <input {$output} {$extra} />
 101          {$tooltip}
 102      </div>
 103  </div>
 104  
 105  EOL;
 106  
 107          return $output;
 108  
 109      }//end _form_common()
 110  }

 391  {
 392      /**
 393       * Returns a properly templated dropdown field.
 394       *
 395       * @param string $data     Either a string with the element name, or an array of key/value pairs of all attributes, which must include a name or id.
 396       * @param array  $options  Array of options for the drop down list
 397       * @param string $selected Either a string of the selected item or an array of selected items
 398       * @param string $label    A string with the label of the element.
 399       * @param string $extra    A string with any additional items to include, like Javascript.
 400       * @param string $tooltip  A string for inline help or a tooltip icon
 401       *
 402       * @return string A string with the formatted input element, label tag and wrapping divs.
 403       */
 404  	function form_dropdown($data, $options=array(), $selected=array(), $label='', $extra='', $tooltip = '')
 405      {
 406          if (! is_array($data))
 407          {
 408              $data = array('name' => $data);
 409          }
 410  
 411          if (! isset($data['id']))
 412          {
 413              $data['id'] = $data['name'];
 414          }
 415  
 416          $output = _parse_form_attributes($data, array());
 417  
 418          if ( ! is_array($selected))
 419          {
 420              $selected = array($selected);
 421          }
 422  
 423          // If no selected state was submitted we will attempt to set it automatically
 424          if (count($selected) === 0)
 425          {
 426              // If the form name appears in the $_POST array we have a winner!
 427              if (isset($_POST[$data['name']]))
 428              {
 429                  $selected = array($_POST[$data['name']]);
 430              }
 431          }
 432  
 433          $options_vals = '';
 434          foreach ($options as $key => $val)
 435          {
 436              $key = (string) $key;
 437  
 438              if (is_array($val) && ! empty($val))
 439              {
 440                  $options_vals .= '<optgroup label="'.$key.'">'.PHP_EOL;
 441  
 442                  foreach ($val as $optgroup_key => $optgroup_val)
 443                  {
 444                      $sel = (in_array($optgroup_key, $selected)) ? ' selected="selected"' : '';
 445  
 446                      $options_vals .= '<option value="'.$optgroup_key.'"'.$sel.'>'.(string) $optgroup_val."</option>\n";
 447                  }
 448  
 449                  $options_vals .= '</optgroup>'.PHP_EOL;
 450              }
 451              else
 452              {
 453                  $sel = (in_array($key, $selected)) ? ' selected="selected"' : '';
 454  
 455                  $options_vals .= '<option value="'.$key.'"'.$sel.'>'.(string) $val."</option>\n";
 456              }
 457          }
 458  
 459          $error = '';
 460  
 461          if (function_exists('form_error'))
 462          {
 463              if (form_error($data['name']))
 464              {
 465                  $error   = ' error';
 466                  $tooltip = '<span class="help-inline">' . form_error($data['name']) . '</span>' . PHP_EOL;
 467              }
 468          }
 469  
 470          $output = <<<EOL
 471  
 472  <div class="control-group {$error}">
 473      <label class="control-label" for="{$data['id']}">{$label}</label>
 474      <div class="controls">
 475           <select {$output} {$extra}>
 476              {$options_vals}
 477          </select>
 478          {$tooltip}
 479      </div>
 480  </div>
 481  
 482  EOL;
 483  
 484          return $output;
 485  
 486      }//end form_dropdown()
 487  }
