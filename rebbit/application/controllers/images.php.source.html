  24   *
  25   * http://mysite.com/images/abc.jpg?assets=images&size=xx
  26   *

  28   * The following parameters are available for use in the params

  31   * assets    - The folder (relative to webroot) that the images can be found in.
  32   *               Optional. Defaults to '/assets'.

  35   * height    - The height (in px) of the image. Not required if 'size' is passed.
  36   * width    - The width (in px) of the image. Not required if 'size' is passed.
  37   * crop    - If 'yes', will crop the image to the size or height/width.

  41   * @category   Helpers

  84  
  85          if (empty($ext))

  87              die('Filename does not include a file extension.');

  89  
  90          // Is it a square?
  91          if (!empty($size))
  92          {
  93              $height = (int)$size;
  94              $width    = (int)$size;
  95              $ratio     = 'no';
  96          }
  97  
  98          // For now, simply return the file....
  99          $img_file = FCPATH . $assets .'/'. $file;
 100  
 101          if (!is_file($img_file))
 102          {
 103              die('Image could not be found.');
 104          }
 105  
 106          $new_file = FCPATH .'assets/cache/'. str_replace('.'. $ext, '', $file) ."_{$width}x{$height}.". $ext;
 107  
 108          if (!is_file($new_file) || $force != 'yes')
 109          {
 110              $config = array(
 111                  'image_library'        => 'gd2',
 112                  'source_image'        => $img_file,
 113                  'new_image'            => $new_file,
 114                  'create_thumb'        => false,
 115                  'maintain_ratio'    => $ratio == 'no' ? false : true,
 116                  'master_dim'        => !empty($width) ? 'width' : 'height', 
 117                  'width'                => !empty($width) ? $width : $height,
 118                  'height'            => !empty($height) ? $height : $width,
 119              );
 120  
 121              $this->load->library('image_lib', $config);
 122  
 123              $this->image_lib->resize();
 124          }
 125  
 126          $this->output
 127              ->set_content_type($ext)
 128              ->set_output(file_get_contents($new_file));
 129      }//end index()
