 305      // set it manually to our tests root
 306      if (is_null($start_folder))

 319

 326              return $files;
 327          }
 328      }
 329  
 330      if (!function_exists('directory_map'))
 331      {
 332          global $ci;
 333          $ci->load->helper('directory');
 334      }
 335  
 336      $folders = directory_map($start_folder, 1);
 337  
 338      // Look through all of our folders & files for
 339      // valid test files.
 340      if (is_array($folders))
 341      {
 342          foreach ($folders as $folder)
 343          {
 344              // Folders get ran back through this function
 345              // and will return an array of valid files.
 346              // Merge this array with ours and call it good.
 347              if (is_dir($start_folder .'/'. $folder) && !in_array($start_folder .'/'. $folder, $ignore_folders))
 348              {
 349                  $f = discover_tests($start_folder .'/'. $folder, true);
 350                  $files = array_merge($files, $f);
 351              }
 352              else if (is_file($start_folder .'/'. $folder))
 353              {
 354                  // Does it appear to be a valid filename?
 355                  if (strpos($folder, 'Test')  OR strpos($folder, '_test'))
 356                  {
 357                      $files[] = $start_folder .'/'. $folder;
 358                  }
 359              }
 360          }
 361      }
 362  
 363      return $files;
 364  }
