 131       *    Send the headers necessary to ensure the page is

 134       *    @access public
 135       *    @static
 136       */
 137      function sendNoCacheHeaders()
 138      {
 139          if (! headers_sent()) {
 140              header("Expires: Mon, 26 Jul 1997 05:00:00 GMT");
 141              header("Last-Modified: " . gmdate("D, d M Y H:i:s") . " GMT");
 142              header("Cache-Control: no-store, no-cache, must-revalidate");
 143              header("Cache-Control: post-check=0, pre-check=0", false);
 144              header("Pragma: no-cache");
 145          }
