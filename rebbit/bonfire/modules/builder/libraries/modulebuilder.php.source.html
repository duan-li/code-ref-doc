 617       * @param string $table_name        The name of the table in the database

 621      private function build_db_sql($field_total, $module_name, $primary_key_field, $table_name, $table_as_field_prefix)

 624          {
 625              return FALSE;
 626          }

 652       * @param int    $fieldno The position number of this field
 653       *
 654       * @return bool

 656      protected function no_match($str, $fieldno)
 657      {
 658          for($counter=1; $this->field_total >= $counter; $counter++)
 659          {
 660              // nothing has been entered into this field so we don't need to check
 661              // or the field being checked is the same as the field we are checking from
 662              if ($_POST["view_field_name$counter"] == '' || $fieldno == $counter)
 663              {
 664                  continue;
 665              }
 666  
 667              if ($str == $_POST["view_field_name$counter"])
 668              {
 669                  $this->CI->form_validation->set_message('no_match', "Field names must be unique!");
 670                  return FALSE;
 671              }
 672          }
 673  
 674          return TRUE;
