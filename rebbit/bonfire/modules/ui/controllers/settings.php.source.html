 118      /**
 119       * Add a shortcut key for an option
 120       *
 121       * @access private
 122       *
 123       * @return bool
 124       */
 125  	private function add()
 126      {
 127  
 128          $this->form_validation->set_rules('new_action', lang('ui_actions'), 'required');
 129          $this->form_validation->set_rules('new_shortcut', lang('ui_shortcuts'), 'required|callback__validate_shortcuts');
 130  
 131          if ($this->form_validation->run() === FALSE)
 132          {
 133              return FALSE;
 134          }
 135  
 136          $action   = $this->input->post('new_action');
 137          $shortcut = $this->input->post('new_shortcut');
 138  
 139          // Read available shortcuts from the application config
 140          $available_actions = config_item('ui.current_shortcuts');
 141  
 142          if (array_key_exists($action, $available_actions))
 143          {
 144              return $this->save_settings(array($action => $shortcut));
 145          }
 146  
 147          return FALSE;
 148  
 149      }//end add()

 154      /**
 155       * Remove a shortcut key
 156       *
 157       * @access private
 158       *
 159       * @return bool
 160       */
 161  	private function remove()
 162      {
 163          $this->form_validation->set_rules('remove_shortcut[]', lang('ui_actions'), 'required|is_array');
 164  
 165          if ($this->form_validation->run() === FALSE)
 166          {
 167              return FALSE;
 168          }
 169  
 170          $action   = key($this->input->post('remove_shortcut'));
 171  
 172          // Read our current settings
 173          $available_actions = $this->settings_lib->find_all_by('module', 'core.ui');
 174  
 175          if (array_key_exists($action, $available_actions))
 176          {
 177              return $this->settings_lib->delete($action, 'core.ui');
 178          }
 179  
 180          return FALSE;
 181  
 182      }//end remove()

 186      /**
 187       * Save multiple shortcut keys at the same time allowing the user to
 188       * edit the settings
 189       *
 190       * @param array $settings Array of shortcuts
 191       *
 192       * @return bool
 193       */
 194  	private function save_settings($settings = array())
 195      {
 196          if (empty($settings))
 197          {
 198              // Read available shortcuts from the application config
 199              $available_actions = config_item('ui.current_shortcuts');
 200  
 201              // We can't use an array like the remove buttons do
 202              // because the text inputs need set_value().
 203              // set_value("shortcut[$action]") is not supported
 204              foreach ($available_actions as $action => $shortcut)
 205              {
 206                  if (isset($_POST["shortcut_$action"]))
 207                  {
 208                      $this->form_validation->set_rules("shortcut_$action", lang('ui_shortcuts'), 'required|callback__validate_shortcuts');
 209  
 210                      $settings[$action] = $this->input->post("shortcut_$action");
 211                  }
 212              }
 213  
 214              if ($this->form_validation->run() === FALSE)
 215              {
 216                  return FALSE;
 217              }
 218  
 219          }//end if
 220  
 221          if (is_array($settings))
 222          {
 223              foreach($settings as $action => $shortcut)
 224              {
 225                  $updated = $this->settings_lib->set($action, $shortcut, 'core.ui');
 226              }
 227          }
 228  
 229          // Log the activity
 230          $this->activity_model->log_activity($this->current_user->id, lang('bf_act_settings_saved').': ' . $this->input->ip_address(), 'ui');
 231  
 232          return $updated;
 233  
 234      }//end save_settings()
