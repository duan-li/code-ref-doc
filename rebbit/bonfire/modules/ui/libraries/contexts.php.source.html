 263      {
 264          self::$ci->benchmark->mark('context_menu_start');

 266          $contexts = self::$contexts;

 268          if (empty($contexts) || !is_array($contexts) || !count($contexts))
 269          {
 270              die(lang('bf_no_contexts'));
 271          }
 272  
 273          // Ensure settings context exists
 274          if (!in_array('settings', $contexts))
 275          {
 276              array_push($contexts, 'settings');

 278

 280          if (!in_array('developer', $contexts))

 285          // Sorting
 286          switch ($order_by)

 297              case 'normal':

 299                  break;
 300          }
 301  
 302          $nav_id = ( trim (self::$outer_id) != '' ) ? ' id="'. self::$outer_id . '"' : '';
 303          $nav = '<ul class="'. self::$outer_class .'" ' . $nav_id . ' >';
 304  
 305          /*
 306              Build out our navigation.
 307          */
 308          foreach ($contexts as $context)
 309          {
 310              if ( has_permission('Site.'. ucfirst($context) .'.View') == true || permission_exists('Site.'. ucfirst($context) .'.View') == false)
 311              {
 312                  $url = site_url(self::$site_area .'/'.$context);
 313                  $class = check_class($context, true);
 314                  $id = 'tb_'. $context;
 315  
 316                  if (lang('bf_context_'. $context))
 317                  {
 318                      $title = lang('bf_context_'. $context);

 327                  // Image

 329                  {

 390       * Builds the main navigation menu for each context.

 394       * @param string $context   The context to build the nav for.

 398       * @return string The HTML necessary to display the menu.

 400  	public static function context_nav($context=NULL, $class='dropdown-menu', $ignore_ul=FALSE)
 401      {
 402          // Get a list of modules with a controller matching
 403          // $context ('content', 'settings', 'reports', or 'developer')
 404          $module_list = module_list();
 405  
 406          foreach ($module_list as $module)
 407          {
 408              if (module_controller_exists($context, $module) === TRUE)
 409              {
 410                  $mod_config = module_config($module);
 411  
 412                  self::$actions[$module] = array(
 413                      'weight'        => isset($mod_config['weights'][$context]) ? $mod_config['weights'][$context] : 0,
 414                      'display_name'    => isset($mod_config['name']) ? $mod_config['name'] : $module,
 415                      'title'         => isset($mod_config['description']) ? $mod_config['description'] : $module,
 416                      'menus'            => isset($mod_config['menus']) ? $mod_config['menus'] : FALSE,
 417                  );
 418  
 419                  self::$actions[$module]['menu_topic'] = isset($mod_config['menu_topic']) ? $mod_config['menu_topic'] : self::$actions[$module]['display_name'];
 420              }
 421          }
 422  
 423          unset($module_list);
 424  
 425          // Do we have any actions?
 426          if (!count(self::$actions))
 427          {
 428              return '<ul class="'. $class .'"></ul>';
 429          }
 430  
 431          // Order our actions by their weight, then alphabetically
 432          self::sort_actions();
 433  
 434          // Build up our menu array
 435          foreach (self::$actions as $module => $config)
 436          {
 437              if (has_permission('Bonfire.'.ucfirst($module).'.View') || has_permission(ucfirst($module).'.'.ucfirst($context).'.View'))
 438              {
 439                  // Drop-down menus?
 440                  if ($config['menus'] && isset($config['menus'][$context]))
 441                  {
 442                      $menu_view = $config['menus'][$context];
 443                  }
 444                  else
 445                  {
 446                      $menu_view = '';
 447                  }
 448  
 449                  $menu_topic = is_array($config['menu_topic']) && isset($config['menu_topic'][$context]) ? $config['menu_topic'][$context] : $config['display_name'];
 450                  self::$menu[$menu_topic][$module] = array(
 451                          'title'            => $config['title'],
 452                          'display_name'    => $config['display_name'],
 453                          'menu_view'        => $menu_view,
 454                          'menu_topic'    => $menu_topic
 455                  );
 456              }
 457          }
 458  
 459          $menu = self::build_sub_menu($context, $ignore_ul);
 460  
 461          self::$actions = array();
 462  
 463          return $menu;
 464  
 465      }//end context_nav()

 473      /**
 474       * Creates everything needed for a new context to run. Includes
 475       * creating permissions, assigning them to certain roles, and
 476       * even creating an application migration for the permissions.
 477       *
 478       * @access public
 479       * @static
 480       *
 481       * @param    string    $name    The name of the context to create.
 482       * @param    array    $roles    The names or id's of the roles to give permissions to view.
 483       * @param    bool    $migrate    If TRUE, will create an app migration file.
 484       *
 485       * @return     bool
 486       */
 487  	public static function create_context($name='', $roles=array(), $migrate=false)
 488      {
 489          if (empty($name))
 490          {
 491              self::$errors = lang('ui_no_context_name');
 492              return false;
 493          }
 494  
 495          /*
 496              1. Try to write it to the config file so that it
 497                  will show in the menu no matter what.
 498          */
 499          self::$ci->load->helper('config_file');
 500  
 501          $contexts = self::$contexts;
 502  
 503          // If it alread exists, we don't need to do anything!
 504          if (!in_array(strtolower($name), $contexts))
 505          {
 506              array_unshift($contexts, strtolower($name));
 507  
 508              if (!write_config('application', array('contexts' => $contexts), null))
 509              {
 510                  self::$errors[] = lang('ui_cant_write_config');
 511                  return false;
 512              }
 513          }
 514  
 515          /*
 516              2. Create our permissions
 517          */
 518          $cname = 'Site.'. ucfirst($name) .'.View';
 519  
 520          // First - create the actual permission
 521          self::$ci->load->model('permissions/permission_model');
 522  
 523          if (!self::$ci->permission_model->permission_exists($cname))
 524          {
 525              $pid = self::$ci->permission_model->insert(array(
 526                  'name'            => $cname,
 527                  'description'    => 'Allow user to view the '. ucwords($name) .' Context.',
 528              ));
 529          }
 530          else
 531          {
 532              $pid = self::$ci->permission_model->find_by('name', $cname)->permission_id;
 533              $exists = true;
 534  
 535          }
 536  
 537          // Do we have any roles to apply this to?
 538          // If we don't we can quite since there won't be anything
 539          // to migrate.
 540          if (count($roles) == 0)
 541          {
 542              return true;
 543          }
 544  
 545          self::$ci->load->model('roles/role_permission_model');
 546  
 547          foreach ($roles as $role)
 548          {
 549              // Assign By Id
 550              if (is_numeric($role))
 551              {
 552                  self::$ci->role_permission_model->delete_role_permissions($role, $pid);
 553                  self::$ci->role_permission_model->create_role_permissions($role, $pid);
 554              }
 555              // Assign By Name
 556              else
 557              {
 558                  self::$ci->role_permission_model->assign_to_role($role, $cname);
 559              }
 560          }
 561  
 562          // If we made it here, we were successful!
 563          return true;
 564      }

 631                  $list .= '<li class="' . self::$submenu_class . '"><a href="#" >'. ucwords($topic_name) .'</a>';

 637
