 217       * Displays all of the emails currently in the queue to be sent.

 219       * @access public

 223  	public function queue()
 224      {
 225          $this->load->library('pagination');
 226  
 227          $offset = $this->uri->segment(5);
 228  
 229          $this->load->model('Emailer_model', 'emailer_model', TRUE);
 230  
 231          // Deleting anything?
 232          if (isset($_POST['delete']))
 233          {
 234              $checked = $this->input->post('checked');
 235              if (is_array($checked) && count($checked))
 236              {
 237                  $result = FALSE;
 238                  foreach ($checked as $pid)
 239                  {
 240                      $result = $this->emailer_model->delete($pid);
 241                  }
 242  
 243                  if ($result)
 244                  {
 245                      Template::set_message(sprintf(lang('em_delete_success'), count($checked)), 'success');
 246                  }
 247                  else
 248                  {
 249                      Template::set_message(sprintf(lang('em_delete_failure'), $this->emailer_model->error), 'error');
 250                  }
 251              }
 252              else
 253              {
 254                  Template::set_message(sprintf(lang('em_delete_error'), $this->emailer_model->error), 'error');
 255              }
 256          }
 257          elseif (isset($_POST['force_process']))
 258          {
 259              $this->load->library('emailer');
 260              $this->emailer->enable_debug(TRUE);
 261  
 262              // Use ob to catch output designed for CRON only
 263              ob_start();
 264              $success = $this->emailer->process_queue();
 265              ob_end_clean();
 266  
 267              if ( ! $success)
 268              {
 269                  Template::set('email_debug', $this->emailer->debug_message);
 270              }
 271          }
 272          elseif (isset($_POST['insert_test']))
 273          {
 274              $this->load->library('emailer');
 275  
 276              $data = array(
 277                  'to'        => $this->settings_lib->item('site.system_email'),
 278                  'subject'    => lang('em_test_mail_subject'),
 279                  'message'    => lang('em_test_mail_body')
 280              );
 281  
 282              $this->emailer->send($data, TRUE);
 283          }
 284  
 285          Template::set('emails', $this->emailer_model->limit($this->limit, $offset)->find_all());
 286          Template::set('total_in_queue', $this->emailer_model->count_by('date_sent IS NULL'));
 287          Template::set('total_sent', $this->emailer_model->count_by('date_sent IS NOT NULL'));

 334      /**
 335       * Create a new email and send to selected recipents
 336       *
 337       * @access public
 338       *
 339       * @return void
 340       */
 341  	public function create()
 342      {
 343  
 344          $this->load->model('users/user_model');
 345          $this->load->library('emailer');
 346  
 347          if (isset($_POST['create']))
 348          {
 349              // validate subject, content and recipients
 350              $this->form_validation->set_rules('email_subject', 'lang:em_email_subject', 'required|trim|min_length[1]|max_length[255]');
 351              $this->form_validation->set_rules('email_content', 'lang:em_email_content', 'required|trim|min_length[1]');
 352              $this->form_validation->set_rules('checked', 'lang:bf_users', 'required');
 353  
 354              if ($this->form_validation->run() === FALSE)
 355              {
 356                  Template::set('email_subject', $this->input->post('email_subject'));
 357                  Template::set('email_content', $this->input->post('email_content'));
 358                  Template::set('checked', $this->input->post('checked'));
 359              }
 360              else
 361              {
 362                  $data = array (
 363                      'subject'    => $this->input->post('email_subject'),
 364                      'message'    => $this->input->post('email_content'),
 365                  );
 366  
 367                  $checked = $this->input->post('checked');
 368                  $success_count = 0;
 369                  if (is_array($checked) && count($checked))
 370                  {
 371                      $result = FALSE;
 372                      foreach ($checked as $user_id)
 373                      {
 374                          //get the email from user_id
 375                          $user = $this->user_model->find($user_id);
 376                          if ($user != NULL){
 377                              $data['to'] = $user->email;
 378                              $result = $this->emailer->send($data,TRUE);
 379                              if ($result) $success_count++;
 380                          }
 381  
 382                      }
 383  
 384                      if ($result)
 385                      {
 386                          Template::set_message($success_count .' '. lang('em_create_email_success'), 'success');
 387                          redirect(SITE_AREA . '/settings/emailer/queue');
 388                      }
 389                      else
 390                      {
 391                          Template::set_message(sprintf(lang('em_create_email_failure'), $this->user_model->error), 'error');
 392                      }
 393                  }
 394                  else
 395                  {
 396                      Template::set_message(sprintf(lang('em_create_email_error'), $this->user_model->error), 'error');
 397                  }//end if
 398              }//end if
 399          }//end if
 400  
 401          $users = $this->user_model->where('users.deleted', 0)->find_all();
 402          Template::set('users', $users);
 403          Template::set('toolbar_title', lang('em_create_email'));
 404          Template::render();
 405  
 406      }//end create()
