 123       * Attempt to log the user in.
 124       *
 125       * @access public
 126       *
 127       * @param string $login    The user's login credentials (email/username)
 128       * @param string $password The user's password
 129       * @param bool   $remember Whether the user should be remembered in the system.
 130       *
 131       * @return bool
 132       */
 133  	public function login($login, $password, $remember=FALSE)
 134      {
 135          if (empty($login) || empty($password))
 136          {
 137              $error = $this->ci->settings_lib->item('auth.login_type') == 'both' ? lang('bf_username') .'/'. lang('bf_email') : ucfirst($this->ci->settings_lib->item('auth.login_type'));
 138              Template::set_message(sprintf(lang('us_fields_required'), $error), 'error');
 139              return FALSE;
 140          }
 141  
 142          $this->ci->load->model('users/User_model', 'user_model');
 143  
 144          // Grab the user from the db
 145          $selects = 'id, email, username, users.role_id, users.deleted, users.active, banned, ban_message, password_hash, password_iterations, force_password_reset';
 146  
 147          if ($this->ci->settings_lib->item('auth.do_login_redirect'))
 148          {
 149              $selects .= ', login_destination';
 150          }
 151  
 152          if ($this->ci->settings_lib->item('auth.login_type') == 'both')
 153          {
 154              $user = $this->ci->user_model->select($selects)->find_by(array('username' => $login, 'email' => $login), null, 'or');
 155          }
 156          else
 157          {
 158              $user = $this->ci->user_model->select($selects)->find_by($this->ci->settings_lib->item('auth.login_type'), $login);
 159          }
 160  
 161          // check to see if a value of FALSE came back, meaning that the username or email or password doesn't exist.
 162          if ($user == FALSE)
 163          {
 164              Template::set_message(lang('us_bad_email_pass'), 'error');
 165              return FALSE;
 166          }
 167  
 168          // check if the account has been activated.
 169          $activation_type = $this->ci->settings_lib->item('auth.user_activation_method');
 170          if ($user->active == 0 && $activation_type > 0) // in case we go to a unix timestamp later, this will still work.
 171          {
 172              if ($activation_type == 1)
 173              {
 174                  Template::set_message(lang('us_account_not_active'), 'error');
 175              }
 176              elseif ($activation_type == 2)
 177              {
 178                  Template::set_message(lang('us_admin_approval_pending'), 'error');
 179              }
 180  
 181              return FALSE;
 182          }
 183  
 184          // check if the account has been soft deleted.
 185          if ($user->deleted >= 1) // in case we go to a unix timestamp later, this will still work.
 186          {
 187              Template::set_message(sprintf(lang('us_account_deleted'), html_escape(settings_item("site.system_email"))), 'error');
 188              return FALSE;
 189          }
 190  
 191          // Try password
 192          if ($this->check_password($password, $user->password_hash))
 193          {
 194              // check if the account has been banned.
 195              if ($user->banned)
 196              {
 197                  $this->increase_login_attempts($login);
 198                  Template::set_message($user->ban_message ? $user->ban_message : lang('us_banned_msg'), 'error');
 199                  return FALSE;
 200              }
 201  
 202              // Check if the user needs to reset their password
 203              if ($user->force_password_reset == 1)
 204              {
 205                  Template::set_message(lang('us_forced_password_reset_note'), 'warning');
 206  
 207                  // Need to generate a reset hash to pass the reset_password checks...
 208                  $this->ci->load->helpers(array('string', 'security'));
 209  
 210                  $pass_code = random_string('alnum', 40);
 211  
 212                  $hash = do_hash($pass_code . $user->email);
 213  
 214                  // Save the hash to the db so we can confirm it later.
 215                  $this->ci->user_model->update_where('id', $user->id, array('reset_hash' => $hash, 'reset_by' => strtotime("+24 hours") ));
 216  
 217                  $this->ci->session->set_userdata('pass_check', $hash);
 218                  $this->ci->session->set_userdata('email', $user->email);
 219                  redirect('/users/reset_password');
 220              }
 221  
 222              $this->clear_login_attempts($login);
 223  
 224              // We've successfully validated the login, so setup the session
 225              $this->setup_session($user->id, $user->username, $user->password_hash, $user->email, $user->role_id, $remember,'', $user->username);
 226  
 227              // Save the login info
 228              $data = array(
 229                  'last_login'            => date('Y-m-d H:i:s', time()),
 230                  'last_ip'                => $this->ip_address,
 231              );
 232              $this->ci->user_model->update($user->id, $data);
 233  
 234              // Clear the cached result of user() (and hence is_logged_in(), user_id() etc).
 235              // Doesn't fix `$this->current_user` in controller (for this page load)...
 236              unset($this->user);
 237  
 238              $trigger_data = array('user_id'=>$user->id, 'role_id'=>$user->role_id);
 239              Events::trigger('after_login', $trigger_data );
 240  
 241              // Save our redirect location
 242              $this->login_destination = isset($user->login_destination) && !empty($user->login_destination) ? $user->login_destination : '';
 243  
 244              return TRUE;
 245          }
 246  
 247          // Bad password
 248          else
 249          {
 250              Template::set_message(lang('us_bad_email_pass'), 'error');
 251              $this->increase_login_attempts($login);
 252          }
 253  
 254          return FALSE;

 298          {
 299              return $this->user;
 300          }
 301  
 302          $this->user = FALSE;
 303  
 304          // Is there any session data we can use?
 305          if ($this->ci->session->userdata('identity') && $this->ci->session->userdata('user_id'))
 306          {
 307              // Grab the user account
 308              $user = $this->ci->user_model->find($this->ci->session->userdata('user_id'));
 309  
 310              if ($user !== FALSE)
 311              {
 312                  // load do_hash()
 313                  $this->ci->load->helper('security');
 314  
 315                  // Ensure user_token is still equivalent to the SHA1 of the user_id and password_hash
 316                  if (do_hash($this->ci->session->userdata('user_id') . $user->password_hash) === $this->ci->session->userdata('user_token'))
 317                  {
 318                      $this->user = $user;
 319                  }
 320              }
 321          }//end if
 322  
 323          if ($this->user !== FALSE)
 324          {
 325              $this->user->id = (int) $this->user->id;
 326              $this->user->role_id = (int) $this->user->role_id;
 327          }
 328  
 329          return $this->user;

 351      /**
 352       * Checks that a user is logged in (and, optionally of the correct role)
 353       * and, if not, send them to the login screen.
 354       *
 355       * If no permission is checked, will simply verify that the user is logged in.
 356       * If a permission is passed in to the first parameter, will check the user's role
 357       * and verify that role has the appropriate permission.
 358       *
 359       * @access public
 360       *
 361       * @param string $permission (Optional) A string representing the permission to check for.
 362       * @param string $uri        (Optional) A string representing an URI to redirect, if FALSE
 363       *
 364       * @return bool TRUE if the user has the appropriate access permissions. Redirect to the previous page if the user doesn't have permissions. Redirect to LOGIN_AREA page if the user is not logged in.
 365       */
 366  	public function restrict($permission=NULL, $uri=NULL)
 367      {
 368          // If user isn't logged in, don't need to check permissions
 369          if ($this->is_logged_in() === FALSE)
 370          {
 371              $this->ci->load->library('Template');
 372              Template::set_message($this->ci->lang->line('us_must_login'), 'error');
 373              Template::redirect(LOGIN_URL);
 374          }
 375  
 376          // Check to see if the user has the proper permissions
 377          if ( ! empty($permission) && ! $this->has_permission($permission))
 378          {
 379              // set message telling them no permission THEN redirect
 380              Template::set_message( lang('us_no_permission'), 'attention');
 381  
 382              if ( ! $uri)
 383              {
 384                  $uri = $this->ci->session->userdata('previous_page');
 385  
 386                  // If previous page was the same (e.g. user pressed F5),
 387                  // but permission has been removed, then redirecting
 388                  // to it will cause an infinite loop.
 389                  if ($uri == current_url())
 390                  {
 391                      $uri = site_url();
 392                  }
 393              }
 394              Template::redirect($uri);
 395          }
 396  
 397          return TRUE;
 398  
 399      }//end restrict()

 471       * @access public

 473       * @param string $permission A string with the permission to check for, ie 'Site.Signin.Allow'
 474       * @param int    $role_id    The id of the role to check the permission against. If role_id is not passed into the method, then it assumes it to be the current user's role_id.
 475       * @param bool   $override   Whether or not access is granted if this permission doesn't exist in the database
 476       *
 477       * @return bool TRUE/FALSE

 479  	public function has_permission($permission, $role_id=NULL, $override = FALSE)
 480      {
 481          // move permission to lowercase for easier checking.
 482          $permission = strtolower($permission);
 483  
 484          // If no role is being provided, assume it's for the current
 485          // logged in user.
 486          if (empty($role_id))
 487          {
 488              $role_id = $this->role_id();
 489          }
 490  
 491          $this->load_permissions();
 492          $this->load_role_permissions($role_id);
 493  
 494          // did we pass?
 495          if (isset($this->permissions[$permission]))
 496          {
 497              $permission_id = $this->permissions[$permission];
 498  
 499              if (isset($this->role_permissions[$role_id][$permission_id]))
 500              {
 501                  return TRUE;
 502              }
 503          }
 504          elseif ($override)
 505          {
 506              return TRUE;
 507          }
 508  
 509          return FALSE;

 653      /**
 654       * Hash a password
 655       *
 656       * @param String $pass        The password to hash
 657       * @param Int $iterations    The number of iterations used in hashing the password
 658       *
 659       * @return Array            An associative array containing the hashed password and number of iterations
 660       */
 661  	public function hash_password($pass, $iterations=0)
 662      {
 663          // The shortest valid hash phpass can currently return is 20 characters,
 664          // which would only happen with CRYPT_EXT_DES
 665          $min_hash_len = 20;
 666  
 667          if (empty($iterations) || ! is_numeric($iterations) || $iterations <= 0)
 668          {
 669              $iterations = $this->ci->settings_lib->item('password_iterations');
 670          }
 671  
 672          // Load the password hash library
 673          if ( ! class_exists('PasswordHash'))
 674          {
 675              require(dirname(__FILE__) . '/../libraries/PasswordHash.php');
 676          }
 677  
 678          $hasher = new PasswordHash($iterations, false);
 679          $password = $hasher->HashPassword($pass);
 680  
 681          unset($hasher);
 682  
 683          if (strlen($password) < $min_hash_len)
 684          {
 685              return false;
 686          }
 687  
 688          return array('hash' => $password, 'iterations' => $iterations);
 689  
 690      }

 849      /**
 850       * Create the auto-login entry in the database. This method uses
 851       * Charles Miller's thoughts at:
 852       * http://fishbowl.pastiche.org/2004/01/19/persistent_login_cookie_best_practice/
 853       *
 854       * @access private
 855       *
 856       * @param int    $user_id    An int representing the user_id.
 857       * @param string $old_token The previous token that was used to login with.
 858       *
 859       * @return bool Whether the autologin was created or not.
 860       */
 861  	private function create_autologin($user_id, $old_token=NULL)
 862      {
 863          if ($this->ci->settings_lib->item('auth.allow_remember') == FALSE)
 864          {
 865              return FALSE;
 866          }
 867  
 868          // load random_string()
 869          $this->ci->load->helper('string');
 870  
 871          // Generate a random string for our token
 872          $token = random_string('alnum', 128);
 873  
 874          // If an old_token is presented, we're refreshing the autologin information
 875          // otherwise we're creating a new one.
 876          if (empty($old_token))
 877          {
 878              // Create a new token
 879              $data = array(
 880                  'user_id'        => $user_id,
 881                  'token'            => $token,
 882                  'created_on'    => date('Y-m-d H:i:s')
 883              );
 884              $this->ci->db->insert('user_cookies', $data);
 885          }
 886          else
 887          {
 888              // Refresh the token
 889              $this->ci->db->where('user_id', $user_id);
 890              $this->ci->db->where('token', $old_token);
 891              $this->ci->db->set('token', $token);
 892              $this->ci->db->set('created_on', date('Y-m-d H:i:s'));
 893              $this->ci->db->update('user_cookies');
 894          }
 895  
 896          if ($this->ci->db->affected_rows())
 897          {
 898              // Create the autologin cookie
 899              $this->ci->input->set_cookie('autologin', $user_id .'~'. $token, $this->ci->settings_lib->item('auth.remember_length'));
 900  
 901              return TRUE;
 902          }
 903          else
 904          {
 905              return FALSE;
 906          }
 907  
 908      }//end create_autologin()()
