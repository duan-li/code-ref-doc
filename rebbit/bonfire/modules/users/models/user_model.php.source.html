 303

 324          }

 707      {

 719  
 720          foreach ($data as $key => $value)
 721          {
 722              $this->db->where('user_id', $user_id);
 723              $this->db->where('meta_key', $key);
 724              $query = $this->db->get($this->meta_table);
 725  
 726              $obj = array(
 727                  'user_id'        => $user_id,
 728                  'meta_key'        => $key,
 729                  'meta_value'    => $value
 730              );
 731  
 732              if ($query->num_rows() == 0)
 733              {
 734                  // Insert
 735                  $result = $this->db->insert($this->meta_table, $obj);
 736              }
 737              // Update
 738              else if ($query->num_rows() > 0)
 739              {
 740                  $row = $query->row();
 741                  $meta_id = $row->meta_id;
 742  
 743                  $this->db->where('user_id', $user_id);
 744                  $this->db->where('meta_key', $key);
 745                  $this->db->set('meta_value', $value);
 746                  $result = $this->db->update($this->meta_table, $obj);
 747              }//end if
 748          }//end foreach
 749

 863       *

 865       * inactive (Admin Activation scenario), so leave_inactive handles this use case.
 866       *
 867       * @access public

 869       * @param int    $user_id        The user to be activated (NULL will match any)
 870       * @param string $code           The activation code to be verified
 871       * @param bool   $leave_inactive Flag whether to remove the activate hash value, but leave active = 0
 872       *
 873       * @return int User Id on success, FALSE on error
 874       */
 875  	public function activate($user_id, $code, $leave_inactive = FALSE)
 876      {
 877          if ($user_id)
 878          {
 879              $this->db->where('id', $user_id);
 880          }
 881  
 882          $query = $this->db->select('id')
 883                            ->where('activate_hash', $code)

 885  
 886          if ($query->num_rows() !== 1)

 888              $this->error = lang('us_err_no_matching_code');

 915       */
 916  	public function deactivate($user_id, $make_hash = TRUE)
 917      {
 918          // create a temp activation code.
 919          $activate_hash = '';
 920          if ($make_hash === true)
 921          {

 923              $activate_hash = do_hash(random_string('alnum', 40) . time());
 924          }
 925

 927

 930              return FALSE;
