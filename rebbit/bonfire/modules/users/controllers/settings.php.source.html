 489      /**
 490       * Save the user
 491       *
 492       * @access private
 493       *
 494       * @param string $type          The type of operation (insert or edit)
 495       * @param int    $id            The id of the user in the case of an edit operation
 496       * @param array  $meta_fields   Array of meta fields fur the user
 497       * @param string $cur_role_name The current role for the user being edited
 498       *
 499       * @return bool
 500       */
 501  	private function save_user($type='insert', $id=0, $meta_fields=array(), $cur_role_name = '')
 502      {
 503          $this->form_validation->set_rules($this->user_model->get_validation_rules($type));
 504  
 505          $extra_unique_rule = '';
 506          $username_required = '';
 507  
 508          if ($type != 'insert') {
 509              $_POST['id'] = $id;
 510              $extra_unique_rule = ',users.id';
 511          }
 512  
 513          if ($this->settings_lib->item('auth.login_type') == 'username'
 514              || $this->settings_lib->item('auth.use_usernames')
 515             ) {
 516              $username_required = 'required|';
 517          }
 518  
 519          $this->form_validation->set_rules('username', 'lang:bf_username', $username_required . 'trim|max_length[30]|unique[users.username' . $extra_unique_rule . ']');
 520          $this->form_validation->set_rules('email', 'lang:bf_email', 'required|trim|valid_email|max_length[120]|unique[users.email' . $extra_unique_rule . ']');
 521  
 522          if (has_permission('Bonfire.Roles.Manage')
 523              && has_permission('Permissions.' . $cur_role_name . '.Manage')
 524             ) {
 525              $this->form_validation->set_rules('role_id', 'lang:us_role', 'required|trim|max_length[2]|is_numeric');
 526          }
 527  
 528          $meta_data = array();
 529          foreach ($meta_fields as $field) {
 530              if ( ! isset($field['admin_only']) || $field['admin_only'] === false
 531                  || (isset($field['admin_only']) && $field['admin_only'] === true
 532                      && isset($this->current_user) && $this->current_user->role_id == 1
 533                     )
 534                 ) {
 535                  $this->form_validation->set_rules($field['name'], $field['label'], $field['rules']);
 536                  $meta_data[$field['name']] = $this->input->post($field['name']);
 537              }
 538          }
 539  
 540          if ($this->form_validation->run() === false) {
 541              return false;
 542          }
 543  
 544          // Compile our core user elements to save.
 545          $data = $this->user_model->prep_data($this->input->post());
 546  
 547          if ($type == 'insert') {
 548              $activation_method = $this->settings_lib->item('auth.user_activation_method');
 549  
 550              // No activation method
 551              if ($activation_method == 0) {
 552                  // Activate the user automatically
 553                  $data['active'] = 1;
 554              }
 555  
 556              $return = $this->user_model->insert($data);
 557              $id = $return;
 558          } else {    // Update
 559              $return = $this->user_model->update($id, $data);
 560          }
 561  
 562          // Save any meta data for this user
 563          if (count($meta_data)) {
 564              $this->user_model->save_meta_for($id, $meta_data);
 565          }
 566  
 567          // Any modules needing to save data?
 568          Events::trigger('save_user', $this->input->post());
 569  
 570          return $return;
