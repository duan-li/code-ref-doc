   2  /**
   3   * Migrations Library
   4   *
   5   * Migrations provide a simple method to version the contents of your database, and make
   6   * those changes easily distributable to other developers in different server environments.
   7   *
   8   * Migrations are stored in specially-named PHP files under *application/db/migrations/*.
   9   * Each migration file must be numbered consecutively, starting at *001* and growing larger
  10   * with each new migration from there. The rest of the filename should tell what the migration does.
  11   *
  12   * For example: 001_install_initial_schema.php
  13   *
  14   * The class inside of the file must extend the abstract Migration class and implement both the
  15   * up() and down() methods to install and uninstall the tables/changes. The class itself should be
  16   * named:
  17   *
  18   * :    class Migration_install_initial_schema extends Migration {
  19   * :
  20   * :        function up() {}
  21   * :
  22   * :        function down() {}
  23   * :    }
  24   *
  25   * @package    Bonfire
  26   * @author     Mat'as Montes
  27   * @author     Phil Sturgeon http://philsturgeon.co.uk/
  28   * @author     Spicer Matthews <spicer@cloudmanic.com> Cloudmanic Labs, LLC http://www.cloudmanic.com/
  29   * @author     Bonfire Dev Team
  30   */

 202      /**
 203       * Migrate to a specific schema version.
 204       *
 205       * Calls each migration step required to get to the schema version of
 206       * choice.
 207       *
 208       * @access public
 209       *
 210       * @param int    $version An int that is the target version to migrate to.
 211       * @param string $type    A string that represents the name of the module, or 'app_' for application migrations. If empty, it applies core migrations.
 212       *
 213       * @return mixed TRUE if already latest, FALSE if failed, int if upgraded
 214       */
 215  	public function version($version, $type='')
 216      {
 217          $schema_version = $this->get_schema_version($type);
 218          $start = $schema_version;
 219          $stop = $version;
 220  
 221          switch ($type)
 222          {
 223              case '':
 224                  $migrations_path = $this->migrations_path;
 225                  break;
 226              case 'app_':
 227                  $migrations_path = APPPATH .'db/migrations/';
 228                  break;
 229              default:
 230                  $migrations_path = module_path(substr($type, 0, -1), 'migrations') .'/';
 231                  break;
 232          }
 233  
 234          if ($version > $schema_version)
 235          {
 236              // Moving Up
 237              $start++;
 238              $stop++;
 239              $step = 1;
 240          }
 241          else
 242          {
 243              // Moving Down
 244              $step = -1;
 245          }
 246  
 247          $method = $step == 1 ? 'up' : 'down';
 248          $migrations = array();
 249  
 250          // We now prepare to actually DO the migrations
 251  
 252          // But first let's make sure that everything is the way it should be
 253          for($i=$start; $i != $stop; $i += $step)
 254          {
 255              $f = glob(sprintf($migrations_path . '%03d_*'.EXT, $i));
 256              // Only one migration per step is permitted
 257              if (count($f) > 1)
 258              {
 259                  $this->error = sprintf($this->_ci->lang->line("multiple_migrations_version"),$i);
 260                  return 0;
 261              }
 262  
 263              // Migration step not found
 264              if (count($f) == 0)
 265              {
 266                  // If trying to migrate up to a version greater than the last
 267                  // existing one, migrate to the last one.
 268                  if ($step == 1)
 269                  {
 270                      break;
 271                  }
 272  
 273                  // If trying to migrate down but we're missing a step,
 274                  // something must definitely be wrong.
 275                  $this->error = sprintf($this->_ci->lang->line("migration_not_found"),$i);
 276                  return 0;
 277              }
 278  
 279              $file = basename($f[0]);
 280              $name = basename($f[0],EXT);
 281  
 282              // Filename validations
 283              if (preg_match('/^\d{3}_(\w+)$/', $name, $match))
 284              {
 285                  $match[1] = strtolower($match[1]);
 286  
 287                  // Cannot repeat a migration at different steps
 288                  if (in_array($match[1], $migrations))
 289                  {
 290                      $this->error = sprintf($this->_ci->lang->line("multiple_migrations_name"),$match[1]);
 291                      return 0;
 292                  }
 293  
 294                  include $f[0];
 295                  $class = 'Migration_'.ucfirst($match[1]);
 296  
 297                  if ( ! class_exists($class))
 298                  {
 299                      $this->error = sprintf($this->_ci->lang->line("migration_class_doesnt_exist"), $class);
 300                      return 0;
 301                  }
 302  
 303                  if ( ! is_callable(array($class,"up")) || ! is_callable(array($class,"down"))) {
 304                      $this->error = sprintf($this->_ci->lang->line('wrong_migration_interface'),$class);
 305                      return 0;
 306                  }
 307  
 308                  $migrations[] = $match[1];
 309              }
 310              else
 311              {
 312                  $this->error = sprintf($this->_ci->lang->line("invalid_migration_filename"),$file, $migrations_path);
 313                  return 0;
 314              }//end if
 315          }//end for
 316  
 317          $version = $i + ($step == 1 ? -1 : 0);
 318  
 319          // If there is nothing to do, bitch and quit
 320          if ($migrations === array())
 321          {
 322              if ($this->verbose)
 323              {
 324                  echo "Nothing to do, bye!\n";
 325              }
 326  
 327              return TRUE;
 328          }
 329  
 330  
 331          if ($this->verbose)
 332          {
 333              echo "<p>Current schema version: ".$schema_version."<br/>";
 334              echo "Moving ".$method." to version ".$version."</p>";
 335              echo "<hr/>";
 336          }
 337  
 338          // Loop through the migrations
 339          foreach($migrations AS $m)
 340          {
 341              if ($this->verbose)
 342              {
 343                  echo "$m:<br />";
 344                  echo "<blockquote>";
 345              }
 346  
 347              $class = 'Migration_'.ucfirst($m);
 348  
 349              $c = new $class;
 350  
 351              if ($c->migration_type == 'forge')
 352              {
 353                  call_user_func(array($c, $method));
 354              }
 355              else if ($c->migration_type == 'sql')
 356              {
 357                  $sql = $c->$method();
 358                  $this->do_sql_migration($sql);
 359              }
 360  
 361              if ($this->verbose)
 362              {
 363                  echo "</blockquote>";
 364                  echo "<hr/>";
 365              }
 366  
 367              $schema_version += $step;
 368              $this->_update_schema_version($schema_version, $type);
 369          }//end foreach
 370  
 371          if ($this->verbose)
 372          {
 373              echo "<p>All done. Schema is at version $schema_version.</p>";
 374          }
 375  
 376          return $schema_version;
 377  
 378      }//end version()

 460      /**
 461       * Search the migrations folder and return a list of available migration files.
 462       *
 463       * @access public
 464       *
 465       * @param string $type A string that represents the name of the module, or 'app_' for application migrations. If empty, it returns core migrations.
 466       *
 467       * @return array An array of migration files
 468       */
 469  	public function get_available_versions($type='')
 470      {
 471          switch ($type)
 472          {
 473              case '':
 474                  $migrations_path = $this->migrations_path;
 475                  break;
 476              case 'app_':
 477                  $migrations_path = APPPATH .'db/migrations/';
 478                  break;
 479              default:
 480                  $migrations_path = module_path(substr($type, 0, -1), 'migrations') .'/';
 481                  break;
 482          }
 483  
 484          // List all *_*.php files in the migrations path
 485          $files = glob($migrations_path .'*_*'.EXT);
 486  
 487          for ($i=0; $i < count($files); $i++)
 488          {
 489              // Remove path and extension
 490              $files[$i] = basename($files[$i],EXT);
 491  
 492              // Mark wrongly formatted files as FALSE for later filtering
 493              if(!preg_match('/^\d{3}_(\w+)$/',$files[$i]))
 494              {
 495                  $files[$i] = FALSE;
 496              }
 497          }
 498  
 499          $migrations = array_filter($files);
 500          sort($migrations);
 501  
 502          return $migrations;
 503  
 504      }//end get_available_versions()

 508      /**
 509       * Execute raw SQL migrations. Will manually break the commands on a ';' so
 510       * that multiple commmands can be run at once. Very handy for using phpMyAdmin
 511       * dumps.
 512       *
 513       * @access public
 514       *
 515       * @param string $sql A string with one or more SQL commands to be run.
 516       *
 517       * @return void
 518       */
 519  	public function do_sql_migration($sql='')
 520      {
 521          if (empty($sql))
 522          {
 523              return;
 524          }
 525  
 526          // Split the sql into usable commands on ';'
 527          $queries = explode(';', $sql);
 528  
 529          foreach ($queries as $q)
 530          {
 531              if (trim($q))
 532              {
 533                  $this->_ci->db->query(trim($q));
 534              }
 535          }
 536  
 537      }//end do_sql_migration()

 550  	public function get_schema_version($type='')
 551      {
 552          if ($this->_check_migrations_column('type'))
 553          {
 554              // new schema table layout
 555              $type = empty($type) ? 'core' : $type;
 556              $row = $this->_ci->db->get_where('schema_version', array('type' => $type))->row();
 557              return isset($row->version) ? $row->version: 0;
 558          }
 559          else
 560          {
 561              $row = $this->_ci->db->get('schema_version')->row();
 562  
 563              $schema = $type .'version';
 564  
 565              return isset($row->$schema) ? $row->$schema : 0;
 566          }
 567  
 568      }//end latest()

 572      /**
 573       * Store the current schema version in the database.
 574       *
 575       * @access private
 576       *
 577       * @param int    $schema_version An integer with the latest Schema version reached
 578       * @param string $type           A string that is appended with '_schema' to create the field name to store in the database.
 579       *
 580       * @return void
 581       */
 582  	private function _update_schema_version($schema_version, $type='')
 583      {
 584          logit('[Migrations] Schema '. $type.' updated to: '. $schema_version);
 585  
 586          if ($this->_check_migrations_column('type'))
 587          {
 588              // new schema table layout
 589              $type = empty($type) ? 'core' : $type;
 590              // If the row doesn't exist, create it...
 591              $query = $this->_ci->db->get_where('schema_version', array('type' => $type));
 592  
 593              if ($schema_version != 0)
 594              {
 595                  if (!$query->num_rows())
 596                  {
 597                      $this->_ci->db->insert('schema_version', array(
 598                          'type'        => $type,
 599                          'version' => $schema_version,
 600                      ));
 601  
 602                  }
 603  
 604                  return $this->_ci->db->update('schema_version', array('version' => $schema_version), array('type' => $type));
 605              }
 606              elseif ($query->num_rows())
 607              {
 608                  return $this->_ci->db->delete('schema_version', array('type' => $type));
 609              }
 610          }
 611          else
 612          {
 613              // If the row doesn't exist, create it...
 614              if (!$this->_check_migrations_column($type .'version'))
 615              {
 616                  $this->_ci->load->dbforge();
 617  
 618                  $this->_ci->dbforge->add_column('schema_version', array(
 619                      $type .'version'    => array(
 620                          'type'            => 'INT',
 621                          'constraint'    => 4,
 622                          'null'            => true,
 623                          'default'        => 0
 624                      )
 625                  ));
 626  
 627              }
 628  
 629              return $this->_ci->db->update('schema_version', array(
 630                  $type.'version' => $schema_version
 631              ));
 632  
 633          }//end if
 634  
 635      }//end _update_schema_version()
