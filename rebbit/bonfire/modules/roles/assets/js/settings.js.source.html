   4  $('#permission_table input:checkbox').change(function() {
   5      $('#permission_table_result').removeClass();
   6      $('#permission_table_result').addClass('alert');
   7      $('#permission_table_result').addClass('alert-info');
   8      var val = $(this).attr('value');
   9      var what = $(this).is(':checked');
  10      var r_name = $(this).closest('td').attr('title');
  11      var p_name = $(this).closest('tr').attr('title');
  12      $.ajax({
  13              url: window.g_url +"/",
  14              type: 'POST',
  15              data: {
  16                  "role_perm": val,
  17                  "action": what,
  18                  "ci_csrf_token": ci_csrf_token()
  19              },
  20              success: function(data) {
  21                  var newtext = data.replace(/g_permission/i, '"'+p_name+'"');
  22                  newtext = newtext.replace(/g_role/i, '"'+r_name+'" role');
  23                  if (newtext.search(':') >= 1) {
  24                      $('#permission_table_result').removeClass('alert-info');
  25                      $('#permission_table_result').addClass('alert-warning');
  26                  } else {
  27                      $('#permission_table_result').removeClass('alert-info');
  28                      $('#permission_table_result').addClass('alert-success');
  29                  }
  30                  $('#permission_table_result').text(newtext);
  31              },
  32              error: function (jqXHR, textStatus, errorThrown) {
  33                  // This callback was added as part of an urgent fix.
  34                  // I expect some better solutions are possible.
  35  
  36                  // E.g. this type of error can occur if you click a checkbox
  37                  // then immediately navigate to another page - before the XHR
  38                  // had time to complete.
  39                  // In general, I get the impression we didn't want _A_JAX;
  40                  // we wanted to use a fake-synchronous XHR technique
  41                  // (real synchronous XHR can have the disadvantage of hanging
  42                  // the entire browser).
  43  
  44                  // This could be a network timeout.   So it's possible that
  45                  // the user toggled _several_ permissions in that time,
  46                  // and none of them have been applied.  The state of the checkboxes
  47                  // would be nicely misleading now.  We need to warn about this.
  48                  alert('An error occurred.  Some or all of your changes have failed.');
  49  
  50                  // Refresh the page.
  51                  // This should either get us back in sync, or bring up an error message.
  52  
  53                  // Make sure we kill the page *immediately*, so the user doesn't
  54                  // continue interacting with a doomed page.
  55                  // We can't rely on them noticing that we've started to reload the page.
  56                  // E.g. modern Firefox has de-emphasized its loading indicators to
  57                  // near-invisibility.
  58                  $('#permission_table').html('');
  59  
  60                  // Use location.replace() to break the back button.
  61                  // Not completely essential, but - I can imagine
  62                  // seeing an error page, then hitting back and trying again.
  63                  // We don't want people to do that, because we know
  64                  // at _least_ one of the checkboxes is showing a wrong value.
  65                  window.location.replace(window.location.href);
  66              }
  67          });
  68  });
