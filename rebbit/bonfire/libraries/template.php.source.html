 354       * A block is a partial view contained in a view file in the
 355       * application/views folder. It can be used for sidebars,
 356       * headers, footers, or any other recurring element within
 357       * a site. It is recommended to set a default when calling
 358       * this function within a layout. The default will be rendered
 359       * if no methods override the view (using the set_block() method).
 360       *
 361       * @access public
 362       * @static
 363       *
 364       * @param string $block_name   The name of the block to render.
 365       * @param string $default_view The view to render if no other view has been set with the set_block() method.
 366       * @param array  $data         An array of data to pass to the view.
 367       * @param bool   $themed       Whether we should look in the themes or standard view locations.
 368       *
 369       * @return void
 370       */
 371  	public static function block($block_name='', $default_view='', $data=array(), $themed=FALSE)
 372      {
 373          if (empty($block_name))
 374          {
 375              self::debug_message('No block name provided.');
 376              return;
 377          }
 378  
 379          // If a block has been set previously use it; otherwise, use the default view.
 380          $block_view_name = isset(self::$blocks[$block_name]) ? self::$blocks[$block_name] : $default_view;
 381  
 382          if (empty($block_view_name) && empty($default_view))
 383          {
 384              self::debug_message('No default block provided for `' . $block_name . '`');
 385              return;
 386          }
 387  
 388          self::debug_message("Looking for block: <b>{$block_view_name}</b>.");
 389  
 390          self::load_view($block_view_name, $data, FALSE, $themed, $output);
 391  
 392          $block_data = array('block' => $block_view_name, 'output' => $output);
 393          Events::trigger('after_block_render', $block_data);
 394  
 395          echo $output;
 396  
 397      }//end block()

 406       * Theme paths allow you to have multiple locations for themes to be
 407       * stored. This might be used for separating themes for different sub-
 408       * applications, or a core theme and user-submitted themes.

 413       * @param string $path A new path where themes can be found.
 414       *
 415       * @return bool
 416       */
 417  	public static function add_theme_path($path=NULL)
 418      {
 419          if (empty($path) || ! is_string($path))
 420          {
 421              return FALSE;
 422          }
 423  
 424          // Make sure the path has a '/' at the end.
 425          if (substr($path, -1) != '/')
 426          {
 427              $path .= '/';
 428          }
 429  
 430          // If the path already exists, we're done here.
 431          if (isset(self::$theme_paths[$path]))
 432          {
 433              return TRUE;
 434          }
 435  
 436          // Make sure the folder actually exists
 437          if (is_dir(FCPATH . $path))
 438          {
 439              array_push(self::$theme_paths, $path);
 440              return FALSE;
 441          }
 442          else
 443          {
 444              self::debug_message("Cannot add theme folder: $path does not exist");
 445              return FALSE;
 446          }
 447  
 448      }//end add_theme_path()

 736       * @param string $message A string to be the message. (Optional) If included, will override any other messages in the system.

 738       *
 739       * @return string A string with the results of inserting the message into the message template.
 740       */
 741  	public static function message($message='', $type='information')
 742      {
 743          // Does session data exist?
 744          if (empty($message) && class_exists('CI_Session'))
 745          {
 746              $message = self::$ci->session->flashdata('message');
 747  
 748              if ( ! empty($message))
 749              {
 750                  // Split out our message parts
 751                  $temp_message = explode('::', $message);
 752                  $type = $temp_message[0];
 753                  $message = $temp_message[1];
 754  
 755                  unset($temp_message);
 756              }
 757          }//end if
 758  
 759          // If message is empty, we need to check our own storage.
 760          if (empty($message))
 761          {
 762              if (empty(self::$message['message']))
 763              {
 764                  return '';
 765              }
 766  
 767              $message = self::$message['message'];
 768              $type = self::$message['type'];
 769          }
 770  
 771          // Grab out message template and replace the placeholders
 772          $template = str_replace('{type}', $type, self::$ci->config->item('template.message_template'));
 773          $template = str_replace('{message}', $message, $template);
 774  
 775          // Clear our session data so we don't get extra messages.
 776          // (This was a very rare occurence, but clearing should resolve the problem.
 777          if (class_exists('CI_Session') && ! self::$ignore_session)
 778          {
 779              self::$ci->session->set_flashdata('message', '');
 780          }
 781  
 782          return $template;

 923      /**
 924       * Searches through the the active theme and the default theme to try to find
 925       * a view file. If found, it returns the rendered view.
 926       *
 927       * @access private
 928       *
 929       * @param string $view The name of the view to find.
 930       * @param array  $data An array of key/value pairs to pass to the views.
 931       *
 932       * @return string The content of the file, if found, else empty.
 933       */
 934  	private static function find_file($view=NULL, $data=NULL)
 935      {
 936          if (empty($view))
 937          {
 938              return FALSE;
 939          }
 940  
 941          if ( ! empty($data))
 942          {
 943              $data = (array)$data;
 944          }
 945  
 946          $output = '';        // Stores the final output
 947          $view_path = '';    // Used to store the location of the file.
 948          $active_theme_set = ! empty(self::$active_theme);    // Is the active theme set?
 949          $view_file = $view . '.php';    // filename for the view
 950  
 951          /*
 952           * In most cases, self::$theme_paths will only include one location, but when it
 953           * does not, the last will take precedence for the search. Previously the loop
 954           * just replaced the $view_path with any files found in later paths. Instead,
 955           * we will just reverse the $theme_paths array and break the loop when we find
 956           * the file.
 957           */
 958          $theme_locations = array_reverse(self::$theme_paths);
 959  
 960          // If there are multiple theme locations, we need to search through all of them.
 961          foreach ($theme_locations as $path)
 962          {
 963              $site_theme_path = self::$site_path . $path . '/';
 964  
 965              /*
 966                  First, check the active theme
 967              */
 968              $active_theme_path = $site_theme_path . self::$active_theme;
 969              self::debug_message('[Find File] Looking for view in active theme: <b>' . $active_theme_path . $view_file . '</b><br/>');
 970  
 971              if ($active_theme_set && is_file($active_theme_path . $view_file))
 972              {
 973                  $view_path = $active_theme_path;
 974                  self::debug_message('Found <b>' . $view . '</b> in Active Theme.<br/>');
 975  
 976                  // If we found the view, we should exit the loop
 977                  break;
 978              }
 979  
 980              /*
 981                  If not in the active theme, try the default theme.
 982                  As long as we break the loop whenever the $view_path is set,
 983                  we should not need to check empty($view_path) here
 984              */
 985              $default_theme_path = $site_theme_path . self::$default_theme;
 986              self::debug_message('[Find File] Looking for view in default theme: <b>' . $default_theme_path . $view_file . '</b><br/>');
 987  
 988              if (is_file($default_theme_path . $view_file))
 989              {
 990                  $view_path = $default_theme_path;
 991                  self::debug_message('Found <b>' . $view . '</b> in Default Theme.<br/>');
 992  
 993                  // If we found the view, we should exit the loop
 994                  break;
 995              }
 996          }
 997  
 998          // If the view was found, its path is stored in the $view_path var. So parse or render it
 999          // based on user settings.
1000          if ( ! empty($view_path))
1001          {
1002              $view_path = str_replace('//', '/', $view_path);
1003              self::debug_message('[Find File] Rendering file at: '. $view_path . $view_file .'<br/><br/>');
1004  
1005              // Grab the output of the view.
1006              if (self::$parse_views === TRUE)
1007              {
1008                  $data = array_merge((array)$data, self::$ci->load->_ci_cached_vars);
1009                  $output = self::$ci->load->_ci_load(array(
1010                      '_ci_path' => $view_path . $view_file,
1011                      '_ci_vars' => $data,
1012                      '_ci_return' => TRUE,
1013                  ));
1014              }
1015              else
1016              {
1017                  $output = self::$ci->load->_ci_load(array(
1018                      '_ci_path' => $view_path . $view_file,
1019                      '_ci_vars' => $data,
1020                      '_ci_return' => TRUE,
1021                  ));
1022              }
1023          }//end if
1024  
1025          return $output;
1026  
1027      }//end find_file()
